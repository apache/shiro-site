050/**
051 * Commandline line utility to hash data such as strings, passwords, resources (files, urls, etc).
052 * <p/>
053 * Usage:
054 * <pre>
055 * java -jar shiro-tools-hasher<em>-version</em>-cli.jar
056 * </pre>
057 * This will print out all supported options with documentation.
058 *
059 * @since 1.2
060 */
061public final class Hasher {
062
063    private static final String HEX_PREFIX = "0x";
064    private static final String DEFAULT_ALGORITHM_NAME = "MD5";
065    private static final String DEFAULT_PASSWORD_ALGORITHM_NAME = DefaultPasswordService.DEFAULT_HASH_ALGORITHM;
066    private static final int DEFAULT_GENERATED_SALT_SIZE = 128;
067    private static final int DEFAULT_NUM_ITERATIONS = 1;
068    private static final int DEFAULT_PASSWORD_NUM_ITERATIONS = DefaultPasswordService.DEFAULT_HASH_ITERATIONS;
069
070    private static final Option ALGORITHM = new Option("a", "algorithm", true, "hash algorithm name.  Defaults to SHA-256 when password hashing, MD5 otherwise.");
071    private static final Option DEBUG = new Option("d", "debug", false, "show additional error (stack trace) information.");
072    private static final Option FORMAT = new Option("f", "format", true, "hash output format.  Defaults to 'shiro1' when password hashing, 'hex' otherwise.  See below for more information.");
073    private static final Option HELP = new Option("help", "help", false, "show this help message.");
074    private static final Option ITERATIONS = new Option("i", "iterations", true, "number of hash iterations.  Defaults to " + DEFAULT_PASSWORD_NUM_ITERATIONS + " when password hashing, 1 otherwise.");
075    private static final Option PASSWORD = new Option("p", "password", false, "hash a password (disable typing echo)");
076    private static final Option PASSWORD_NC = new Option("pnc", "pnoconfirm", false, "hash a password (disable typing echo) but disable password confirmation prompt.");
077    private static final Option RESOURCE = new Option("r", "resource", false, "read and hash the resource located at <value>.  See below for more information.");
078    private static final Option SALT = new Option("s", "salt", true, "use the specified salt.  <arg> is plaintext.");
079    private static final Option SALT_BYTES = new Option("sb", "saltbytes", true, "use the specified salt bytes.  <arg> is hex or base64 encoded text.");
080    private static final Option SALT_GEN = new Option("gs", "gensalt", false, "generate and use a random salt. Defaults to true when password hashing, false otherwise.");
081    private static final Option NO_SALT_GEN = new Option("ngs", "nogensalt", false, "do NOT generate and use a random salt (valid during password hashing).");
082    private static final Option SALT_GEN_SIZE = new Option("gss", "gensaltsize", true, "the number of salt bits (not bytes!) to generate.  Defaults to 128.");
083    private static final Option PRIVATE_SALT = new Option("ps", "privatesalt", true, "use the specified private salt.  <arg> is plaintext.");
084    private static final Option PRIVATE_SALT_BYTES = new Option("psb", "privatesaltbytes", true, "use the specified private salt bytes.  <arg> is hex or base64 encoded text.");
085
086    private static final String SALT_MUTEX_MSG = createMutexMessage(SALT, SALT_BYTES);
087
088    private static final HashFormatFactory HASH_FORMAT_FACTORY = new DefaultHashFormatFactory();
089
090    static {
091        ALGORITHM.setArgName("name");
092        SALT_GEN_SIZE.setArgName("numBits");
093        ITERATIONS.setArgName("num");
094        SALT.setArgName("sval");
095        SALT_BYTES.setArgName("encTxt");
096    }

098    public static void main(String[] args) {
099
100        CommandLineParser parser = new DefaultParser();
101
102        Options options = new Options();
103        options.addOption(HELP).addOption(DEBUG).addOption(ALGORITHM).addOption(ITERATIONS);
104        options.addOption(RESOURCE).addOption(PASSWORD).addOption(PASSWORD_NC);
105        options.addOption(SALT).addOption(SALT_BYTES).addOption(SALT_GEN).addOption(SALT_GEN_SIZE).addOption(NO_SALT_GEN);
106        options.addOption(PRIVATE_SALT).addOption(PRIVATE_SALT_BYTES);
107        options.addOption(FORMAT);
108
109        boolean debug = false;
110        String algorithm = null; //user unspecified
111        int iterations = 0; //0 means unspecified by the end-user
112        boolean resource = false;
113        boolean password = false;
114        boolean passwordConfirm = true;
115        String saltString = null;
116        String saltBytesString = null;
117        boolean generateSalt = false;
118        int generatedSaltSize = DEFAULT_GENERATED_SALT_SIZE;
119        String privateSaltString = null;
120        String privateSaltBytesString = null;
121
122        String formatString = null;
123
124        char[] passwordChars = null;
125
126        try {
127            CommandLine line = parser.parse(options, args);
128
129            if (line.hasOption(HELP.getOpt())) {
130                printHelpAndExit(options, null, debug, 0);
131            }
132            if (line.hasOption(DEBUG.getOpt())) {
133                debug = true;
134            }
135            if (line.hasOption(ALGORITHM.getOpt())) {
136                algorithm = line.getOptionValue(ALGORITHM.getOpt());
137            }
138            if (line.hasOption(ITERATIONS.getOpt())) {
139                iterations = getRequiredPositiveInt(line, ITERATIONS);
140            }
141            if (line.hasOption(PASSWORD.getOpt())) {
142                password = true;
143                generateSalt = true;
144            }
145            if (line.hasOption(RESOURCE.getOpt())) {
146                resource = true;
147            }
148            if (line.hasOption(PASSWORD_NC.getOpt())) {
149                password = true;
150                generateSalt = true;
151                passwordConfirm = false;
152            }
153            if (line.hasOption(SALT.getOpt())) {
154                saltString = line.getOptionValue(SALT.getOpt());
155            }
156            if (line.hasOption(SALT_BYTES.getOpt())) {
157                saltBytesString = line.getOptionValue(SALT_BYTES.getOpt());
158            }
159            if (line.hasOption(NO_SALT_GEN.getOpt())) {
160                generateSalt = false;
161            }
162            if (line.hasOption(SALT_GEN.getOpt())) {
163                generateSalt = true;
164            }
165            if (line.hasOption(SALT_GEN_SIZE.getOpt())) {
166                generateSalt = true;
167                generatedSaltSize = getRequiredPositiveInt(line, SALT_GEN_SIZE);
168                if (generatedSaltSize % 8 != 0) {
169                    throw new IllegalArgumentException("Generated salt size must be a multiple of 8 (e.g. 128, 192, 256, 512, etc).");
170                }
171            }
172            if (line.hasOption(PRIVATE_SALT.getOpt())) {
173                privateSaltString = line.getOptionValue(PRIVATE_SALT.getOpt());
174            }
175            if (line.hasOption(PRIVATE_SALT_BYTES.getOpt())) {
176                privateSaltBytesString = line.getOptionValue(PRIVATE_SALT_BYTES.getOpt());
177            }
178            if (line.hasOption(FORMAT.getOpt())) {
179                formatString = line.getOptionValue(FORMAT.getOpt());
180            }
181
182            String sourceValue;
183
184            Object source;
185
186            if (password) {
187                passwordChars = readPassword(passwordConfirm);
188                source = passwordChars;
189            } else {
190                String[] remainingArgs = line.getArgs();
191                if (remainingArgs == null || remainingArgs.length != 1) {
192                    printHelpAndExit(options, null, debug, -1);
193                }
194
195                assert remainingArgs != null;
196                sourceValue = toString(remainingArgs);
197
198                if (resource) {
199                    if (!ResourceUtils.hasResourcePrefix(sourceValue)) {
200                        source = toFile(sourceValue);
201                    } else {
202                        source = ResourceUtils.getInputStreamForPath(sourceValue);
203                    }
204                } else {
205                    source = sourceValue;
206                }
207            }
208
209            if (algorithm == null) {
210                if (password) {
211                    algorithm = DEFAULT_PASSWORD_ALGORITHM_NAME;
212                } else {
213                    algorithm = DEFAULT_ALGORITHM_NAME;
214                }
215            }
216
217            if (iterations < DEFAULT_NUM_ITERATIONS) {
218                //Iterations were not specified.  Default to 350,000 when password hashing, and 1 for everything else:
219                if (password) {
220                    iterations = DEFAULT_PASSWORD_NUM_ITERATIONS;
221                } else {
222                    iterations = DEFAULT_NUM_ITERATIONS;
223                }
224            }
225
226            ByteSource publicSalt = getSalt(saltString, saltBytesString, generateSalt, generatedSaltSize);
227            ByteSource privateSalt = getSalt(privateSaltString, privateSaltBytesString, false, generatedSaltSize);
228            HashRequest hashRequest = new SimpleHashRequest(algorithm, ByteSource.Util.bytes(source), publicSalt, iterations);
229
230            DefaultHashService hashService = new DefaultHashService();
231            hashService.setPrivateSalt(privateSalt);
232            Hash hash = hashService.computeHash(hashRequest);
233
234            if (formatString == null) {
235                //Output format was not specified.  Default to 'shiro1' when password hashing, and 'hex' for
236                //everything else:
237                if (password) {
238                    formatString = Shiro1CryptFormat.class.getName();
239                } else {
240                    formatString = HexFormat.class.getName();
241                }
242            }
243
244            HashFormat format = HASH_FORMAT_FACTORY.getInstance(formatString);
245
246            if (format == null) {
247                throw new IllegalArgumentException("Unrecognized hash format '" + formatString + "'.");
248            }
249
250            String output = format.format(hash);
251
252            System.out.println(output);
253
254        } catch (IllegalArgumentException iae) {
255            exit(iae, debug);
256        } catch (UnknownAlgorithmException uae) {
257            exit(uae, debug);
258        } catch (IOException ioe) {
259            exit(ioe, debug);
260        } catch (Exception e) {
261            printHelpAndExit(options, e, debug, -1);
262        } finally {
263            if (passwordChars != null && passwordChars.length > 0) {
264                for (int i = 0; i < passwordChars.length; i++) {
265                    passwordChars[i] = ' ';
266                }
267            }
268        }
269    }

271    private static String createMutexMessage(Option... options) {
272        StringBuilder sb = new StringBuilder();
273        sb.append("The ");
274
275        for (int i = 0; i < options.length; i++) {
276            if (i > 0) {
277                sb.append(", ");
278            }
279            Option o = options[0];
280            sb.append("-").append(o.getOpt()).append("/--").append(o.getLongOpt());
281        }
282        sb.append(" and generated salt options are mutually exclusive.  Only one of them may be used at a time");
283        return sb.toString();
284    }

286    private static void exit(Exception e, boolean debug) {
287        printException(e, debug);
288        System.exit(-1);
289    }

291    private static int getRequiredPositiveInt(CommandLine line, Option option) {
292        String iterVal = line.getOptionValue(option.getOpt());
293        try {
294            return Integer.parseInt(iterVal);
295        } catch (NumberFormatException e) {
296            String msg = "'" + option.getLongOpt() + "' value must be a positive integer.";
297            throw new IllegalArgumentException(msg, e);
298        }
299    }

301    private static ByteSource getSalt(String saltString, String saltBytesString, boolean generateSalt, int generatedSaltSize) {
302
303        if (saltString != null) {
304            if (generateSalt || (saltBytesString != null)) {
305                throw new IllegalArgumentException(SALT_MUTEX_MSG);
306            }
307            return ByteSource.Util.bytes(saltString);
308        }
309
310        if (saltBytesString != null) {
311            if (generateSalt) {
312                throw new IllegalArgumentException(SALT_MUTEX_MSG);
313            }
314
315            String value = saltBytesString;
316            boolean base64 = true;
317            if (saltBytesString.startsWith(HEX_PREFIX)) {
318                //hex:
319                base64 = false;
320                value = value.substring(HEX_PREFIX.length());
321            }
322            byte[] bytes;
323            if (base64) {
324                bytes = Base64.decode(value);
325            } else {
326                bytes = Hex.decode(value);
327            }
328            return ByteSource.Util.bytes(bytes);
329        }
330
331        if (generateSalt) {
332            SecureRandomNumberGenerator generator = new SecureRandomNumberGenerator();
333            int byteSize = generatedSaltSize / 8; //generatedSaltSize is in *bits* - convert to byte size:
334            return generator.nextBytes(byteSize);
335        }
336
337        //no salt used:
338        return null;
339    }

341    private static void printException(Exception e, boolean debug) {
342        if (e != null) {
343            System.out.println();
344            if (debug) {
345                System.out.println("Error: ");
346                e.printStackTrace(System.out);
347                System.out.println(e.getMessage());
348
349            } else {
350                System.out.println("Error: " + e.getMessage());
351                System.out.println();
352                System.out.println("Specify -d or --debug for more information.");
353            }
354        }
355    }

445    private static char[] readPassword(boolean confirm) {
446        if (!JavaEnvironment.isAtLeastVersion16()) {
447            String msg = "Password hashing (prompt without echo) uses the java.io.Console to read passwords " +
448                    "safely.  This is only available on Java 1.6 platforms and later.";
449            throw new IllegalArgumentException(msg);
450        }
451        java.io.Console console = System.console();
452        if (console == null) {
453            throw new IllegalStateException("java.io.Console is not available on the current JVM.  Cannot read passwords.");
454        }
455        char[] first = console.readPassword("%s", "Password to hash: ");
456        if (first == null || first.length == 0) {
457            throw new IllegalArgumentException("No password specified.");
458        }
459        if (confirm) {
460            char[] second = console.readPassword("%s", "Password to hash (confirm): ");
461            if (!Arrays.equals(first, second)) {
462                String msg = "Password entries do not match.";
463                throw new IllegalArgumentException(msg);
464            }
465        }
466        return first;
467    }

469    private static File toFile(String path) {
470        String resolved = path;
471        if (path.startsWith("~/") || path.startsWith(("~\\"))) {
472            resolved = path.replaceFirst("\\~", System.getProperty("user.home"));
473        }
474        return new File(resolved);
475    }

477    private static String toString(String[] strings) {
478        int len = strings != null ? strings.length : 0;
479        if (len == 0) {
480            return null;
481        }
482        return StringUtils.toDelimitedString(strings, " ");
483    }
