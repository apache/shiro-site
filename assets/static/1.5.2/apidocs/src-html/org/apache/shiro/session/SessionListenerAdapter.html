019package org.apache.shiro.session;
020
021/**
022 * Simple adapter implementation of the {@link SessionListener} interface, effectively providing
023 * no-op implementations of all methods.
024 *
025 * @since 1.0
026 */
027public class SessionListenerAdapter implements SessionListener {
028
029    /**
030     * Adapter no-op implementation - does nothing and returns immediately.
031     *
032     * @param session the session that has started.
033     */
034    public void onStart(Session session) {
035        //no-op
036    }
037
038    /**
039     * Adapter no-op implementation - does nothing and returns immediately.
040     *
041     * @param session the session that has stopped.
042     */
043    public void onStop(Session session) {
044        //no-op
045    }
046
047    /**
048     * Adapter no-op implementation - does nothing and returns immediately.
049     *
050     * @param session the session that has expired.
051     */
052    public void onExpiration(Session session) {
053        //no-op
054    }
055}
