033/**
034 * Requires the requesting user to be {@link org.apache.shiro.subject.Subject#isAuthenticated() authenticated} for the
035 * request to continue, and if they're not, requires the user to login via the HTTP Basic protocol-specific challenge.
036 * Upon successful login, they're allowed to continue on to the requested resource/url.
037 * <p/>
038 * This implementation is a 'clean room' Java implementation of Basic HTTP Authentication specification per
039 * <a href="ftp://ftp.isi.edu/in-notes/rfc2617.txt">RFC 2617</a>.
040 * <p/>
041 * Basic authentication functions as follows:
042 * <ol>
043 * <li>A request comes in for a resource that requires authentication.</li>
044 * <li>The server replies with a 401 response status, sets the <code>WWW-Authenticate</code> header, and the contents of a
045 * page informing the user that the incoming resource requires authentication.</li>
046 * <li>Upon receiving this <code>WWW-Authenticate</code> challenge from the server, the client then takes a
047 * username and a password and puts them in the following format:
048 * <p><code>username:password</code></p></li>
049 * <li>This token is then base 64 encoded.</li>
050 * <li>The client then sends another request for the same resource with the following header:<br/>
051 * <p><code>Authorization: Basic <em>Base64_encoded_username_and_password</em></code></p></li>
052 * </ol>
053 * The {@link #onAccessDenied(javax.servlet.ServletRequest, javax.servlet.ServletResponse)} method will
054 * only be called if the subject making the request is not
055 * {@link org.apache.shiro.subject.Subject#isAuthenticated() authenticated}
056 *
057 * @see <a href="ftp://ftp.isi.edu/in-notes/rfc2617.txt">RFC 2617</a>
058 * @see <a href="http://en.wikipedia.org/wiki/Basic_access_authentication">Basic Access Authentication</a>
059 * @since 0.9
060 */
061public class BasicHttpAuthenticationFilter extends AuthenticatingFilter {
062
063    /**
064     * This class's private logger.
065     */
066    private static final Logger log = LoggerFactory.getLogger(BasicHttpAuthenticationFilter.class);
067
068    /**
069     * HTTP Authorization header, equal to <code>Authorization</code>
070     */
071    protected static final String AUTHORIZATION_HEADER = "Authorization";
072
073    /**
074     * HTTP Authentication header, equal to <code>WWW-Authenticate</code>
075     */
076    protected static final String AUTHENTICATE_HEADER = "WWW-Authenticate";
077
078    /**
079     * The name that is displayed during the challenge process of authentication, defauls to <code>application</code>
080     * and can be overridden by the {@link #setApplicationName(String) setApplicationName} method.
081     */
082    private String applicationName = "application";
083
084    /**
085     * The authcScheme to look for in the <code>Authorization</code> header, defaults to <code>BASIC</code>
086     */
087    private String authcScheme = HttpServletRequest.BASIC_AUTH;
088
089    /**
090     * The authzScheme value to look for in the <code>Authorization</code> header, defaults to <code>BASIC</code>
091     */
092    private String authzScheme = HttpServletRequest.BASIC_AUTH;

094    /**
095     * Returns the name to use in the ServletResponse's <b><code>WWW-Authenticate</code></b> header.
096     * <p/>
097     * Per RFC 2617, this name name is displayed to the end user when they are asked to authenticate.  Unless overridden
098     * by the {@link #setApplicationName(String) setApplicationName(String)} method, the default value is 'application'.
099     * <p/>
100     * Please see {@link #setApplicationName(String) setApplicationName(String)} for an example of how this functions.
101     *
102     * @return the name to use in the ServletResponse's 'WWW-Authenticate' header.
103     */
104    public String getApplicationName() {
105        return applicationName;
106    }
107
108    /**
109     * Sets the name to use in the ServletResponse's <b><code>WWW-Authenticate</code></b> header.
110     * <p/>
111     * Per RFC 2617, this name name is displayed to the end user when they are asked to authenticate.  Unless overridden
112     * by this method, the default value is &quot;application&quot;
113     * <p/>
114     * For example, setting this property to the value <b><code>Awesome Webapp</code></b> will result in the
115     * following header:
116     * <p/>
117     * <code>WWW-Authenticate: Basic realm=&quot;<b>Awesome Webapp</b>&quot;</code>
118     * <p/>
119     * Side note: As you can see from the header text, the HTTP Basic specification calls
120     * this the authentication 'realm', but we call this the 'applicationName' instead to avoid confusion with
121     * Shiro's Realm constructs.
122     *
123     * @param applicationName the name to use in the ServletResponse's 'WWW-Authenticate' header.
124     */
125    public void setApplicationName(String applicationName) {
126        this.applicationName = applicationName;
127    }

132     * <p/>
133     * Unless overridden by the {@link #setAuthzScheme(String) setAuthzScheme(String)} method, the
134     * default value is <code>BASIC</code>.
135     *
136     * @return the Http 'Authorization' header value that this filter will respond to as indicating a login request
137     */
138    public String getAuthzScheme() {

143     * Sets the HTTP <b><code>Authorization</code></b> header value that this filter will respond to as indicating a

145     * <p/>
146     * Unless overridden by this method, the default value is <code>BASIC</code>
147     *
148     * @param authzScheme the HTTP <code>Authorization</code> header value that this filter will respond to as
149     *                    indicating a login request.
150     */
151    public void setAuthzScheme(String authzScheme) {
152        this.authzScheme = authzScheme;
153    }

164        return authcScheme;

168     * Sets the HTTP <b><code>WWW-Authenticate</code></b> header scheme that this filter will use when sending the

170     *
171     * @param authcScheme the HTTP <code>WWW-Authenticate</code> header scheme that this filter will use when
172     *                    sending the Http Basic challenge response.

175    public void setAuthcScheme(String authcScheme) {
176        this.authcScheme = authcScheme;

179    /**
180     * Processes unauthenticated requests. It handles the two-stage request/challenge authentication protocol.
181     *
182     * @param request  incoming ServletRequest
183     * @param response outgoing ServletResponse
184     * @return true if the request should be processed; false if the request should not continue to be processed
185     */
186    protected boolean onAccessDenied(ServletRequest request, ServletResponse response) throws Exception {
187        boolean loggedIn = false; //false by default or we wouldn't be in this method
188        if (isLoginAttempt(request, response)) {
189            loggedIn = executeLogin(request, response);
190        }
191        if (!loggedIn) {
192            sendChallenge(request, response);
193        }
194        return loggedIn;
195    }
196
197    /**
198     * Determines whether the incoming request is an attempt to log in.
199     * <p/>
200     * The default implementation obtains the value of the request's
201     * {@link #AUTHORIZATION_HEADER AUTHORIZATION_HEADER}, and if it is not <code>null</code>, delegates
202     * to {@link #isLoginAttempt(String) isLoginAttempt(authzHeaderValue)}. If the header is <code>null</code>,
203     * <code>false</code> is returned.
204     *
205     * @param request  incoming ServletRequest
206     * @param response outgoing ServletResponse
207     * @return true if the incoming request is an attempt to log in based, false otherwise
208     */
209    protected boolean isLoginAttempt(ServletRequest request, ServletResponse response) {
210        String authzHeader = getAuthzHeader(request);
211        return authzHeader != null && isLoginAttempt(authzHeader);
212    }
213
214    /**
215     * Delegates to {@link #isLoginAttempt(javax.servlet.ServletRequest, javax.servlet.ServletResponse) isLoginAttempt}.
216     */
217    @Override
218    protected final boolean isLoginRequest(ServletRequest request, ServletResponse response) {
219        return this.isLoginAttempt(request, response);
220    }
221
222    /**
223     * Returns the {@link #AUTHORIZATION_HEADER AUTHORIZATION_HEADER} from the specified ServletRequest.
224     * <p/>
225     * This implementation merely casts the request to an <code>HttpServletRequest</code> and returns the header:
226     * <p/>
227     * <code>HttpServletRequest httpRequest = {@link WebUtils#toHttp(javax.servlet.ServletRequest) toHttp(reaquest)};<br/>
228     * return httpRequest.getHeader({@link #AUTHORIZATION_HEADER AUTHORIZATION_HEADER});</code>
229     *
230     * @param request the incoming <code>ServletRequest</code>
231     * @return the <code>Authorization</code> header's value.
232     */
233    protected String getAuthzHeader(ServletRequest request) {
234        HttpServletRequest httpRequest = WebUtils.toHttp(request);
235        return httpRequest.getHeader(AUTHORIZATION_HEADER);
236    }
237
238    /**
239     * Default implementation that returns <code>true</code> if the specified <code>authzHeader</code>
240     * starts with the same (case-insensitive) characters specified by the
241     * {@link #getAuthzScheme() authzScheme}, <code>false</code> otherwise.
242     * <p/>
243     * That is:
244     * <p/>
245     * <code>String authzScheme = getAuthzScheme().toLowerCase();<br/>
246     * return authzHeader.toLowerCase().startsWith(authzScheme);</code>
247     *
248     * @param authzHeader the 'Authorization' header value (guaranteed to be non-null if the
249     *                    {@link #isLoginAttempt(javax.servlet.ServletRequest, javax.servlet.ServletResponse)} method is not overriden).
250     * @return <code>true</code> if the authzHeader value matches that configured as defined by
251     *         the {@link #getAuthzScheme() authzScheme}.
252     */
253    protected boolean isLoginAttempt(String authzHeader) {
254        String authzScheme = getAuthzScheme().toLowerCase();
255        return authzHeader.toLowerCase().startsWith(authzScheme);
256    }
257
258    /**
259     * Builds the challenge for authorization by setting a HTTP <code>401</code> (Unauthorized) status as well as the
260     * response's {@link #AUTHENTICATE_HEADER AUTHENTICATE_HEADER}.
261     * <p/>
262     * The header value constructed is equal to:
263     * <p/>
264     * <code>{@link #getAuthcScheme() getAuthcScheme()} + " realm=\"" + {@link #getApplicationName() getApplicationName()} + "\"";</code>
265     *
266     * @param request  incoming ServletRequest, ignored by this implementation
267     * @param response outgoing ServletResponse
268     * @return false - this sends the challenge to be sent back
269     */
270    protected boolean sendChallenge(ServletRequest request, ServletResponse response) {
271        if (log.isDebugEnabled()) {
272            log.debug("Authentication required: sending 401 Authentication challenge response.");
273        }
274        HttpServletResponse httpResponse = WebUtils.toHttp(response);
275        httpResponse.setStatus(HttpServletResponse.SC_UNAUTHORIZED);
276        String authcHeader = getAuthcScheme() + " realm=\"" + getApplicationName() + "\"";
277        httpResponse.setHeader(AUTHENTICATE_HEADER, authcHeader);
278        return false;
279    }
280
281    /**
282     * Creates an AuthenticationToken for use during login attempt with the provided credentials in the http header.
283     * <p/>
284     * This implementation:
285     * <ol><li>acquires the username and password based on the request's
286     * {@link #getAuthzHeader(javax.servlet.ServletRequest) authorization header} via the
287     * {@link #getPrincipalsAndCredentials(String, javax.servlet.ServletRequest) getPrincipalsAndCredentials} method</li>
288     * <li>The return value of that method is converted to an <code>AuthenticationToken</code> via the
289     * {@link #createToken(String, String, javax.servlet.ServletRequest, javax.servlet.ServletResponse) createToken} method</li>
290     * <li>The created <code>AuthenticationToken</code> is returned.</li>
291     * </ol>
292     *
293     * @param request  incoming ServletRequest
294     * @param response outgoing ServletResponse
295     * @return the AuthenticationToken used to execute the login attempt
296     */
297    protected AuthenticationToken createToken(ServletRequest request, ServletResponse response) {
298        String authorizationHeader = getAuthzHeader(request);
299        if (authorizationHeader == null || authorizationHeader.length() == 0) {
300            // Create an empty authentication token since there is no
301            // Authorization header.
302            return createToken("", "", request, response);
303        }
304
305        if (log.isDebugEnabled()) {
306            log.debug("Attempting to execute login with headers [" + authorizationHeader + "]");
307        }
308
309        String[] prinCred = getPrincipalsAndCredentials(authorizationHeader, request);
310        if (prinCred == null || prinCred.length < 2) {
311            // Create an authentication token with an empty password,
312            // since one hasn't been provided in the request.
313            String username = prinCred == null || prinCred.length == 0 ? "" : prinCred[0];
314            return createToken(username, "", request, response);
315        }
316
317        String username = prinCred[0];
318        String password = prinCred[1];
319
320        return createToken(username, password, request, response);
321    }
322
323    /**
324     * Returns the username obtained from the
325     * {@link #getAuthzHeader(javax.servlet.ServletRequest) authorizationHeader}.
326     * <p/>
327     * Once the {@code authzHeader} is split per the RFC (based on the space character ' '), the resulting split tokens
328     * are translated into the username/password pair by the
329     * {@link #getPrincipalsAndCredentials(String, String) getPrincipalsAndCredentials(scheme,encoded)} method.
330     *
331     * @param authorizationHeader the authorization header obtained from the request.
332     * @param request             the incoming ServletRequest
333     * @return the username (index 0)/password pair (index 1) submitted by the user for the given header value and request.
334     * @see #getAuthzHeader(javax.servlet.ServletRequest)
335     */
336    protected String[] getPrincipalsAndCredentials(String authorizationHeader, ServletRequest request) {
337        if (authorizationHeader == null) {
338            return null;
339        }
340        String[] authTokens = authorizationHeader.split(" ");
341        if (authTokens == null || authTokens.length < 2) {
342            return null;
343        }
344        return getPrincipalsAndCredentials(authTokens[0], authTokens[1]);
345    }
346
347    /**
348     * Returns the username and password pair based on the specified <code>encoded</code> String obtained from
349     * the request's authorization header.
350     * <p/>
351     * Per RFC 2617, the default implementation first Base64 decodes the string and then splits the resulting decoded
352     * string into two based on the ":" character.  That is:
353     * <p/>
354     * <code>String decoded = Base64.decodeToString(encoded);<br/>
355     * return decoded.split(":");</code>
356     *
357     * @param scheme  the {@link #getAuthcScheme() authcScheme} found in the request
358     *                {@link #getAuthzHeader(javax.servlet.ServletRequest) authzHeader}.  It is ignored by this implementation,
359     *                but available to overriding implementations should they find it useful.
360     * @param encoded the Base64-encoded username:password value found after the scheme in the header
361     * @return the username (index 0)/password (index 1) pair obtained from the encoded header data.
362     */
363    protected String[] getPrincipalsAndCredentials(String scheme, String encoded) {
364        String decoded = Base64.decodeToString(encoded);
365        return decoded.split(":", 2);
366    }
