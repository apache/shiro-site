019package org.apache.shiro.config.event;
020
021import org.apache.shiro.event.Subscribe;
022import org.slf4j.Logger;
023import org.slf4j.LoggerFactory;
024
025/**
026 * A stock bean listener implementation that logs all BeanEvents as TRACE log statements.
027 *
028 * @since 1.3
029 */
030public class LoggingBeanEventListener {
031
032    private static final Logger logger = LoggerFactory.getLogger(LoggingBeanEventListener.class);
033    private static final String SUFFIX = BeanEvent.class.getSimpleName();
034
035    @Subscribe
036    public void onEvent(BeanEvent e) {
037        String className = e.getClass().getSimpleName();
038        int i = className.lastIndexOf(SUFFIX);
039        String subclassPrefix = i > 0 ? className.substring(0, i) : className;
040        logger.trace("{} bean '{}' [{}]", new Object[]{subclassPrefix, e.getBeanName(), e.getBean()});
041    }
042}
