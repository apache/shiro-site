019    package org.apache.shiro.web.session.mgt;
020    
021    import org.apache.shiro.session.mgt.SessionManager;
022    
023    /**
024     * {@link SessionManager} specific to web-enabled applications.
025     *
026     * @since 1.2
027     * @see ServletContainerSessionManager
028     * @see DefaultWebSessionManager
029     */
030    public interface WebSessionManager extends SessionManager {
031    
032        /**
033         * Returns {@code true} if session management and storage is managed by the underlying Servlet container or
034         * {@code false} if managed by Shiro directly (called 'native' sessions).
035         * <p/>
036         * If sessions are enabled, Shiro can make use of Sessions to retain security information from
037         * request to request.  This method indicates whether Shiro would use the Servlet container sessions to fulfill its
038         * needs, or if it would use its own native session management instead (which can support enterprise features
039         * - like distributed caching - in a container-independent manner).
040         *
041         * @return {@code true} if session management and storage is managed by the underlying Servlet container or
042         *         {@code false} if managed by Shiro directly (called 'native' sessions).
043         */
044        boolean isServletContainerSessions();
045    }
