044/**
045 * Web-application capable {@link org.apache.shiro.session.mgt.SessionManager SessionManager} implementation.
046 *
047 * @since 0.9
048 */
049public class DefaultWebSessionManager extends DefaultSessionManager implements WebSessionManager {
050
051    private static final Logger log = LoggerFactory.getLogger(DefaultWebSessionManager.class);
052
053    private Cookie sessionIdCookie;
054    private boolean sessionIdCookieEnabled;
055
056    public DefaultWebSessionManager() {
057        Cookie cookie = new SimpleCookie(ShiroHttpSession.DEFAULT_SESSION_ID_NAME);
058        cookie.setHttpOnly(true); //more secure, protects against XSS attacks
059        this.sessionIdCookie = cookie;
060        this.sessionIdCookieEnabled = true;
061    }
062
063    public Cookie getSessionIdCookie() {
064        return sessionIdCookie;
065    }
066
067    @SuppressWarnings({"UnusedDeclaration"})
068    public void setSessionIdCookie(Cookie sessionIdCookie) {
069        this.sessionIdCookie = sessionIdCookie;
070    }
071
072    public boolean isSessionIdCookieEnabled() {
073        return sessionIdCookieEnabled;
074    }
075
076    @SuppressWarnings({"UnusedDeclaration"})
077    public void setSessionIdCookieEnabled(boolean sessionIdCookieEnabled) {
078        this.sessionIdCookieEnabled = sessionIdCookieEnabled;
079    }
080
081    private void storeSessionId(Serializable currentId, HttpServletRequest request, HttpServletResponse response) {
082        if (currentId == null) {
083            String msg = "sessionId cannot be null when persisting for subsequent requests.";
084            throw new IllegalArgumentException(msg);
085        }
086        Cookie template = getSessionIdCookie();
087        Cookie cookie = new SimpleCookie(template);
088        String idString = currentId.toString();
089        cookie.setValue(idString);
090        cookie.saveTo(request, response);
091        log.trace("Set session ID cookie for session with id {}", idString);
092    }
093
094    private void removeSessionIdCookie(HttpServletRequest request, HttpServletResponse response) {
095        getSessionIdCookie().removeFrom(request, response);
096    }
097
098    private String getSessionIdCookieValue(ServletRequest request, ServletResponse response) {
099        if (!isSessionIdCookieEnabled()) {
100            log.debug("Session ID cookie is disabled - session id will not be acquired from a request cookie.");
101            return null;
102        }
103        if (!(request instanceof HttpServletRequest)) {
104            log.debug("Current request is not an HttpServletRequest - cannot get session ID cookie.  Returning null.");
105            return null;
106        }
107        HttpServletRequest httpRequest = (HttpServletRequest) request;
108        return getSessionIdCookie().readValue(httpRequest, WebUtils.toHttp(response));
109    }
110
111    private Serializable getReferencedSessionId(ServletRequest request, ServletResponse response) {
112
113        String id = getSessionIdCookieValue(request, response);
114        if (id != null) {
115            request.setAttribute(ShiroHttpServletRequest.REFERENCED_SESSION_ID_SOURCE,
116                    ShiroHttpServletRequest.COOKIE_SESSION_ID_SOURCE);
117        } else {
118            //not in a cookie, or cookie is disabled - try the request params as a fallback (i.e. URL rewriting):
119            String name = getSessionIdName();
120            id = request.getParameter(name);
121            if (id == null) {
122                //try lowercase:
123                id = request.getParameter(name.toLowerCase());
124            }
125            if (id != null) {
126                request.setAttribute(ShiroHttpServletRequest.REFERENCED_SESSION_ID_SOURCE,
127                        ShiroHttpServletRequest.URL_SESSION_ID_SOURCE);
128            }
129        }
130        if (id != null) {
131            request.setAttribute(ShiroHttpServletRequest.REFERENCED_SESSION_ID, id);
132            //automatically mark it valid here.  If it is invalid, the
133            //onUnknownSession method below will be invoked and we'll remove the attribute at that time.
134            request.setAttribute(ShiroHttpServletRequest.REFERENCED_SESSION_ID_IS_VALID, Boolean.TRUE);
135        }
136        return id;
137    }
138
139    //since 1.2.1
140    private String getSessionIdName() {
141        String name = this.sessionIdCookie != null ? this.sessionIdCookie.getName() : null;
142        if (name == null) {
143            name = ShiroHttpSession.DEFAULT_SESSION_ID_NAME;
144        }
145        return name;
146    }
147
148    protected Session createExposedSession(Session session, SessionContext context) {
149        if (!WebUtils.isWeb(context)) {
150            return super.createExposedSession(session, context);
151        }
152        ServletRequest request = WebUtils.getRequest(context);
153        ServletResponse response = WebUtils.getResponse(context);
154        SessionKey key = new WebSessionKey(session.getId(), request, response);
155        return new DelegatingSession(this, key);
156    }
157
158    protected Session createExposedSession(Session session, SessionKey key) {
159        if (!WebUtils.isWeb(key)) {
160            return super.createExposedSession(session, key);
161        }
162
163        ServletRequest request = WebUtils.getRequest(key);
164        ServletResponse response = WebUtils.getResponse(key);
165        SessionKey sessionKey = new WebSessionKey(session.getId(), request, response);
166        return new DelegatingSession(this, sessionKey);
167    }
168
169    /**
170     * Stores the Session's ID, usually as a Cookie, to associate with future requests.
171     *
172     * @param session the session that was just {@link #createSession created}.
173     */
174    @Override
175    protected void onStart(Session session, SessionContext context) {
176        super.onStart(session, context);
177
178        if (!WebUtils.isHttp(context)) {
179            log.debug("SessionContext argument is not HTTP compatible or does not have an HTTP request/response " +
180                    "pair. No session ID cookie will be set.");
181            return;
182
183        }
184        HttpServletRequest request = WebUtils.getHttpRequest(context);
185        HttpServletResponse response = WebUtils.getHttpResponse(context);
186
187        if (isSessionIdCookieEnabled()) {
188            Serializable sessionId = session.getId();
189            storeSessionId(sessionId, request, response);
190        } else {
191            log.debug("Session ID cookie is disabled.  No cookie has been set for new session with id {}", session.getId());
192        }
193
194        request.removeAttribute(ShiroHttpServletRequest.REFERENCED_SESSION_ID_SOURCE);
195        request.setAttribute(ShiroHttpServletRequest.REFERENCED_SESSION_IS_NEW, Boolean.TRUE);
196    }
197
198    @Override
199    public Serializable getSessionId(SessionKey key) {
200        Serializable id = super.getSessionId(key);
201        if (id == null && WebUtils.isWeb(key)) {
202            ServletRequest request = WebUtils.getRequest(key);
203            ServletResponse response = WebUtils.getResponse(key);
204            id = getSessionId(request, response);
205        }
206        return id;
207    }
208
209    protected Serializable getSessionId(ServletRequest request, ServletResponse response) {
210        return getReferencedSessionId(request, response);
211    }
212
213    @Override
214    protected void onExpiration(Session s, ExpiredSessionException ese, SessionKey key) {
215        super.onExpiration(s, ese, key);
216        onInvalidation(key);
217    }
218
219    @Override
220    protected void onInvalidation(Session session, InvalidSessionException ise, SessionKey key) {
221        super.onInvalidation(session, ise, key);
222        onInvalidation(key);
223    }
224
225    private void onInvalidation(SessionKey key) {
226        ServletRequest request = WebUtils.getRequest(key);
227        if (request != null) {
228            request.removeAttribute(ShiroHttpServletRequest.REFERENCED_SESSION_ID_IS_VALID);
229        }
230        if (WebUtils.isHttp(key)) {
231            log.debug("Referenced session was invalid.  Removing session ID cookie.");
232            removeSessionIdCookie(WebUtils.getHttpRequest(key), WebUtils.getHttpResponse(key));
233        } else {
234            log.debug("SessionKey argument is not HTTP compatible or does not have an HTTP request/response " +
235                    "pair. Session ID cookie will not be removed due to invalidated session.");
236        }
237    }
238
239    @Override
240    protected void onStop(Session session, SessionKey key) {
241        super.onStop(session, key);
242        if (WebUtils.isHttp(key)) {
243            HttpServletRequest request = WebUtils.getHttpRequest(key);
244            HttpServletResponse response = WebUtils.getHttpResponse(key);
245            log.debug("Session has been stopped (subject logout or explicit stop).  Removing session ID cookie.");
246            removeSessionIdCookie(request, response);
247        } else {
248            log.debug("SessionKey argument is not HTTP compatible or does not have an HTTP request/response " +
249                    "pair. Session ID cookie will not be removed due to stopped session.");
250        }
251    }
252
253    /**
254     * This is a native session manager implementation, so this method returns {@code false} always.
255     *
256     * @return {@code false} always
257     * @since 1.2
258     */
259    public boolean isServletContainerSessions() {
260        return false;
261    }
