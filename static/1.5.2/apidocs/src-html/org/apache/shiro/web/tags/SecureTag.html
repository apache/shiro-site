019package org.apache.shiro.web.tags;
020
021import javax.servlet.jsp.JspException;
022import javax.servlet.jsp.tagext.TagSupport;
023
024import org.slf4j.Logger;
025import org.slf4j.LoggerFactory;
026
027import org.apache.shiro.SecurityUtils;
028import org.apache.shiro.subject.Subject;
029
030/**
031 * @since 0.1
032 */
033public abstract class SecureTag extends TagSupport {
034
035    //TODO - complete JavaDoc
036
037    private static final Logger log = LoggerFactory.getLogger(SecureTag.class);
038
039    public SecureTag() {
040    }
041
042    protected Subject getSubject() {
043        return SecurityUtils.getSubject();
044    }
045
046    protected void verifyAttributes() throws JspException {
047    }
048
049    public int doStartTag() throws JspException {
050
051        verifyAttributes();
052
053        return onDoStartTag();
054    }
055
056    public abstract int onDoStartTag() throws JspException;
057}
