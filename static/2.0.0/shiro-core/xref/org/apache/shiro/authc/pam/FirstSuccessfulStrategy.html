19  package org.apache.shiro.authc.pam;
20  
21  import org.apache.shiro.authc.AuthenticationException;
22  import org.apache.shiro.authc.AuthenticationInfo;
23  import org.apache.shiro.authc.AuthenticationToken;
24  import org.apache.shiro.realm.Realm;
25  import org.apache.shiro.subject.PrincipalCollection;
26  
27  import java.util.Collection;
28  
29  /**
30   * {@link AuthenticationStrategy} implementation that only accepts the account data from
31   * the first successfully consulted Realm and ignores all subsequent realms.  This is slightly
32   * different behavior than {@link AtLeastOneSuccessfulStrategy}, so please review both to see
33   * which one meets your needs better.
34   *
35   * @see AtLeastOneSuccessfulStrategy AtLeastOneSuccessfulAuthenticationStrategy
36   * @since 0.9
37   */
38  public class FirstSuccessfulStrategy extends AbstractAuthenticationStrategy {
39  
40      private boolean stopAfterFirstSuccess;
41  
42      public void setStopAfterFirstSuccess(boolean stopAfterFirstSuccess) {
43          this.stopAfterFirstSuccess = stopAfterFirstSuccess;
44      }
45  
46      public boolean getStopAfterFirstSuccess() {
47          return stopAfterFirstSuccess;
48      }
49  
50      /**
51       * Returns {@code null} immediately, relying on this class's {@link #merge merge} implementation to return
52       * only the first {@code info} object it encounters, ignoring all subsequent ones.
53       */
54      public AuthenticationInfo beforeAllAttempts(Collection<? extends Realm> realms, AuthenticationToken token)
55              throws AuthenticationException {
56          return null;
57      }
58  
59      /**
60       * Throws ShortCircuitIterationException if stopAfterFirstSuccess is set and authentication is
61       * successful with a previously consulted realm.
62       * Returns the <code>aggregate</code> method argument, without modification
63       * otherwise.
64       */
65      public AuthenticationInfo beforeAttempt(Realm realm, AuthenticationToken token, AuthenticationInfo aggregate)
66              throws AuthenticationException {
67          if (getStopAfterFirstSuccess() && aggregate != null && !isEmpty(aggregate.getPrincipals())) {
68              throw new ShortCircuitIterationException();
69          }
70          return aggregate;
71      }
72  
73      private static boolean isEmpty(PrincipalCollection pc) {
74          return pc == null || pc.isEmpty();
75      }
76  
77      /**
78       * Returns the specified {@code aggregate} instance if is non null and valid (that is, has principals and they are
79       * not empty) immediately, or, if it is null or not valid, the {@code info} argument is returned instead.
80       * <p/>
81       * This logic ensures that the first valid info encountered is the one retained and all subsequent ones are ignored,
82       * since this strategy mandates that only the info from the first successfully authenticated realm be used.
83       */
84      protected AuthenticationInfo merge(AuthenticationInfo info, AuthenticationInfo aggregate) {
85          if (aggregate != null && !isEmpty(aggregate.getPrincipals())) {
86              return aggregate;
87          }
88          return info != null ? info : aggregate;
89      }
90  }
