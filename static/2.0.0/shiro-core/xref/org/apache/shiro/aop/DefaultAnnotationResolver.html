19  package org.apache.shiro.aop;
20  
21  import java.lang.annotation.Annotation;
22  import java.lang.reflect.Method;
23  
24  /**
25   * Default {@code AnnotationResolver} implementation that merely inspects the
26   * {@link MethodInvocation MethodInvocation}'s {@link MethodInvocation#getMethod() target method},
27   * and returns {@code targetMethod}.{@link Method#getAnnotation(Class) getAnnotation(class)}.
28   * <p/>
29   * Unfortunately Java's default reflection API for Annotations is not very robust, and this logic
30   * may not be enough - if the incoming method invocation represents a method from an interface,
31   * this default logic would not discover the annotation if it existed on the method implementation
32   * directly (as opposed to being defined directly in the interface definition).
33   * <p/>
34   * More complex class hierarchy traversal logic is required to exhaust a method's target object's
35   * classes, parent classes, interfaces and parent interfaces.  That logic will likely be added
36   * to this implementation in due time, but for now, this implementation relies on the JDK's default
37   * {@link Method#getAnnotation(Class) Method.getAnnotation(class)} logic.
38   *
39   * @since 1.1
40   */
41  public class DefaultAnnotationResolver implements AnnotationResolver {
42  
43      /**
44       * Returns {@code methodInvocation.}{@link org.apache.shiro.aop.MethodInvocation#getMethod() getMethod()}.
45       * {@link Method#getAnnotation(Class) getAnnotation(clazz)}.
46       *
47       * @param mi    the intercepted method to be invoked.
48       * @param clazz the annotation class to use to find an annotation instance on the method.
49       * @return the discovered annotation or {@code null} if an annotation instance could not be
50       * found.
51       */
52      public Annotation getAnnotation(MethodInvocation mi, Class<? extends Annotation> clazz) {
53          if (mi == null) {
54              throw new IllegalArgumentException("method argument cannot be null");
55          }
56          Method m = mi.getMethod();
57          if (m == null) {
58              String msg = MethodInvocation.class.getName() + " parameter incorrectly constructed.  getMethod() returned null";
59              throw new IllegalArgumentException(msg);
60  
61          }
62          Annotation annotation = m.getAnnotation(clazz);
63          if (annotation == null) {
64              Object miThis = mi.getThis();
65              //SHIRO-473 - miThis could be null for static methods, just return null
66              annotation = miThis != null ? miThis.getClass().getAnnotation(clazz) : null;
67          }
68          return annotation;
69      }
70  }
