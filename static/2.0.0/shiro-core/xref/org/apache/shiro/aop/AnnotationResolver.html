19  package org.apache.shiro.aop;
20  
21  import java.lang.annotation.Annotation;
22  
23  /**
24   * Defines an AOP-framework-independent way of determining if an Annotation exists on a Method.
25   *
26   * @since 1.1
27   */
28  public interface AnnotationResolver {
29  
30      /**
31       * Returns an {@link Annotation} instance of the specified type based on the given
32       * {@link MethodInvocation MethodInvocation} argument, or {@code null} if no annotation
33       * of that type could be found. First checks the invoked method itself and if not found,
34       * then the class for the existence of the same annotation.
35       *
36       * @param mi    the intercepted method to be invoked.
37       * @param clazz the annotation class of the annotation to find.
38       * @return the method's annotation of the specified type or {@code null} if no annotation of
39       * that type could be found.
40       */
41      Annotation getAnnotation(MethodInvocation mi, Class<? extends Annotation> clazz);
42  }
