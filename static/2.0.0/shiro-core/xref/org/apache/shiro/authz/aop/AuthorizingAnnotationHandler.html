19  package org.apache.shiro.authz.aop;
20  
21  import java.lang.annotation.Annotation;
22  
23  import org.apache.shiro.aop.AnnotationHandler;
24  import org.apache.shiro.authz.AuthorizationException;
25  
26  /**
27   * An AnnotationHandler that executes authorization (access control) behavior based on directive(s) found in a
28   * JSR-175 Annotation.
29   *
30   * @since 0.9.0
31   */
32  public abstract class AuthorizingAnnotationHandler extends AnnotationHandler {
33  
34      /**
35       * Constructs an <code>AuthorizingAnnotationHandler</code> who processes annotations of the
36       * specified type.  Immediately calls <code>super(annotationClass)</code>.
37       *
38       * @param annotationClass the type of annotation this handler will process.
39       */
40      public AuthorizingAnnotationHandler(Class<? extends Annotation> annotationClass) {
41          super(annotationClass);
42      }
43  
44      /**
45       * Ensures the calling Subject is authorized to execute based on the directive(s) found in the given
46       * annotation.
47       * <p/>
48       * As this is an AnnotationMethodInterceptor, the implementations of this method typically inspect the annotation
49       * and perform a corresponding authorization check based.
50       *
51       * @param a the <code>Annotation</code> to check for performing an authorization check.
52       * @throws org.apache.shiro.authz.AuthorizationException if the class/instance/method is not allowed to proceed/execute.
53       */
54      public abstract void assertAuthorized(Annotation a) throws AuthorizationException;
55  }
