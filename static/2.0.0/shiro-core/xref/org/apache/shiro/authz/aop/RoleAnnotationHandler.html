19  package org.apache.shiro.authz.aop;
20  
21  import org.apache.shiro.authz.AuthorizationException;
22  import org.apache.shiro.authz.annotation.Logical;
23  import org.apache.shiro.authz.annotation.RequiresRoles;
24  
25  import java.lang.annotation.Annotation;
26  import java.util.Arrays;
27  
28  /**
29   * Checks to see if a @{@link org.apache.shiro.authz.annotation.RequiresRoles RequiresRoles} annotation is declared,
30   * and if so, performs a role check to see if the calling <code>Subject</code> is allowed to proceed.
31   *
32   * @since 0.9.0
33   */
34  public class RoleAnnotationHandler extends AuthorizingAnnotationHandler {
35  
36      /**
37       * Default no-argument constructor that ensures this handler looks for
38       * {@link org.apache.shiro.authz.annotation.RequiresRoles RequiresRoles} annotations.
39       */
40      public RoleAnnotationHandler() {
41          super(RequiresRoles.class);
42      }
43  
44      /**
45       * Ensures that the calling <code>Subject</code> has the Annotation's specified roles, and if not, throws an
46       * <code>AuthorizingException</code> indicating that access is denied.
47       *
48       * @param a the RequiresRoles annotation to use to check for one or more roles
49       * @throws org.apache.shiro.authz.AuthorizationException if the calling <code>Subject</code> does not have the role(s)
50       *                                                       necessary to proceed.
51       */
52      public void assertAuthorized(Annotation a) throws AuthorizationException {
53          if (!(a instanceof RequiresRoles)) {
54              return;
55          }
56  
57          RequiresRoles rrAnnotation = (RequiresRoles) a;
58          String[] roles = rrAnnotation.value();
59  
60          if (roles.length == 1) {
61              getSubject().checkRole(roles[0]);
62              return;
63          }
64          if (Logical.AND.equals(rrAnnotation.logical())) {
65              getSubject().checkRoles(Arrays.asList(roles));
66              return;
67          }
68          if (Logical.OR.equals(rrAnnotation.logical())) {
69              // Avoid processing exceptions unnecessarily - "delay" throwing the exception by calling hasRole first
70              boolean hasAtLeastOneRole = false;
71              for (String role : roles) {
72                  if (getSubject().hasRole(role)) {
73                      hasAtLeastOneRole = true;
74                  }
75              }
76              // Cause the exception if none of the role match, note that the exception message will be a bit misleading
77              if (!hasAtLeastOneRole) {
78                  getSubject().checkRole(roles[0]);
79              }
80          }
81      }
82  
83  }
