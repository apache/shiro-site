19  package org.apache.shiro.authz.aop;
20  
21  import org.apache.shiro.authz.AuthorizationException;
22  
23  import javax.annotation.security.RolesAllowed;
24  import java.lang.annotation.Annotation;
25  
26  /**
27   * Checks to see if a @{@link RolesAllowed} annotation is declared, and if so, performs
28   * a role check to see if the calling <code>Subject</code> is allowed to proceed.
29   *
30   * @since 2.0
31   */
32  public class RolesAllowedAnnotationHandler extends AuthorizingAnnotationHandler {
33  
34      /**
35       * Default no-argument constructor that ensures this handler looks for
36       * {@link org.apache.shiro.authz.annotation.RequiresRoles RequiresRoles} annotations.
37       */
38      public RolesAllowedAnnotationHandler() {
39          super(RolesAllowed.class);
40      }
41  
42      /**
43       * Ensures that the calling <code>Subject</code> has one of the Annotation's specified roles, and if not, throws an
44       * <code>AuthorizingException</code> indicating that access is denied.
45       *
46       * @param a the RolesAllowed annotation to use to check for one or more roles
47       * @throws org.apache.shiro.authz.AuthorizationException if the calling <code>Subject</code> does not have the role
48       *                                                       necessary to proceed.
49       */
50      @Override
51      public void assertAuthorized(Annotation a) throws AuthorizationException {
52          if (!(a instanceof RolesAllowed)) {
53              return;
54          }
55  
56          RolesAllowed raAnnotation = (RolesAllowed) a;
57          String[] roles = raAnnotation.value();
58  
59          if (roles.length == 1) {
60              getSubject().checkRole(roles[0]);
61              return;
62          }
63  
64          // Logical OR
65  
66          // Avoid processing exceptions unnecessarily - "delay" throwing the exception by calling hasRole first
67          boolean hasAtLeastOneRole = false;
68          for (String role : roles) {
69              if (getSubject().hasRole(role)) {
70                  hasAtLeastOneRole = true;
71              }
72          }
73          // Cause the exception if none of the role match, note that the exception message will be a bit misleading
74          if (!hasAtLeastOneRole) {
75              getSubject().checkRole(roles[0]);
76          }
77      }
78  }
