19  package org.apache.shiro.authz.annotation;
20  
21  import java.lang.annotation.Documented;
22  import java.lang.annotation.ElementType;
23  import java.lang.annotation.Retention;
24  import java.lang.annotation.RetentionPolicy;
25  import java.lang.annotation.Target;
26  
27  /**
28   * <p>
29   * Requires the current executor's Subject to imply a particular permission in
30   * order to execute the annotated method.  If the executor's associated
31   * {@link org.apache.shiro.subject.Subject Subject} determines that the
32   * executor does not imply the specified permission, the method will not be executed.
33   * </p>
34   *
35   * <p>For example, this declaration:
36   * <p/>
37   * <code>&#64;RequiresPermissions( {"file:read", "write:aFile.txt"} )<br/>
38   * void someMethod();</code>
39   * <p/>
40   * indicates the current user must be able to both <tt>read</tt> and <tt>write</tt>
41   * to the file <tt>aFile.txt</tt> in order for the <tt>someMethod()</tt> to execute, otherwise
42   * an {@link org.apache.shiro.authz.AuthorizationException AuthorizationException} will be thrown.
43   *
44   * @see org.apache.shiro.subject.Subject#checkPermission
45   * @since 0.1
46   */
47  @Target({ElementType.TYPE, ElementType.METHOD})
48  @Retention(RetentionPolicy.RUNTIME)
49  @Documented
50  public @interface RequiresPermissions {
51  
52      /**
53       * The permission string which will be passed to {@link org.apache.shiro.subject.Subject#isPermitted(String)}
54       * to determine if the user is allowed to invoke the code protected by this annotation.
55       */
56      String[] value();
57  
58      /**
59       * The logical operation for the permission checks in case multiple roles are specified. AND is the default
60       *
61       * @since 1.1.0
62       */
63      Logical logical() default Logical.AND;
64  
65  }
