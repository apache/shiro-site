19  package org.apache.shiro.authz.aop;
20  
21  import org.apache.shiro.aop.AnnotationResolver;
22  
23  /**
24   * Checks to see if a @{@link org.apache.shiro.authz.annotation.RequiresGuest RequiresGuest} annotation
25   * is declared, and if so, ensures the calling <code>Subject</code> does <em>not</em>
26   * have an {@link org.apache.shiro.subject.Subject#getPrincipal() identity} before invoking the method.
27   * <p>
28   * This annotation essentially ensures that <code>subject.
29   * {@link org.apache.shiro.subject.Subject#getPrincipal() getPrincipal()} == null</code>.
30   *
31   * @since 0.9.0
32   */
33  public class GuestAnnotationMethodInterceptor extends AuthorizingAnnotationMethodInterceptor {
34  
35      /**
36       * Default no-argument constructor that ensures this interceptor looks for
37       * {@link org.apache.shiro.authz.annotation.RequiresGuest RequiresGuest} annotations in a method
38       * declaration.
39       */
40      public GuestAnnotationMethodInterceptor() {
41          super(new GuestAnnotationHandler());
42      }
43  
44      /**
45       * @param resolver
46       * @since 1.1
47       */
48      public GuestAnnotationMethodInterceptor(AnnotationResolver resolver) {
49          super(new GuestAnnotationHandler(), resolver);
50      }
51  
52  }
