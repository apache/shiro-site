19  package org.apache.shiro.session.mgt;
20  
21  /**
22   * Interface that should be implemented by classes that can control validating sessions on a regular
23   * basis.  This interface is used as a delegate for session validation
24   * by the {@link org.apache.shiro.session.mgt.DefaultSessionManager}
25   *
26   * @see org.apache.shiro.session.mgt.DefaultSessionManager#setSessionValidationScheduler(SessionValidationScheduler)
27   * @since 0.1
28   */
29  public interface SessionValidationScheduler {
30  
31      /**
32       * Returns <code>true</code> if this Scheduler is enabled and ready to begin validation at the appropriate time,
33       * <code>false</code> otherwise.
34       * <p/>
35       * It does <em>not</em> indicate if the validation is actually executing at that instant - only that it is prepared
36       * to do so at the appropriate time.
37       *
38       * @return <code>true</code> if this Scheduler is enabled and ready to begin validation at the appropriate time,
39       * <code>false</code> otherwise.
40       */
41      boolean isEnabled();
42  
43      /**
44       * Enables the session validation job.
45       */
46      void enableSessionValidation();
47  
48      /**
49       * Disables the session validation job.
50       */
51      void disableSessionValidation();
52  
53  }
