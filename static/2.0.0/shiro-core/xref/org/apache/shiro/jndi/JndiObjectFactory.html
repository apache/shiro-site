19  package org.apache.shiro.jndi;
20  
21  import org.apache.shiro.lang.util.Factory;
22  
23  import javax.naming.NamingException;
24  
25  /**
26   * A factory implementation intended to be used to look up objects in jndi.
27   *
28   * @param <T>
29   * @since 1.2
30   */
31  public class JndiObjectFactory<T> extends JndiLocator implements Factory<T> {
32  
33      private String resourceName;
34      private Class<? extends T> requiredType;
35  
36      @SuppressWarnings("unchecked")
37      public T getInstance() {
38          try {
39              if (requiredType != null) {
40                  return requiredType.cast(this.lookup(resourceName, requiredType));
41              } else {
42                  return (T) this.lookup(resourceName);
43              }
44          } catch (NamingException e) {
45              final String typeName = requiredType != null ? requiredType.getName() : "object";
46              throw new IllegalStateException("Unable to look up " + typeName + " with jndi name '" + resourceName + "'.", e);
47          }
48      }
49  
50      public String getResourceName() {
51          return resourceName;
52      }
53  
54      public void setResourceName(String resourceName) {
55          this.resourceName = resourceName;
56      }
57  
58      public Class<? extends T> getRequiredType() {
59          return requiredType;
60      }
61  
62      public void setRequiredType(Class<? extends T> requiredType) {
63          this.requiredType = requiredType;
64      }
65  }
