20  package org.apache.shiro.testing.jaxrs.app.json;
21  
22  import javax.enterprise.context.Dependent;
23  import javax.json.bind.Jsonb;
24  import javax.json.bind.JsonbBuilder;
25  import javax.json.bind.JsonbConfig;
26  import javax.json.bind.config.PropertyNamingStrategy;
27  import javax.ws.rs.ext.ContextResolver;
28  import javax.ws.rs.ext.Provider;
29  
30  @Dependent
31  @Provider
32  public class JsonbConfigProvider implements ContextResolver<Jsonb> {
33      @Override
34      public Jsonb getContext(Class<?> type) {
35          JsonbConfig config = getJsonbConfig();
36          return JsonbBuilder.newBuilder().withConfig(config).build();
37      }
38  
39      private JsonbConfig getJsonbConfig() {
40          return new JsonbConfig()
41                  .withPropertyNamingStrategy(PropertyNamingStrategy.LOWER_CASE_WITH_UNDERSCORES)
42                  .withAdapters(new StormtrooperIdConverter());
43      }
44  }
