14  package org.apache.shiro.testing.jakarta.ee;
15  
16  import java.util.Map;
17  import javax.enterprise.context.ApplicationScoped;
18  import javax.inject.Named;
19  
20  import lombok.Getter;
21  import lombok.Setter;
22  import org.apache.shiro.authc.SimpleAccount;
23  import org.apache.shiro.authz.SimpleRole;
24  import org.apache.shiro.lang.util.StringUtils;
25  import org.apache.shiro.realm.text.IniRealm;
26  import org.apache.shiro.subject.SimplePrincipalCollection;
27  
28  @Named
29  @ApplicationScoped
30  public class PropertyRealm extends IniRealm {
31      @Getter
32      @Setter
33      private IniRealm iniRealm;
34  
35      @Override
36      protected void onInit() {
37          setName(getClass().getName());
38          setIni(iniRealm.getIni());
39          super.onInit();
40      }
41  
42      @Override
43      @SuppressWarnings("MagicNumber")
44      protected void processUserDefinitions(Map<String, String> userDefs) {
45          if (userDefs == null || userDefs.isEmpty()) {
46              return;
47          }
48          for (Map.Entry<String, String> entry : userDefs.entrySet()) {
49              String username = entry.getKey();
50              String value = entry.getValue();
51  
52              String[] passwordAndRolesArray = StringUtils.split(value);
53  
54              // the first token is expected to be the password.
55              String password = passwordAndRolesArray[0];
56  
57              SimpleAccount account = getUser(username);
58              if (account == null) {
59                  var pc = new SimplePrincipalCollection();
60                  pc.add(username, getName());
61                  pc.add(5, getName());
62                  pc.add(new PropertyPrincipal(username), getName());
63                  account = new SimpleAccount(pc, password, getName());
64                  add(account);
65              }
66              account.setCredentials(password);
67  
68              if (passwordAndRolesArray.length > 1) {
69                  for (int i = 1; i < passwordAndRolesArray.length; i++) {
70                      String rolename = passwordAndRolesArray[i];
71                      account.addRole(rolename);
72  
73                      SimpleRole role = getRole(rolename);
74                      if (role != null) {
75                          account.addObjectPermissions(role.getPermissions());
76                      }
77                  }
78              } else {
79                  account.setRoles(null);
80              }
81          }
82      }
83  }
