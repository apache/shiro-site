19  package org.apache.shiro.web.jaxrs;
20  
21  import org.apache.shiro.authz.UnauthenticatedException;
22  
23  import javax.ws.rs.core.Application;
24  import javax.ws.rs.core.Feature;
25  import javax.ws.rs.core.FeatureContext;
26  import javax.ws.rs.ext.Provider;
27  
28  
29  /**
30   * Shiro JAX-RS feature which includes {@link UnauthorizedExceptionExceptionMapper}, {@link SubjectPrincipalRequestFilter}, and
31   * {@link ShiroAnnotationFilterFeature}.
32   * <p>
33   * Typically a JAX-RS {@link Application} class will include this Feature class in the
34   * classes returned from {@link Application#getClasses()} method, for example:
35   * <blockquote><pre>
36   *     public class SampleApplication extends Application {
37   *
38   *         @Override
39   *         public Set<Class<?>> getClasses() {
40   *             Set<Class<?>> classes = new HashSet<Class<?>>();
41   *
42   *             // register Shiro
43   *             classes.add(ShiroFeature.class);
44   *             ...
45   *             return classes;
46   *         }
47   *     }
48   * </pre></blockquote>
49   * NOTE: Apache CXF requires this annotation on this feature (jersey and resteasy do not)
50   *
51   * @since 1.4
52   */
53  @Provider
54  public class ShiroFeature implements Feature {
55  
56      @Override
57      public boolean configure(FeatureContext context) {
58  
59          context.register(UnauthorizedExceptionExceptionMapper.class);
60          context.register(UnauthenticatedException.class);
61          context.register(SubjectPrincipalRequestFilter.class);
62          context.register(ShiroAnnotationFilterFeature.class);
63  
64          return true;
65      }
66  }
