49  /**
50   * TypeListener that injects setter methods on Shiro objects.
51   */
52  class BeanTypeListener implements TypeListener {
53      public static final Package SHIRO_GUICE_PACKAGE = ShiroModule.class.getPackage();
54      public static final Package SHIRO_PACKAGE = SecurityUtils.class.getPackage();
55  
56  
57      /**
58       * @since 1.4
59       */
60      public static final String BEAN_TYPE_MAP_NAME = "__SHIRO_BEAN_TYPES__";
61  
62      static final Key<?> MAP_KEY = Key.get(Types.mapOf(TypeLiteral.class, BeanTypeKey.class), Names.named(BEAN_TYPE_MAP_NAME));
63  
64      private static final Matcher<Class> SHIRO_MATCHER = Matchers.inSubpackage(SHIRO_PACKAGE.getName());
65      private static final Matcher<Class> SHIRO_GUICE_MATCHER = Matchers.inSubpackage(SHIRO_GUICE_PACKAGE.getName());
66      private static final Matcher<Class> CLASS_MATCHER = ShiroMatchers.anyPackage
67              .and(SHIRO_MATCHER.and(Matchers.not(SHIRO_GUICE_MATCHER)));
68  
69      static final Matcher<TypeLiteral> MATCHER = ShiroMatchers.typeLiteral(CLASS_MATCHER);
70  
71      private static final Set<Class<?>> WRAPPER_TYPES = new HashSet<Class<?>>(Arrays.asList(
72              Byte.class,
73              Boolean.class,
74              Character.class,
75              Double.class,
76              Float.class,
77              Integer.class,
78              Long.class,
79              Short.class,
80              Void.class));
81  
82      private final BeanUtilsBean beanUtilsBean;
83  
84      BeanTypeListener() {
85          // SHIRO-619
86          beanUtilsBean = new BeanUtilsBean();
87          beanUtilsBean.getPropertyUtils().addBeanIntrospector(
88                  SuppressPropertiesBeanIntrospector.SUPPRESS_CLASS);
89      }
90  
91      public <I> void hear(TypeLiteral<I> type, final TypeEncounter<I> encounter) {
92          PropertyDescriptor[] propertyDescriptors = beanUtilsBean.getPropertyUtils().getPropertyDescriptors(type.getRawType());
93          final Map<PropertyDescriptor, Key<?>> propertyDependencies
94                  = new HashMap<PropertyDescriptor, Key<?>>(propertyDescriptors.length);
95          final Provider<Injector> injectorProvider = encounter.getProvider(Injector.class);
96          for (PropertyDescriptor propertyDescriptor : propertyDescriptors) {
97              if (propertyDescriptor.getWriteMethod() != null
98                      && Modifier.isPublic(propertyDescriptor.getWriteMethod().getModifiers())) {
99                  Type propertyType = propertyDescriptor.getWriteMethod().getGenericParameterTypes()[0];
100                 propertyDependencies.put(propertyDescriptor, createDependencyKey(propertyDescriptor, propertyType));
101             }
102         }
103         encounter.register(new MembersInjector<I>() {
104             public void injectMembers(I instance) {
105                 for (Map.Entry<PropertyDescriptor, Key<?>> dependency : propertyDependencies.entrySet()) {
106                     try {
107                         final Injector injector = injectorProvider.get();
108 
109                         Object value = injector.getInstance(getMappedKey(injector, dependency.getValue()));
110                         dependency.getKey().getWriteMethod().invoke(instance, value);
111 
112                     } catch (ConfigurationException e) {
113                         // This is ok, it simply means that we can't fulfill this dependency.
114                         // Is there a better way to do this?
115                     } catch (InvocationTargetException e) {
116                         throw new RuntimeException("Couldn't set property " + dependency.getKey().getDisplayName(), e);
117                     } catch (IllegalAccessException e) {
118                         throw new RuntimeException("We shouldn't have ever reached this point,"
119                                 + " we don't try to inject to non-accessible methods.", e);
120                     }
121                 }
122 
123             }
124         });
125     }
126 
127     private static Key<?> getMappedKey(Injector injector, Key<?> key) {
128         Map<TypeLiteral, BeanTypeKey> beanTypeMap = getBeanTypeMap(injector);
129         if (key.getAnnotation() == null && beanTypeMap.containsKey(key.getTypeLiteral())) {
130             return beanTypeMap.get(key.getTypeLiteral()).key;
131         } else {
132             return key;
133         }
134     }
135 
136     @SuppressWarnings({"unchecked"})
137     private static Map<TypeLiteral, BeanTypeKey> getBeanTypeMap(Injector injector) {
138         return (Map<TypeLiteral, BeanTypeKey>) injector.getInstance(MAP_KEY);
139     }
140 
141     private static Key<?> createDependencyKey(PropertyDescriptor propertyDescriptor, Type propertyType) {
142         if (requiresName(propertyType)) {
143             return Key.get(propertyType, Names.named("shiro." + propertyDescriptor.getName()));
144         } else {
145             return Key.get(propertyType);
146         }
147     }
148 
149     private static boolean requiresName(Type propertyType) {
150         if (propertyType instanceof Class) {
151             Class<?> aClass = (Class<?>) propertyType;
152             return aClass.isPrimitive()
153                     || aClass.isEnum()
154                     || WRAPPER_TYPES.contains(aClass)
155                     || CharSequence.class.isAssignableFrom(aClass);
156         } else {
157             return false;
158         }
159     }
160 
161     static void ensureBeanTypeMapExists(Binder binder) {
162         beanTypeMapBinding(binder).addBinding(TypeLiteral.get(BeanTypeKey.class)).toInstance(new BeanTypeKey(null));
163     }
164 
165     static <T> void bindBeanType(Binder binder, TypeLiteral<T> typeLiteral, Key<? extends T> key) {
166         beanTypeMapBinding(binder).addBinding(typeLiteral).toInstance(new BeanTypeKey(key));
167     }
168 
169     private static MapBinder<TypeLiteral, BeanTypeKey> beanTypeMapBinding(Binder binder) {
170         return MapBinder.newMapBinder(binder, TypeLiteral.class, BeanTypeKey.class, Names.named(BEAN_TYPE_MAP_NAME));
171     }
172 
173     private static final class BeanTypeKey {
174         Key<?> key;
175 
176         private BeanTypeKey(Key<?> key) {
177             this.key = key;
178         }
179     }
180 }
