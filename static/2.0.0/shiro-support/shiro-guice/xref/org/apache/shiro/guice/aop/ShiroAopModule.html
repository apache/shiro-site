19  package org.apache.shiro.guice.aop;
20  
21  import com.google.inject.AbstractModule;
22  import com.google.inject.matcher.AbstractMatcher;
23  import com.google.inject.matcher.Matchers;
24  import org.apache.shiro.aop.AnnotationMethodInterceptor;
25  import org.apache.shiro.aop.AnnotationResolver;
26  import org.apache.shiro.aop.DefaultAnnotationResolver;
27  import org.apache.shiro.authz.aop.AuthenticatedAnnotationMethodInterceptor;
28  import org.apache.shiro.authz.aop.GuestAnnotationMethodInterceptor;
29  import org.apache.shiro.authz.aop.PermissionAnnotationMethodInterceptor;
30  import org.apache.shiro.authz.aop.RoleAnnotationMethodInterceptor;
31  import org.apache.shiro.authz.aop.UserAnnotationMethodInterceptor;
32  
33  import java.lang.annotation.Annotation;
34  import java.lang.reflect.Method;
35  
36  /**
37   * Install this module to enable Shiro AOP functionality in Guice.  You may extend it to add your own Shiro
38   * interceptors, override the default ones, or provide a specific {@link org.apache.shiro.aop.AnnotationResolver}.
39   */
40  public class ShiroAopModule extends AbstractModule {
41      @Override
42      protected final void configure() {
43          AnnotationResolver resolver = createAnnotationResolver();
44          configureDefaultInterceptors(resolver);
45          configureInterceptors(resolver);
46      }
47  
48      protected final void bindShiroInterceptor(final AnnotationMethodInterceptor methodInterceptor) {
49          bindInterceptor(Matchers.any(), new AbstractMatcher<Method>() {
50              public boolean matches(Method method) {
51                  Class<? extends Annotation> annotation = methodInterceptor.getHandler().getAnnotationClass();
52                  return method.getAnnotation(annotation) != null
53                          || method.getDeclaringClass().getAnnotation(annotation) != null;
54              }
55          }, new AopAllianceMethodInterceptorAdapter(methodInterceptor));
56      }
57  
58      protected AnnotationResolver createAnnotationResolver() {
59          return new DefaultAnnotationResolver();
60      }
61  
62      protected void configureDefaultInterceptors(AnnotationResolver resolver) {
63          bindShiroInterceptor(new RoleAnnotationMethodInterceptor(resolver));
64          bindShiroInterceptor(new PermissionAnnotationMethodInterceptor(resolver));
65          bindShiroInterceptor(new AuthenticatedAnnotationMethodInterceptor(resolver));
66          bindShiroInterceptor(new UserAnnotationMethodInterceptor(resolver));
67          bindShiroInterceptor(new GuestAnnotationMethodInterceptor(resolver));
68      }
69  
70      protected void configureInterceptors(AnnotationResolver resolver) {
71  
72      }
73  }
