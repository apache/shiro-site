19  package org.apache.shiro.spring.boot.autoconfigure;
20  
21  import org.apache.shiro.spring.boot.autoconfigure.exception.NoRealmBeanConfiguredException;
22  import org.springframework.boot.diagnostics.AbstractFailureAnalyzer;
23  import org.springframework.boot.diagnostics.FailureAnalysis;
24  
25  public class ShiroNoRealmConfiguredFailureAnalyzer extends AbstractFailureAnalyzer<NoRealmBeanConfiguredException> {
26  
27      @Override
28      protected FailureAnalysis analyze(Throwable rootFailure, NoRealmBeanConfiguredException cause) {
29          return new FailureAnalysis("No bean of type 'org.apache.shiro.realm.Realm' found.",
30                  "Please create bean of type 'Realm' or add a shiro.ini in the root classpath (src/main/resources/shiro.ini) "
31                          + "or in the META-INF folder (src/main/resources/META-INF/shiro.ini).",
32                  cause);
33      }
34  
35  }
