14  package org.apache.shiro.ee.filters;
15  
16  import static org.apache.shiro.ee.filters.FormAuthenticationFilter.LOGIN_PREDICATE_ATTR_NAME;
17  import static org.apache.shiro.ee.filters.FormAuthenticationFilter.LOGIN_WAITTIME_ATTR_NAME;
18  import static org.apache.shiro.ee.filters.FormResubmitSupport.FORM_IS_RESUBMITTED;
19  import static org.apache.shiro.ee.filters.FormResubmitSupport.SESSION_EXPIRED_PARAMETER;
20  import static org.apache.shiro.ee.filters.LogoutFilter.LOGOUT_PREDICATE_ATTR_NAME;
21  import static org.apache.shiro.ee.listeners.EnvironmentLoaderListener.isFormResubmitDisabled;
22  import java.util.concurrent.TimeUnit;
23  import javax.enterprise.context.ApplicationScoped;
24  import javax.inject.Named;
25  import javax.servlet.http.HttpServletRequest;
26  import javax.servlet.http.HttpServletResponse;
27  import lombok.AccessLevel;
28  import lombok.NoArgsConstructor;
29  import lombok.SneakyThrows;
30  import lombok.extern.slf4j.Slf4j;
31  import org.apache.shiro.SecurityUtils;
32  import org.apache.shiro.authc.AuthenticationException;
33  import org.apache.shiro.authc.UsernamePasswordToken;
34  import static org.apache.shiro.web.filter.authc.FormAuthenticationFilter.DEFAULT_ERROR_KEY_ATTRIBUTE_NAME;
35  import static org.omnifaces.exceptionhandler.ViewExpiredExceptionHandler.wasViewExpired;
36  import org.omnifaces.util.Faces;
37  
38  /**
39   * Methods to redirect to saved requests upon logout
40   * functionality includes saving a previous form state and resubmitting
41   * if the form times out
42   */
43  @Slf4j
44  @NoArgsConstructor(access = AccessLevel.PRIVATE)
45  @SuppressWarnings("HideUtilityClassConstructor")
46  public class Forms {
47      /**
48       * JSF access points
49       */
50      @Named("authc")
51      @ApplicationScoped
52      public static class AuthenticationMethods {
53          /**
54           * let Shiro filter handle the login,
55           * this method should only get called if login fails
56           * login wait time is handled by Shiro configuration
57           */
58          public void login() {
59              if (isLoginFailure()) {
60                  loginFailed();
61              } else if (!redirectIfLoggedIn()) {
62                  throw new IllegalStateException("Not enough context to log in, need username / password");
63              }
64          }
65  
66          /**
67           * manual login, zero wait time
68           *
69           * @param username
70           * @param password
71           */
72          public void login(String username, String password) {
73              login(username, password, false);
74          }
75  
76          /**
77           * manual login with timeout
78           *
79           * @param username
80           * @param password
81           * @param rememberMe
82           */
83          public void login(String username, String password, boolean rememberMe) {
84              Forms.login(username, password, rememberMe);
85          }
86  
87          public void logout() {
88              Forms.logout();
89          }
90  
91          public boolean isLoggedIn() {
92              return Forms.isLoggedIn();
93          }
94  
95          public boolean redirectIfLoggedIn() {
96              return redirectIfLoggedIn("");
97          }
98  
99          public boolean redirectIfLoggedIn(String view) {
100             if (isLoggedIn()) {
101                 redirectToView(Faces.getRequestAttribute(LOGOUT_PREDICATE_ATTR_NAME), view);
102                 return true;
103             } else {
104                 return false;
105             }
106         }
107 
108         public boolean isSessionExpired() {
109             return Forms.isSessionExpired();
110         }
111 
112         public boolean isLoginFailure() {
113             return Faces.getRequestAttribute(DEFAULT_ERROR_KEY_ATTRIBUTE_NAME) != null
114                     || Faces.getFlashAttribute(DEFAULT_ERROR_KEY_ATTRIBUTE_NAME) != null;
115         }
116     }
117 
118     @FunctionalInterface
119     public interface FallbackPredicate {
120         boolean useFallback(String path, HttpServletRequest request);
121     }
122 
123     /**
124      * Jakarta Faces variant
125      * redirect to saved request, possibly resubmitting an existing form
126      * the saved request is via a cookie
127      *
128      * @param useFallbackPath
129      * @param fallbackPath
130      */
131     public static void redirectToSaved(FallbackPredicate useFallbackPath, String fallbackPath) {
132         FormResubmitSupport.redirectToSaved(Faces.getRequest(), Faces.getResponse(), useFallbackPath, fallbackPath,
133                 !isFormResubmitDisabled(Faces.getRequest().getServletContext()));
134     }
135 
136     /**
137      * Jakarta Faces variant
138      * redirects to current view after a form submit, or a logout, for example
139      */
140     public static void redirectToView() {
141         FormResubmitSupport.redirectToView(Faces.getRequest(), Faces.getResponse());
142     }
143 
144     public static void redirectToView(FallbackPredicate useFallbackPath, String fallbackPath) {
145         FormResubmitSupport.redirectToView(Faces.getRequest(), Faces.getResponse(), useFallbackPath, fallbackPath);
146     }
147 
148     /**
149      * manually login, used via {@link PassThruAuthenticationFilter}
150      * @param username
151      * @param password
152      * @param rememberMe
153      */
154     @SneakyThrows(InterruptedException.class)
155     public static void login(String username, String password, boolean rememberMe) {
156         try {
157             SecurityUtils.getSubject().login(new UsernamePasswordToken(username, password, rememberMe));
158             redirectToSaved(Faces.getRequestAttribute(LOGIN_PREDICATE_ATTR_NAME), "");
159         } catch (AuthenticationException e) {
160             Faces.setFlashAttribute(DEFAULT_ERROR_KEY_ATTRIBUTE_NAME, e);
161             int loginFailedWaitTime = Faces.getRequestAttribute(LOGIN_WAITTIME_ATTR_NAME);
162             if (loginFailedWaitTime != 0) {
163                 TimeUnit.SECONDS.sleep(loginFailedWaitTime);
164             }
165             redirectToView();
166         }
167     }
168 
169     /**
170      * JSF login failure method
171      */
172     public static void loginFailed() {
173         Faces.setFlashAttribute(DEFAULT_ERROR_KEY_ATTRIBUTE_NAME, Faces.getRequestAttribute(DEFAULT_ERROR_KEY_ATTRIBUTE_NAME));
174         Faces.removeRequestAttribute(DEFAULT_ERROR_KEY_ATTRIBUTE_NAME);
175         redirectToView();
176     }
177 
178     public static void logout() {
179         Forms.logout(Faces.getRequestAttribute(LOGOUT_PREDICATE_ATTR_NAME), "");
180     }
181 
182     /**
183      * Faces variant
184      * @param useFallback
185      * @param fallbackPath
186      */
187     public static void logout(FallbackPredicate useFallback, String fallbackPath) {
188         logout(Faces.getRequest(), Faces.getResponse(), useFallback, fallbackPath);
189     }
190 
191     /**
192      * makes sure that there is no double-logout
193      *
194      * @param request
195      * @param response
196      * @param useFallback
197      * @param fallbackPath
198      */
199     public static void logout(HttpServletRequest request, HttpServletResponse response,
200             FallbackPredicate useFallback, String fallbackPath) {
201         if (!Boolean.TRUE.toString().equals(request.getHeader(FORM_IS_RESUBMITTED))) {
202             SecurityUtils.getSubject().logout();
203             FormResubmitSupport.redirectToView(request, response, useFallback, fallbackPath);
204         }
205     }
206 
207     public static boolean isLoggedIn() {
208         var subject = SecurityUtils.getSubject();
209         return subject.isAuthenticated() || subject.isRemembered();
210     }
211 
212     public static boolean isSessionExpired() {
213         return wasViewExpired() || Boolean.parseBoolean(Faces.getRequestParameter(SESSION_EXPIRED_PARAMETER));
214     }
215 }
