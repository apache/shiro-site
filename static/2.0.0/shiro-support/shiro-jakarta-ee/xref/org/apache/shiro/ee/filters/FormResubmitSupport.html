139     static void savePostDataForResubmit(HttpServletRequest request, HttpServletResponse response, @NonNull String loginUrl) {
140         if (isPostRequest(request) && unwrapSecurityManager(SecurityUtils.getSecurityManager())
141                 instanceof DefaultSecurityManager) {
142             String postData = getPostData(request);
143             var cacheKey = UUID.randomUUID();
144             var dsm = (DefaultSecurityManager) unwrapSecurityManager(SecurityUtils.getSecurityManager());
145             if (dsm.getCacheManager() != null) {
146                 var cache = dsm.getCacheManager().getCache(FORM_DATA_CACHE);
147                 var rememberMeManager = (AbstractRememberMeManager) dsm.getRememberMeManager();
148                 if (rememberMeManager != null && rememberMeManager.getCipherService() != null) {
149                     cache.put(cacheKey, rememberMeManager.getCipherService()
150                             .encrypt(postData.getBytes(StandardCharsets.UTF_8),
151                                     rememberMeManager.getEncryptionCipherKey()).getBytes());
152                 } else {
153                     log.warn("Post-data was saved in plain text due to rememberMeManager not being available");
154                     cache.put(cacheKey, postData);
155                 }
156                 addCookie(response, request.getServletContext(), SHIRO_FORM_DATA_KEY,
157                         cacheKey.toString(), getCookieAge(request, dsm));
158             } else {
159                 log.warn("Shiro Cache manager is not configured, cannot store form data");
160             }
161         }
162         boolean isFacesGetRequest = HttpMethod.GET.equalsIgnoreCase(request.getMethod());
163         doFacesRedirect(request, response, request.getContextPath() + loginUrl
164                 + (isFacesGetRequest ? "" : "?%s=true"), SESSION_EXPIRED_PARAMETER);
165     }

180     static String getSavedFormDataFromKey(@NonNull String savedFormDataKey) {
181         String savedFormData = null;
182         if (unwrapSecurityManager(SecurityUtils.getSecurityManager()) instanceof DefaultSecurityManager) {
183             var dsm = (DefaultSecurityManager) unwrapSecurityManager(SecurityUtils.getSecurityManager());
184             if (dsm.getCacheManager() != null) {
185                 var cache = dsm.getCacheManager().getCache(FORM_DATA_CACHE);
186                 var cacheKey = UUID.fromString(savedFormDataKey);
187                 var rememberMeManager = (AbstractRememberMeManager) dsm.getRememberMeManager();
188                 if (rememberMeManager != null && rememberMeManager.getCipherService() != null) {
189                     var cachedData = Optional.ofNullable((byte[]) cache.get(cacheKey));
190                     savedFormData = cachedData.map(encryptedData ->
191                             decrypt(encryptedData, rememberMeManager)).orElse(savedFormData);
192                 } else {
193                     savedFormData = (String) cache.get(cacheKey);
194                 }
195                 cache.remove(cacheKey);
196             }
197         }
198         return savedFormData;
199     }

201     static String decrypt(byte[] encrypted, AbstractRememberMeManager rememberMeManager) {
202         return new String(rememberMeManager.getCipherService()
203                 .decrypt(encrypted, rememberMeManager.getDecryptionCipherKey()).getClonedBytes(),
204                 StandardCharsets.UTF_8);
205     }

207     static void saveRequest(HttpServletRequest request, HttpServletResponse response, boolean useReferer) {
208         String path = useReferer ? getReferer(request)
209                 : Servlets.getRequestURLWithQueryString(request);
210         if (path != null) {
211             Servlets.addResponseCookie(request, response, WebUtils.SAVED_REQUEST_KEY,
212                     path, null, request.getContextPath(),
213                     // cookie age = session timeout
214                     getCookieAge(request, SecurityUtils.getSecurityManager()));
215         }
216     }

218     static void saveRequestReferer(boolean rv, HttpServletRequest request, HttpServletResponse response) {
219         if (rv && HttpMethod.GET.equalsIgnoreCase(request.getMethod())) {
220             if (Servlets.getRequestCookie(request, WebUtils.SAVED_REQUEST_KEY) == null) {
221                 // only save refer when there is no saved request cookie already,
222                 // and only as a last resort
223                 saveRequest(request, response, true);
224             }
225         }
226     }

228     static String getReferer(HttpServletRequest request) {
229         String referer = request.getHeader("referer");
230         if (referer != null) {
231             // do not switch to https if custom port is specified
232             if (!referer.matches("^http:\\/\\/[A-z|.|[0-9]]+:[0-9]+(\\/.*|$)")) {
233                 referer = referer.replaceFirst("^http:", "https:");
234             }
235         }
236         return referer;
237     }

241      * Resubmits the form that caused the logout upon successfull login.Form resumnission supports JSF and Ajax forms

243      * @param response

248     @SneakyThrows({IOException.class, InterruptedException.class})
249     static void redirectToSaved(HttpServletRequest request, HttpServletResponse response,
250             FallbackPredicate useFallbackPath, String fallbackPath, boolean resubmit) {
251         String savedRequest = Servlets.getRequestCookie(request, WebUtils.SAVED_REQUEST_KEY);
252         if (savedRequest != null) {
253             doRedirectToSaved(request, response, savedRequest, resubmit);
254         } else {
255             redirectToView(request, response, useFallbackPath, fallbackPath);
256         }
257     }

275     private static void doRedirectToSaved(HttpServletRequest request, HttpServletResponse response,
276             @NonNull String savedRequest, boolean resubmit) throws IOException, InterruptedException {
277         deleteCookie(response, request.getServletContext(), WebUtils.SAVED_REQUEST_KEY);
278         String savedFormDataKey = Servlets.getRequestCookie(request, SHIRO_FORM_DATA_KEY);
279         boolean doRedirectAtEnd = true;
280         if (savedFormDataKey != null && resubmit) {
281             String formData = getSavedFormDataFromKey(savedFormDataKey);
282             if (formData != null) {
283                 Optional.ofNullable(resubmitSavedForm(formData, savedRequest,
284                         request, response, request.getServletContext(), false))
285                         .ifPresent(path -> doFacesRedirect(request, response, path));
286                 doRedirectAtEnd = false;
287             } else {
288                 deleteCookie(response, request.getServletContext(), SHIRO_FORM_DATA_KEY);
289             }
290         }
291         if (doRedirectAtEnd) {
292             doFacesRedirect(request, response, savedRequest);
293         }
294     }

304     /**
305      * redirects to current view after a form submit,
306      * or the fallback path if predicate succeeds
307      *
308      * @param request
309      * @param response
310      * @param useFallbackPath
311      * @param fallbackPath
312      */
313     @SneakyThrows
314     static void redirectToView(HttpServletRequest request, HttpServletResponse response,
315             FallbackPredicate useFallbackPath, String fallbackPath) {
316         boolean useFallback = useFallbackPath.useFallback(request.getRequestURI(), request);
317         String referer = getReferer(request);
318         String redirectPath = Servlets.getRequestURLWithQueryString(request);
319         if (useFallback && referer != null && !isLoginUrl(request)) {
320             // the following is used in the logout flow only,
321             // because login flow saves the request automatically, without
322             // needing a referrer
323             useFallback = useFallbackPath.useFallback(referer, request);
324             redirectPath = referer;
325         }
326         if (useFallback) {
327             doFacesRedirect(request, response, request.getContextPath() + fallbackPath);
328         } else {
329             doFacesRedirect(request, response, redirectPath);
330         }
331     }

363     static String resubmitSavedForm(@NonNull String savedFormData, @NonNull String savedRequest,
364             HttpServletRequest originalRequest, HttpServletResponse originalResponse,
365             ServletContext servletContext, boolean rememberedAjaxResubmit)
366             throws InterruptedException, IOException {
367         if (log.isDebugEnabled()) {
368             log.debug("saved form data: {}", savedFormData);
369             log.debug("Set Cookie Headers: {}", originalResponse.getHeaders(SET_COOKIE));
370             log.debug("Original Request Headers: {}", Collections.list(originalRequest.getHeaderNames()));
371             log.debug("Original Request Cookie Header: {}", Collections.list(originalRequest.getHeaders(COOKIE)));
372         }
373         if (Boolean.TRUE.toString().equals(originalRequest.getHeader(FORM_IS_RESUBMITTED))) {
374             log.debug("Form resubmit: internal auth failure");
375             originalResponse.setStatus(AUTHFAIL);
376             return resubmitResponseCleanup(originalRequest);
377         }
378         URI overriddenRequestURI = overrideSavedRequestURI(URI.create(savedRequest));
379         HttpClient client = buildHttpClient(overriddenRequestURI, servletContext, originalRequest);
380         HttpResponse<String> response;
381         PartialAjaxResult decodedFormData;
382         try {
383             decodedFormData = parseFormData(savedFormData, overriddenRequestURI, client, servletContext);
384             HttpRequest postRequest = constructPostRequest(overriddenRequestURI, decodedFormData.result);
385             response = sendResubmitRequest(client, postRequest);
386         } catch (IOException e) {
387             log.warn("Unable to resubmit form to {}" + System.lineSeparator()
388                     + "perhaps set org.apache.shiro.form-resubmit-host or "
389                     + "org.apache.shiro.form-resubmit-port system property?", overriddenRequestURI, e);
390             return savedRequest;
391         }
392         if (rememberedAjaxResubmit && !decodedFormData.isStatelessRequest) {
393             HttpRequest redirectRequest = constructPostRequest(overriddenRequestURI, savedFormData);
394             var redirectResponse = client.send(redirectRequest, HttpResponse.BodyHandlers.ofString());
395             log.debug("Redirect request: {}, response: {}", redirectRequest, redirectResponse);
396             return processResubmitResponse(redirectResponse, originalRequest, originalResponse,
397                     response.headers(), savedRequest, servletContext, true, rememberedAjaxResubmit);
398         } else {
399             deleteCookie(originalResponse, servletContext, SHIRO_FORM_DATA_KEY);
400             return processResubmitResponse(response, originalRequest, originalResponse,
401                     response.headers(), savedRequest, servletContext,
402                     (rememberedAjaxResubmit && decodedFormData.isStatelessRequest) ? false
403                             : decodedFormData.isPartialAjaxRequest, rememberedAjaxResubmit);
404         }
405     }

407     @SneakyThrows(URISyntaxException.class)
408     private static URI overrideSavedRequestURI(URI savedRequestURI) {
409         if (RESUBMIT_HOST.isPresent() || RESUBMIT_PORT.isPresent()) {
410             var uri = new URI(savedRequestURI.getScheme(), savedRequestURI.getRawUserInfo(),
411                     RESUBMIT_HOST.orElse(savedRequestURI.getHost()), RESUBMIT_PORT.orElse(savedRequestURI.getPort()),
412                     savedRequestURI.getRawPath(), savedRequestURI.getRawQuery(), savedRequestURI.getRawFragment());
413             log.debug("Form Resubmit - Overriding URI {} with {}", savedRequestURI, uri);
414             return uri;
415         } else {
416             return savedRequestURI;
417         }
418     }

420     private static HttpRequest constructPostRequest(URI request, String body) {
421         return HttpRequest.newBuilder().uri(request)
422                 .POST(HttpRequest.BodyPublishers.ofString(body))
423                 .headers(CONTENT_TYPE, APPLICATION_FORM_URLENCODED,
424                         FORM_IS_RESUBMITTED, Boolean.TRUE.toString())
425                 .build();
426     }

428     private static HttpResponse<String>
429     sendResubmitRequest(HttpClient client, HttpRequest request) throws IOException, InterruptedException {
430         HttpResponse<String> response = client.send(request, HttpResponse.BodyHandlers.ofString());
431         if (log.isDebugEnabled()) {
432             log.debug("Resubmit request: {}, response: {}", request, response);
433             log.debug("Response Headers: {}", response.headers().map());
434         }
435         if (response.statusCode() == AUTHFAIL) {
436             log.debug("processing authfail");
437             var cookieManager = (CookieManager) client.cookieHandler().get();
438             cookieStreamFromHeader(response.headers().allValues(SET_COOKIE))
439                     .forEach(cookie -> cookieManager.getCookieStore().add(request.uri(), cookie));
440             response = client.send(request, HttpResponse.BodyHandlers.ofString());
441             if (log.isDebugEnabled()) {
442                 log.debug("Resubmit request(authfail): {}, response: {}", request, response);
443                 log.debug("Response Headers(authfail): {}", response.headers().map());
444             }
445         }
446         return response;
447     }

449     private static PartialAjaxResult parseFormData(String savedFormData, URI savedRequest,
450             HttpClient client, ServletContext servletContext) throws IOException, InterruptedException {
451         boolean isStateless = true;
452         if (!isJSFClientStateSavingMethod(servletContext)) {
453             String decodedFormData = URLDecoder.decode(savedFormData, StandardCharsets.UTF_8);
454             if (isJSFStatefulForm(decodedFormData)) {
455                 isStateless = false;
456                 savedFormData = getJSFNewViewState(savedRequest, client, decodedFormData);
457             }
458         }
459         return noJSFAjaxRequests(savedFormData, isStateless);
460     }

462     @SuppressWarnings("fallthrough")
463     private static String processResubmitResponse(HttpResponse<String> response,
464             HttpServletRequest originalRequest, HttpServletResponse originalResponse,
465             HttpHeaders headers, String savedRequest, ServletContext servletContext,
466             boolean isPartialAjaxRequest, boolean rememberedAjaxResubmit) throws IOException {
467         switch (response.statusCode()) {
468             case FOUND:
469                 if (rememberedAjaxResubmit) {
470                     originalResponse.setStatus(OK);
471                 } else {
472                     // can't use Faces.redirect() here
473                     originalResponse.setStatus(response.statusCode());
474                     originalResponse.setHeader(LOCATION, response.headers().firstValue(LOCATION).orElseThrow());
475                 }
476             case OK:
477                 // do not duplicate the session cookie(s)
478                 transformCookieHeader(headers.allValues(SET_COOKIE))
479                         .entrySet().stream().filter(not(entry -> entry.getKey()
480                                 .startsWith(getSessionCookieName(servletContext, SecurityUtils.getSecurityManager()))))
481                         .forEach(entry -> addCookie(originalResponse, servletContext,
482                                 entry.getKey(), entry.getValue(), -1));
483                 if (isPartialAjaxRequest) {
484                     originalResponse.setHeader(CONTENT_TYPE, TEXT_XML);
485                     originalResponse.setCharacterEncoding(StandardCharsets.UTF_8.name());
486                     originalResponse.getWriter().append(String.format(
487                             "<partial-response><redirect url=\"%s\"></redirect></partial-response>",
488                             savedRequest));
489                 } else {
490                     originalResponse.getWriter().append(response.body());
491                 }
492                 return resubmitResponseCleanup(originalRequest);
493             default:
494                 return savedRequest;
495         }
496     }

498     private static String resubmitResponseCleanup(HttpServletRequest originalRequest) {
499         originalRequest.setAttribute(DONT_ADD_ANY_MORE_COOKIES, Boolean.TRUE);
500         if (hasFacesContext()) {
501             Faces.responseComplete();
502         }
503         return null;
504     }

506     private static HttpClient buildHttpClient(URI savedRequest, ServletContext servletContext,
507             HttpServletRequest originalRequest) {
508         CookieManager cookieManager = new CookieManager();
509         var session = SecurityUtils.getSubject().getSession();
510         var sessionCookieName = getSessionCookieName(servletContext, SecurityUtils.getSecurityManager());
511         var sessionCookie = new HttpCookie(sessionCookieName, session.getId().toString());
512         sessionCookie.setPath(servletContext.getContextPath());
513         sessionCookie.setVersion(0);
514         cookieManager.getCookieStore().add(savedRequest, sessionCookie);
515         log.debug("Setting Cookie {}", sessionCookieName);
516         for (Cookie origCookie : originalRequest.getCookies()) {
517             if (!origCookie.getName().equals(sessionCookieName)) {
518                 try {
519                     log.debug("Setting Cookie {}", origCookie.getName());
520                     HttpCookie cookie = new HttpCookie(origCookie.getName(), origCookie.getValue());
521                     cookie.setPath(servletContext.getContextPath());
522                     cookie.setVersion(0);
523                     cookieManager.getCookieStore().add(savedRequest, cookie);
524                 } catch (IllegalArgumentException e) {
525                     log.warn("Form Resubmit: Ignoring invalid cookie [{} - {}]",
526                             origCookie.getName(), origCookie.getValue(), e);
527                 }
528             }
529         }
530         return HttpClient.newBuilder().cookieHandler(cookieManager).build();
531     }

533     public static DefaultWebSessionManager getNativeSessionManager(SecurityManager securityManager) {
534         DefaultWebSessionManager rv = null;
535         SecurityManager unwrapped = unwrapSecurityManager(securityManager);
536         if (unwrapped instanceof SessionsSecurityManager) {
537             var ssm = (SessionsSecurityManager) unwrapped;
538             var sm = ssm.getSessionManager();
539             if (sm instanceof DefaultWebSessionManager) {
540                 rv = (DefaultWebSessionManager) sm;
541             }
542         }
543         return rv;

555     private static String getJSFNewViewState(URI savedRequest, HttpClient client, String savedFormData)
556             throws IOException, InterruptedException {
557         var getRequest = HttpRequest.newBuilder().uri(savedRequest).GET().build();
558         HttpResponse<String> htmlResponse = sendResubmitRequest(client, getRequest);
559         if (htmlResponse.statusCode() == OK) {
560             savedFormData = extractJSFNewViewState(htmlResponse.body(), savedFormData);
561         }
562         return savedFormData;
563     }

565     static String extractJSFNewViewState(@NonNull String responseBody, @NonNull String savedFormData) {
566         Elements elts = Jsoup.parse(responseBody).select(String.format("input[name=%s]", FACES_VIEW_STATE));
567         if (!elts.isEmpty()) {
568             String viewState = elts.first().attr("value");
569 
570             var matcher = VIEW_STATE_PATTERN.matcher(savedFormData);
571             if (matcher.matches()) {
572                 savedFormData = matcher.replaceFirst(String.format("$1%s%s$3",
573                         FACES_VIEW_STATE_EQUALS, viewState));
574                 log.debug("Encoded w/Replaced ViewState: {}", savedFormData);
575             }
576         }
577         return savedFormData;
578     }
