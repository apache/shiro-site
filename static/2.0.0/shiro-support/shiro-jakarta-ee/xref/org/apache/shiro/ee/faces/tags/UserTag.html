14  package org.apache.shiro.ee.faces.tags;
15  
16  import javax.faces.view.facelets.TagConfig;
17  
18  /**
19   * Tag that renders the tag body if the current user known to the system, either from a successful login attempt
20   * (not necessarily during the current session) or from 'RememberMe' services.
21   * <p>
22   * <p><b>Note:</b> This is <em>less</em> restrictive than the <code>AuthenticatedTag</code> since it only assumes
23   * the user is who they say they are, either via a current session login <em>or</em> via Remember Me services, which
24   * makes no guarantee the user is who they say they are.  The <code>AuthenticatedTag</code> however
25   * guarantees that the current user has logged in <em>during their current session</em>, proving they really are
26   * who they say they are.
27   * <p>
28   * <p>The logically opposite tag of this one is the {@link GuestTag}.
29   */
30  public class UserTag extends GuestTag {
31      public UserTag(TagConfig config) {
32          super(config);
33      }
34  
35      @Override
36      protected boolean checkAuthentication() {
37          return !super.checkAuthentication();
38      }
39  }
