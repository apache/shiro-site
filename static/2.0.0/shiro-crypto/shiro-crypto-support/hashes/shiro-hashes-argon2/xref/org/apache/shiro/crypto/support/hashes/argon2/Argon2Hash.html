44  /**
45   * The Argon2 key derivation function (KDF) is a modern algorithm to shade and hash passwords.
46   *
47   * <p>The default implementation ({@code argon2id}) is designed to use both memory and cpu to make
48   * brute force attacks unfeasible.</p>
49   *
50   * <p>The defaults are taken from
51   * <a href="https://argon2-cffi.readthedocs.io/en/stable/parameters.html">argon2-cffi.readthedocs.io</a>.
52   * The RFC suggests to use 1 GiB of memory for frontend and 4 GiB for backend authentication.</p>
53   *
54   * <p>Example crypt string is:
55   *      {@code $argon2i$v=19$m=16384,t=100,p=2$M3ByeyZKLjFRREJqQi87WQ$5kRCtDjL6RoIWGq9bL27DkFNunucg1hW280PmP0XDtY}.</p>
56   *
57   * <p>Default values are taken from
58   *      <a href="https://datatracker.ietf.org/doc/draft-irtf-cfrg-argon2/?include_text=1">draft-irtf-cfrg-argon2-13</a>.
59   * This implementation is using the parameters from section 4, paragraph 2 (memory constrained environment).</p>
60   *
61   * @since 2.0
62   */
63  class Argon2Hash extends AbstractCryptHash {
64  
65      /**
66       * Number of default lanes, p=4 is the default recommendation, taken from draft-irtf-cfrg-argon2-13, 4.2.
67       */
68      public static final int DEFAULT_PARALLELISM = 4;
69  
70      /**
71       * 256 bits tag size is the default recommendation, taken from draft-irtf-cfrg-argon2-13, 4.2.
72       */
73      public static final int DEFAULT_OUTPUT_LENGTH_BITS = 256;
74  
75      public static final String DEFAULT_ALGORITHM_NAME = "argon2id";
76  
77      public static final int DEFAULT_ALGORITHM_VERSION = Argon2Parameters.ARGON2_VERSION_13;
78  
79      /**
80       * Number of iterations, default taken from draft-irtf-cfrg-argon2-13, 4.2.
81       */
82      public static final int DEFAULT_ITERATIONS = 1;
83  
84      /**
85       * Amount of memory, default (64 MiB) taken from draft-irtf-cfrg-argon2-13, 4.2.
86       */
87      public static final int DEFAULT_MEMORY_KIB = 64 * 1024;

89      private static final long serialVersionUID = 2647354947284558921L;

91      private static final Logger LOG = LoggerFactory.getLogger(Argon2Hash.class);

93      private static final Set<String> ALGORITHMS_ARGON2 = new HashSet<>(Arrays.asList("argon2id", "argon2i", "argon2d"));
94  
95      private static final Pattern DELIMITER_COMMA = Pattern.compile(",");

97      /**
98       * 128 bits of salt is the recommended salt length, taken from draft-irtf-cfrg-argon2-13, 4.2.
99       */
100     private static final int SALT_LENGTH_BITS = 128;
101 
102     private final int argonVersion;
103 
104     private final int iterations;
105 
106     private final int memoryKiB;
107 
108     private final int parallelism;

110     Argon2Hash(String algorithmName, int argonVersion, byte[] hashedData,
111                ByteSource salt, int iterations, int memoryAsKB, int parallelism) {
112         super(algorithmName, hashedData, salt);
113         this.argonVersion = argonVersion;
114         this.iterations = iterations;
115         this.memoryKiB = memoryAsKB;
116         this.parallelism = parallelism;
117 
118         checkValidIterations();
119     }

121     public static Set<String> getAlgorithmsArgon2() {
122         return unmodifiableSet(ALGORITHMS_ARGON2);
123     }

125     protected static ByteSource createSalt() {
126         return createSalt(new SecureRandom());
127     }

129     @SuppressWarnings("checkstyle:MagicNumber")
130     public static ByteSource createSalt(SecureRandom random) {
131         return new SimpleByteSource(random.generateSeed(SALT_LENGTH_BITS / 8));
132     }

134     public static Argon2Hash fromString(String input) {
135         // expected:
136         // $argon2i$v=19$m=4096,t=3,p=4$M3ByeyZKLjFRREJqQi87WQ$5kRCtDjL6RoIWGq9bL27DkFNunucg1hW280PmP0XDtY
137         if (!input.startsWith("$")) {
138             throw new UnsupportedOperationException("Unsupported input: " + input);
139         }
140 
141         final String[] parts = AbstractCryptHash.DELIMITER.split(input.substring(1));
142         final String algorithmName = parts[0].trim();
143 
144         if (!ALGORITHMS_ARGON2.contains(algorithmName)) {
145             throw new UnsupportedOperationException("Unsupported algorithm: " + algorithmName
146                     + ". Expected one of: " + ALGORITHMS_ARGON2);
147         }
148 
149         final int version = parseVersion(parts[1]);
150         final String parameters = parts[2];
151         final int memoryPowTwo = parseMemory(parameters);
152         final int iterations = parseIterations(parameters);
153         final int parallelism = parseParallelism(parameters);
154         final ByteSource salt = new SimpleByteSource(Base64.decode(parts[3]));
155         final byte[] hashedData = Base64.decode(parts[4]);
156 
157         return new Argon2Hash(algorithmName, version, hashedData, salt, iterations, memoryPowTwo, parallelism);
158     }
159 
160     private static int parseParallelism(String parameters) {
161         String parameter = DELIMITER_COMMA.splitAsStream(parameters)
162                 .filter(parm -> parm.startsWith("p="))
163                 .findAny()
164                 .orElseThrow(() -> new IllegalArgumentException("Did not found memory parameter 'p='. Got: ["
165                         + parameters + "]."));
166         return Integer.parseInt(parameter.substring(2));
167     }
168 
169     private static int parseIterations(String parameters) {
170         String parameter = DELIMITER_COMMA.splitAsStream(parameters)
171                 .filter(parm -> parm.startsWith("t="))
172                 .findAny()
173                 .orElseThrow(() -> new IllegalArgumentException("Did not found memory parameter 't='. Got: ["
174                         + parameters + "]."));
175 
176         return Integer.parseInt(parameter.substring(2));
177     }
178 
179     private static int parseMemory(String parameters) {
180         String parameter = DELIMITER_COMMA.splitAsStream(parameters)
181                 .filter(parm -> parm.startsWith("m="))
182                 .findAny()
183                 .orElseThrow(() -> new IllegalArgumentException("Did not found memory parameter 'm='. Got: ["
184                         + parameters + "]."));
185 
186         return Integer.parseInt(parameter.substring(2));
187     }
188 
189     private static int parseVersion(final String part) {
190         if (!part.startsWith("v=")) {
191             throw new IllegalArgumentException("Did not find version parameter 'v='. Got: [" + part + "].");
192         }
193 
194         return Integer.parseInt(part.substring(2));
195     }

197     public static Argon2Hash generate(final char[] source) {
198         return generate(new SimpleByteSource(source), createSalt(), DEFAULT_ITERATIONS);
199     }

201     public static Argon2Hash generate(final ByteSource source, final ByteSource salt, final int iterations) {
202         return generate(DEFAULT_ALGORITHM_NAME, source, requireNonNull(salt, "salt"), iterations);
203     }
204 
205     public static Argon2Hash generate(String algorithmName, ByteSource source, ByteSource salt, int iterations) {
206         return generate(algorithmName, DEFAULT_ALGORITHM_VERSION, source, salt, iterations,
207                 DEFAULT_MEMORY_KIB, DEFAULT_PARALLELISM, DEFAULT_OUTPUT_LENGTH_BITS);
208     }
209 
210     public static Argon2Hash generate(
211             String algorithmName,
212             int argonVersion,
213             ByteSource source,
214             ByteSource salt,
215             int iterations,
216             int memoryAsKB,
217             int parallelism,
218             int outputLengthBits
219     ) {
220         final int type;
221         switch (requireNonNull(algorithmName, "algorithmName")) {
222             case "argon2i":
223                 type = Argon2Parameters.ARGON2_i;
224                 break;
225             case "argon2d":
226                 type = Argon2Parameters.ARGON2_d;
227                 break;
228             case "argon2":
229                 // fall through
230             case "argon2id":
231                 type = Argon2Parameters.ARGON2_id;
232                 break;
233             default:
234                 throw new IllegalArgumentException("Unknown argon2 algorithm: " + algorithmName);
235         }
236 
237         final Argon2Parameters parameters = new Argon2Parameters.Builder(type)
238                 .withVersion(argonVersion)
239                 .withIterations(iterations)
240                 .withParallelism(parallelism)
241                 .withSalt(requireNonNull(salt, "salt").getBytes())
242                 .withMemoryAsKB(memoryAsKB)
243                 .build();
244 
245         final Argon2BytesGenerator gen = new Argon2BytesGenerator();
246         gen.init(parameters);
247 
248         @SuppressWarnings("checkstyle:MagicNumber") final byte[] hash = new byte[outputLengthBits / 8];
249         gen.generateBytes(source.getBytes(), hash);
250 
251         return new Argon2Hash(algorithmName, argonVersion, hash, new SimpleByteSource(salt), iterations, memoryAsKB, parallelism);
252     }

254     @Override
255     protected void checkValidAlgorithm() {
256         if (!ALGORITHMS_ARGON2.contains(getAlgorithmName())) {
257             final String message = String.format(
258                     Locale.ENGLISH,
259                     "Given algorithm name [%s] not valid for argon2. "
260                             + "Valid algorithms: [%s].",
261                     getAlgorithmName(),
262                     ALGORITHMS_ARGON2
263             );
264             throw new IllegalArgumentException(message);
265         }
266     }
267 
268     protected void checkValidIterations() {
269         int iterations = this.getIterations();
270         if (iterations < 1) {
271             final String message = String.format(
272                     Locale.ENGLISH,
273                     "Expected argon2 iterations >= 1, but was [%d].",
274                     iterations
275             );
276             throw new IllegalArgumentException(message);
277         }
278     }

280     @Override
281     public int getIterations() {
282         return this.iterations;
283     }

285     @Override
286     public boolean matchesPassword(ByteSource plaintextBytes) {
287         try {
288             @SuppressWarnings("checkstyle:MagicNumber")
289             Argon2Hash compare = generate(
290                     this.getAlgorithmName(),
291                     this.argonVersion,
292                     plaintextBytes,
293                     this.getSalt(),
294                     this.getIterations(),
295                     this.memoryKiB,
296                     this.parallelism,
297                     this.getBytes().length * 8);
298 
299             return this.equals(compare);
300         } catch (IllegalArgumentException illegalArgumentException) {
301             // cannot recreate hash. Do not log password.
302             LOG.warn("Cannot recreate a hash using the same parameters.", illegalArgumentException);
303             return false;
304         }
305     }

309     public int getSaltLength() {
310         return SALT_LENGTH_BITS / 8;

313     @Override
314     public String formatToCryptString() {
315         // expected:
316         // $argon2i$v=19$m=4096,t=3,p=4$M3ByeyZKLjFRREJqQi87WQ$5kRCtDjL6RoIWGq9bL27DkFNunucg1hW280PmP0XDtY
317         Encoder encoder = java.util.Base64.getEncoder().withoutPadding();
318         String saltBase64 = encoder.encodeToString(this.getSalt().getBytes());
319         String dataBase64 = encoder.encodeToString(this.getBytes());
320 
321         return new StringJoiner("$", "$", "")
322                 .add(this.getAlgorithmName())
323                 .add("v=" + this.argonVersion)
324                 .add(formatParameters())
325                 .add(saltBase64)
326                 .add(dataBase64)
327                 .toString();
328     }
329 
330     private CharSequence formatParameters() {
331         return String.format(
332                 Locale.ENGLISH,
333                 "t=%d,m=%d,p=%d",
334                 getIterations(),
335                 getMemoryKiB(),
336                 getParallelism()
337         );
338     }

352     @Override
353     public boolean equals(Object other) {
354         if (this == other) {
355             return true;
356         }
357         if (other == null || getClass() != other.getClass()) {
358             return false;
359         }
360         if (!super.equals(other)) {
361             return false;
362         }
363         Argon2Hash that = (Argon2Hash) other;
364         return argonVersion == that.argonVersion && iterations == that.iterations
365                     && memoryKiB == that.memoryKiB && parallelism == that.parallelism;
366     }

369     public int hashCode() {
370         return Objects.hash(super.hashCode(), argonVersion, iterations, memoryKiB, parallelism);
371     }

377                 .add("version=" + argonVersion)
378                 .add("iterations=" + iterations)
379                 .add("memoryKiB=" + memoryKiB)
380                 .add("parallelism=" + parallelism)
381                 .toString();
382     }
