38  /**
39   * A class representing the <a href="http://en.wikipedia.org/wiki/INI_file">INI</a> text configuration format.
40   * <p/>
41   * An Ini instance is a map of {@link Ini.Section Section}s, keyed by section name.  Each
42   * {@code Section} is itself a map of {@code String} name/value pairs.  Name/value pairs are guaranteed to be unique
43   * within each {@code Section} only - not across the entire {@code Ini} instance.
44   *
45   * @since 1.0
46   */
47  @SuppressWarnings({"checkstyle:MethodCount", "checkstyle:CyclomaticComplexity"})
48  public final class Ini implements Map<String, Ini.Section> {

51       * empty string means the first unnamed section

92      /**
93       * Creates a new {@code Ini} instance with the specified defaults.
94       *
95       * @param defaults the default sections and/or key-value pairs to copy into the new instance.
96       */
97      public Ini(Ini defaults) {
98          this();
99          if (defaults == null) {
100             throw new NullPointerException("Defaults cannot be null.");
101         }
102         for (Section section : defaults.getSections()) {
103             Section copy = new Section(section);
104             this.sections.put(section.getName(), copy);
105         }
106     }

108     /**
109      * Returns {@code true} if no sections have been configured, or if there are sections, but the sections themselves
110      * are all empty, {@code false} otherwise.
111      *
112      * @return {@code true} if no sections have been configured, or if there are sections, but the sections themselves
113      * are all empty, {@code false} otherwise.
114      */
115     public boolean isEmpty() {
116         Collection<Section> sections = this.sections.values();
117         if (!sections.isEmpty()) {
118             for (Section section : sections) {
119                 if (!section.isEmpty()) {
120                     return false;
121                 }
122             }
123         }
124         return true;
125     }

161      * Ensures a section with the specified name exists, adding a new one if it does not yet exist.
162      *
163      * @param sectionName the name of the section to ensure existence
164      * @return the section created if it did not yet exist, or the existing Section that already existed.
165      */
166     public Section addSection(String sectionName) {
167         String name = cleanName(sectionName);
168         Section section = getSection(name);
169         if (section == null) {
170             section = new Section(name);
171             this.sections.put(name, section);
172         }
173         return section;
174     }

187     private static String cleanName(String sectionName) {
188         String name = StringUtils.clean(sectionName);
189         if (name == null) {
190             LOGGER.trace("Specified name was null or empty.  Defaulting to the default section (name = \"\")");
191             name = DEFAULT_SECTION_NAME;
192         }
193         return name;
194     }

196     /**
197      * Sets a name/value pair for the section with the given {@code sectionName}.  If the section does not yet exist,
198      * it will be created.  If the {@code sectionName} is null or empty, the name/value pair will be placed in the
199      * default (unnamed, empty string) section.
200      *
201      * @param sectionName   the name of the section to add the name/value pair
202      * @param propertyName  the name of the property to add
203      * @param propertyValue the property value
204      */
205     public void setSectionProperty(String sectionName, String propertyName, String propertyValue) {
206         String name = cleanName(sectionName);
207         Section section = getSection(name);
208         if (section == null) {
209             section = addSection(name);
210         }
211         section.put(propertyName, propertyValue);
212     }

241     /**
242      * Creates a new {@code Ini} instance loaded with the INI-formatted data in the resource at the given path.  The
243      * resource path may be any value interpretable by the
244      * {@link ResourceUtils#getInputStreamForPath(String) ResourceUtils.getInputStreamForPath} method.
245      *
246      * @param resourcePath the resource location of the INI data to load when creating the {@code Ini} instance.
247      * @return a new {@code Ini} instance loaded with the INI-formatted data in the resource at the given path.
248      * @throws ConfigurationException if the path cannot be loaded into an {@code Ini} instance.
249      */
250     public static Ini fromResourcePath(String resourcePath) throws ConfigurationException {
251         if (!StringUtils.hasLength(resourcePath)) {
252             throw new IllegalArgumentException("Resource Path argument cannot be null or empty.");
253         }
254         Ini ini = new Ini();
255         ini.loadFromPath(resourcePath);
256         return ini;
257     }

259     /**
260      * Loads data from the specified resource path into this current {@code Ini} instance.  The
261      * resource path may be any value interpretable by the
262      * {@link ResourceUtils#getInputStreamForPath(String) ResourceUtils.getInputStreamForPath} method.
263      *
264      * @param resourcePath the resource location of the INI data to load into this instance.
265      * @throws ConfigurationException if the path cannot be loaded
266      */
267     public void loadFromPath(String resourcePath) throws ConfigurationException {
268         InputStream is;
269         try {
270             is = ResourceUtils.getInputStreamForPath(resourcePath);
271         } catch (IOException e) {
272             throw new ConfigurationException(e);
273         }
274         load(is);
275     }

287     /**
288      * Loads the INI-formatted text backed by the given InputStream into this instance.  This implementation will
289      * close the input stream after it has finished loading.  It is expected that the stream's contents are
290      * UTF-8 encoded.
291      *
292      * @param is the {@code InputStream} from which to read the INI-formatted text
293      * @throws ConfigurationException if unable
294      */
295     public void load(InputStream is) throws ConfigurationException {
296         if (is == null) {
297             throw new NullPointerException("InputStream argument cannot be null.");
298         }
299         InputStreamReader isr;
300         try {
301             isr = new InputStreamReader(is, DEFAULT_CHARSET_NAME);
302         } catch (UnsupportedEncodingException e) {
303             throw new ConfigurationException(e);
304         }
305         load(isr);
306     }
307 
308     /**
309      * Loads the INI-formatted text backed by the given Reader into this instance.  This implementation will close the
310      * reader after it has finished loading.
311      *
312      * @param reader the {@code Reader} from which to read the INI-formatted text
313      */
314     public void load(Reader reader) {
315         Scanner scanner = new Scanner(reader);
316         try {
317             load(scanner);
318         } finally {
319             try {
320                 scanner.close();
321             } catch (Exception e) {
322                 LOGGER.debug("Unable to cleanly close the InputStream scanner.  Non-critical - ignoring.", e);
323             }
324         }
325     }

327     /**
328      * Merges the contents of <code>m</code>'s {@link Section} objects into self.
329      * This differs from {@link Ini#putAll(Map)}, in that each section is merged with the existing one.
330      * For example the following two ini blocks are merged and the result is the third<BR/>
331      * <p>
332      * Initial:
333      * <pre>
334      * <code>[section1]
335      * key1 = value1
336      *
337      * [section2]
338      * key2 = value2
339      * </code> </pre>
340      * <p>
341      * To be merged:
342      * <pre>
343      * <code>[section1]
344      * foo = bar
345      *
346      * [section2]
347      * key2 = new value
348      * </code> </pre>
349      * <p>
350      * Result:
351      * <pre>
352      * <code>[section1]
353      * key1 = value1
354      * foo = bar
355      *
356      * [section2]
357      * key2 = new value
358      * </code> </pre>
359      *
360      * </p>
361      *
362      * @param m map to be merged
363      * @since 1.4
364      */
365     public void merge(Map<String, Section> m) {
366 
367         if (m != null) {
368             for (Entry<String, Section> entry : m.entrySet()) {
369                 Section section = this.getSection(entry.getKey());
370                 if (section == null) {
371                     section = addSection(entry.getKey());
372                 }
373                 section.putAll(entry.getValue());
374             }
375         }
376     }
377 
378     private void addSection(String name, StringBuilder content) {
379         if (content.length() > 0) {
380             String contentString = content.toString();
381             String cleaned = StringUtils.clean(contentString);
382             if (cleaned != null) {
383                 Section section = new Section(name, contentString);
384                 if (!section.isEmpty()) {
385                     sections.put(name, section);
386                 }
387             }
388         }
389     }

391     /**
392      * Loads the INI-formatted text backed by the given Scanner.  This implementation will close the
393      * scanner after it has finished loading.
394      *
395      * @param scanner the {@code Scanner} from which to read the INI-formatted text
396      */
397     public void load(Scanner scanner) {
398 
399         String sectionName = DEFAULT_SECTION_NAME;
400         StringBuilder sectionContent = new StringBuilder();
401 
402         while (scanner.hasNextLine()) {
403 
404             String rawLine = scanner.nextLine();
405             String line = StringUtils.clean(rawLine);
406 
407             if (line == null || line.startsWith(COMMENT_POUND) || line.startsWith(COMMENT_SEMICOLON)) {
408                 //skip empty lines and comments:
409                 continue;
410             }
411 
412             String newSectionName = getSectionName(line);
413             if (newSectionName != null) {
414                 //found a new section - convert the currently buffered one into a Section object
415                 addSection(sectionName, sectionContent);
416 
417                 //reset the buffer for the new section:
418                 sectionContent = new StringBuilder();
419 
420                 sectionName = newSectionName;
421 
422                 if (LOGGER.isDebugEnabled()) {
423                     LOGGER.debug("Parsing " + SECTION_PREFIX + sectionName + SECTION_SUFFIX);
424                 }
425             } else {
426                 //normal line - add it to the existing content buffer:
427                 sectionContent.append(rawLine).append("\n");
428             }
429         }
430 
431         //finish any remaining buffered content:
432         addSection(sectionName, sectionContent);
433     }

435     protected static boolean isSectionHeader(String line) {
436         String s = StringUtils.clean(line);
437         return s != null && s.startsWith(SECTION_PREFIX) && s.endsWith(SECTION_SUFFIX);
438     }

440     protected static String getSectionName(String line) {
441         String s = StringUtils.clean(line);
442         if (isSectionHeader(s)) {
443             return cleanName(s.substring(1, s.length() - 1));
444         }
445         return null;
446     }

461     public String toString() {
462         if (this.sections == null || this.sections.isEmpty()) {
463             return "<empty INI>";
464         } else {
465             StringBuilder sb = new StringBuilder("sections=");
466             int i = 0;
467             for (Ini.Section section : this.sections.values()) {
468                 if (i > 0) {
469                     sb.append(",");
470                 }
471                 sb.append(section.toString());
472                 i++;
473             }
474             return sb.toString();
475         }

522     /**
523      * An {@code Ini.Section} is String-key-to-String-value Map, identifiable by a
524      * {@link #getName() name} unique within an {@link Ini} instance.
525      */
526     public static final class Section implements Map<String, String> {
527         private final String name;
528         private final Map<String, String> props;
529 
530         private Section(String name) {
531             if (name == null) {
532                 throw new NullPointerException("name");
533             }
534             this.name = name;
535             this.props = new LinkedHashMap<String, String>();
536         }
537 
538         private Section(String name, String sectionContent) {
539             if (name == null) {
540                 throw new NullPointerException("name");
541             }
542             this.name = name;
543             Map<String, String> props;
544             if (StringUtils.hasText(sectionContent)) {
545                 props = toMapProps(sectionContent);
546             } else {
547                 props = new LinkedHashMap<>();
548             }
549             this.props = props;
550         }
551 
552         private Section(Section defaults) {
553             this(defaults.getName());
554             putAll(defaults.props);
555         }

557         //Protected to access in a test case - NOT considered part of Shiro's public API
558 
559         protected static boolean isContinued(String line) {
560             if (!StringUtils.hasText(line)) {
561                 return false;
562             }
563             int length = line.length();
564             //find the number of backslashes at the end of the line.  If an even number, the
565             //backslashes are considered escaped.  If an odd number, the line is considered continued on the next line
566             int backslashCount = 0;
567             for (int i = length - 1; i > 0; i--) {
568                 if (line.charAt(i) == ESCAPE_TOKEN) {
569                     backslashCount++;
570                 } else {
571                     break;
572                 }
573             }
574             return backslashCount % 2 != 0;
575         }
576 
577         private static boolean isKeyValueSeparatorChar(char c) {
578             return Character.isWhitespace(c) || c == ':' || c == '=';
579         }
580 
581         private static boolean isCharEscaped(CharSequence s, int index) {
582             return index > 0 && s.charAt(index) == ESCAPE_TOKEN;
583         }
584 
585         //Protected to access in a test case - NOT considered part of Shiro's public API
586         protected static String[] splitKeyValue(String keyValueLine) {
587             String line = StringUtils.clean(keyValueLine);
588             if (line == null) {
589                 return null;
590             }
591             StringBuilder keyBuffer = new StringBuilder();
592             StringBuilder valueBuffer = new StringBuilder();
593 
594             //we'll build the value next:
595             boolean buildingKey = true;
596 
597             for (int i = 0; i < line.length(); i++) {
598                 char c = line.charAt(i);
599 
600                 if (buildingKey) {
601                     if (isKeyValueSeparatorChar(c) && !isCharEscaped(line, i) && !isCharEscaped(line, i - 1)) {
602                         //now start building the value
603                         buildingKey = false;
604                     } else if (!isCharEscaped(line, i)) {
605                         keyBuffer.append(c);
606                     }
607                 } else {
608                     if (valueBuffer.length() != 0 || !isKeyValueSeparatorChar(c) || isCharEscaped(line, i)) {
609                         valueBuffer.append(c);
610                     }
611                 }
612             }
613 
614             String key = StringUtils.clean(keyBuffer.toString());
615             String value = StringUtils.clean(valueBuffer.toString());
616 
617             if (key == null || value == null) {
618                 String msg = "Line argument must contain a key and a value.  Only one string token was found.";
619                 throw new IllegalArgumentException(msg);
620             }
621 
622             LOGGER.trace("Discovered key/value pair: {} = {}", key, value);
623 
624             return new String[] {key, value};
625         }
626 
627         private static Map<String, String> toMapProps(String content) {
628             Map<String, String> props = new LinkedHashMap<String, String>();
629             String line;
630             StringBuilder lineBuffer = new StringBuilder();
631             Scanner scanner = new Scanner(content);
632             while (scanner.hasNextLine()) {
633                 line = StringUtils.clean(scanner.nextLine());
634                 if (isContinued(line)) {
635                     //strip off the last continuation backslash:
636                     line = line.substring(0, line.length() - 1);
637                     lineBuffer.append(line);
638                     continue;
639                 } else {
640                     lineBuffer.append(line);
641                 }
642                 line = lineBuffer.toString();
643                 lineBuffer = new StringBuilder();
644                 String[] kvPair = splitKeyValue(line);
645                 props.put(kvPair[0], kvPair[1]);
646             }
647 
648             return props;
649         }
