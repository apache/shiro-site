19  package org.apache.shiro.samples.spring;
20  
21  import org.apache.shiro.crypto.hash.Sha256Hash;
22  import org.slf4j.Logger;
23  import org.slf4j.LoggerFactory;
24  import org.springframework.beans.factory.InitializingBean;
25  import org.springframework.jdbc.core.JdbcTemplate;
26  
27  import javax.sql.DataSource;
28  
29  /**
30   * A data populator that creates a set of security tables and test data that can be used by the
31   * Shiro Spring sample application to demonstrate the use of the {@link org.apache.shiro.realm.jdbc.JdbcRealm}
32   * The tables created by this class follow the default table and column names that
33   * {@link org.apache.shiro.realm.jdbc.JdbcRealm} uses.
34   */
35  public class BootstrapDataPopulator implements InitializingBean {
36  
37      private static final String CREATE_TABLES = "create table users (\n"
38              + "    username varchar(255) primary key,\n"
39              + "    password varchar(255) not null\n"
40              + ");\n"
41              + "\n"
42              + "create table roles (\n"
43              + "    role_name varchar(255) primary key\n"
44              + ");\n"
45              + "\n"
46              + "create table user_roles (\n"
47              + "    username varchar(255) not null,\n"
48              + "    role_name varchar(255) not null,\n"
49              + "    constraint user_roles_uq unique ( username, role_name )\n"
50              + ");\n"
51              + "\n"
52              + "create table roles_permissions (\n"
53              + "    role_name varchar(255) not null,\n"
54              + "    permission varchar(255) not null,\n"
55              + "    primary key (role_name, permission)\n"
56              + ");";
57  
58      private static final Logger LOGGER = LoggerFactory.getLogger(BootstrapDataPopulator.class);
59  
60      protected DataSource dataSource;
61  
62      public void setDataSource(DataSource dataSource) {
63          this.dataSource = dataSource;
64      }
65  
66      public void afterPropertiesSet() throws Exception {
67          //because we're using an in-memory hsqldb for the sample app, a new one will be created each time the
68          //app starts, so create the tables and insert the 2 sample users on bootstrap:
69  
70          JdbcTemplate jdbcTemplate = new JdbcTemplate(this.dataSource);
71          jdbcTemplate.execute(CREATE_TABLES);
72  
73          //password is 'user1' SHA hashed and base64 encoded:
74          //The first argument to the hash constructor is the actual value to be hashed.  The 2nd is the
75          //salt.  In this simple demo scenario, the username and the password are the same, but to clarify the
76          //distinction, you would see this in practice:
77          //new Sha256Hash( <password>, <cryptographically strong randomly generated salt> (not the username!) )
78          String query = "insert into users values ('user1', '" + new Sha256Hash("user1", "user1").toBase64() + "' )";
79          jdbcTemplate.execute(query);
80          LOGGER.debug("Created user1.");
81  
82          //password is 'user2' SHA hashed and base64 encoded:
83          query = "insert into users values ( 'user2', '" + new Sha256Hash("user2", "user2").toBase64() + "' )";
84          jdbcTemplate.execute(query);
85          LOGGER.debug("Created user2.");
86  
87          query = "insert into roles values ( 'role1' )";
88          jdbcTemplate.execute(query);
89          LOGGER.debug("Created role1");
90  
91          query = "insert into roles values ( 'role2' )";
92          jdbcTemplate.execute(query);
93          LOGGER.debug("Created role2");
94  
95          query = "insert into roles_permissions values ( 'role1', 'permission1')";
96          jdbcTemplate.execute(query);
97          LOGGER.debug("Created permission 1 for role 1");
98  
99          query = "insert into roles_permissions values ( 'role1', 'permission2')";
100         jdbcTemplate.execute(query);
101         LOGGER.debug("Created permission 2 for role 1");
102 
103         query = "insert into roles_permissions values ( 'role2', 'permission1')";
104         jdbcTemplate.execute(query);
105         LOGGER.debug("Created permission 1 for role 2");
106 
107         query = "insert into user_roles values ( 'user1', 'role1' )";
108         jdbcTemplate.execute(query);
109         query = "insert into user_roles values ( 'user1', 'role2' )";
110         jdbcTemplate.execute(query);
111         LOGGER.debug("Assigned user1 roles role1 and role2");
112 
113         query = "insert into user_roles values ( 'user2', 'role2' )";
114         jdbcTemplate.execute(query);
115         LOGGER.debug("Assigned user2 role role2");
116     }
117 
118 }
