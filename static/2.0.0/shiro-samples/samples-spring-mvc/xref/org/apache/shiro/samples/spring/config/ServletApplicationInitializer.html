19  package org.apache.shiro.samples.spring.config;
20  
21  import org.springframework.web.WebApplicationInitializer;
22  import org.springframework.web.context.ContextLoaderListener;
23  import org.springframework.web.context.support.AnnotationConfigWebApplicationContext;
24  import org.springframework.web.filter.DelegatingFilterProxy;
25  import org.springframework.web.servlet.DispatcherServlet;
26  
27  import javax.servlet.DispatcherType;
28  import javax.servlet.FilterRegistration;
29  import javax.servlet.ServletContext;
30  import javax.servlet.ServletRegistration;
31  import java.util.EnumSet;
32  
33  /**
34   * Initializes Spring Environment without the need for a web.xml
35   */
36  public class ServletApplicationInitializer implements WebApplicationInitializer {
37  
38      @Override
39      public void onStartup(ServletContext container) {
40  
41          //now add the annotations
42          AnnotationConfigWebApplicationContext appContext = getContext();
43  
44          // Manage the lifecycle of the root application context
45          container.addListener(new ContextLoaderListener(appContext));
46  
47          FilterRegistration.Dynamic shiroFilter = container.addFilter("shiroFilterFactoryBean", DelegatingFilterProxy.class);
48          shiroFilter.setInitParameter("targetFilterLifecycle", "true");
49          shiroFilter.addMappingForUrlPatterns(EnumSet.allOf(DispatcherType.class), false, "/*");
50  
51  
52          ServletRegistration.Dynamic remotingDispatcher = container.addServlet("remoting", new DispatcherServlet(appContext));
53          remotingDispatcher.setLoadOnStartup(1);
54          remotingDispatcher.addMapping("/remoting/*");
55  
56  
57          ServletRegistration.Dynamic dispatcher = container.addServlet("DispatcherServlet", new DispatcherServlet(appContext));
58          dispatcher.setLoadOnStartup(1);
59          dispatcher.addMapping("/");
60  
61      }
62  
63      private AnnotationConfigWebApplicationContext getContext() {
64          AnnotationConfigWebApplicationContext context = new AnnotationConfigWebApplicationContext();
65          context.setConfigLocation(getClass().getPackage().getName());
66          return context;
67      }
68  
69  }
