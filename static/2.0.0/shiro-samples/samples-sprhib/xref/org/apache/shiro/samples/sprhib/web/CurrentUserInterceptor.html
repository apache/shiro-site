19  package org.apache.shiro.samples.sprhib.web;
20  
21  import org.apache.shiro.samples.sprhib.model.User;
22  import org.apache.shiro.samples.sprhib.service.UserService;
23  import org.springframework.beans.factory.annotation.Autowired;
24  import org.springframework.stereotype.Component;
25  import org.springframework.web.servlet.HandlerInterceptor;
26  import org.springframework.web.servlet.ModelAndView;
27  
28  import javax.servlet.http.HttpServletRequest;
29  import javax.servlet.http.HttpServletResponse;
30  
31  /**
32   * A Spring MVC interceptor that adds the currentUser into the request as a request attribute
33   * before the JSP is rendered.  This operation is assumed to be fast because the User should be
34   * cached in the Hibernate second-level cache.
35   */
36  @Component
37  public class CurrentUserInterceptor implements HandlerInterceptor {
38  
39      private UserService userService;
40  
41      @Autowired
42      public void setUserService(UserService userService) {
43          this.userService = userService;
44      }
45  
46      @Override
47      public void postHandle(HttpServletRequest httpServletRequest,
48                             HttpServletResponse httpServletResponse,
49                             Object obj, ModelAndView modelAndView)
50              throws Exception {
51          // Add the current user into the request
52          User currentUser = userService.getCurrentUser();
53          if (currentUser != null) {
54              httpServletRequest.setAttribute("currentUser", currentUser);
55          }
56      }
57  }
