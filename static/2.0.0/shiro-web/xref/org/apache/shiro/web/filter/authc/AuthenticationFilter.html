19  package org.apache.shiro.web.filter.authc;
20  
21  import org.apache.shiro.subject.Subject;
22  import org.apache.shiro.web.filter.AccessControlFilter;
23  import org.apache.shiro.web.util.WebUtils;
24  
25  import javax.servlet.ServletRequest;
26  import javax.servlet.ServletResponse;
27  
28  /**
29   * Base class for all Filters that require the current user to be authenticated. This class encapsulates the
30   * logic of checking whether a user is already authenticated in the system while subclasses are required to perform
31   * specific logic for unauthenticated requests.
32   *
33   * @since 0.9
34   */
35  public abstract class AuthenticationFilter extends AccessControlFilter {
36  
37      @SuppressWarnings("checkstyle:JavadocVariable")
38      public static final String DEFAULT_SUCCESS_URL = "/";
39  
40      private String successUrl = DEFAULT_SUCCESS_URL;
41  
42      /**
43       * Returns the success url to use as the default location a user is sent after logging in.  Typically a redirect
44       * after login will redirect to the originally request URL; this property is provided mainly as a fallback in case
45       * the original request URL is not available or not specified.
46       * <p/>
47       * The default value is {@link #DEFAULT_SUCCESS_URL}.
48       *
49       * @return the success url to use as the default location a user is sent after logging in.
50       */
51      public String getSuccessUrl() {
52          return successUrl;
53      }
54  
55      /**
56       * Sets the default/fallback success url to use as the default location a user is sent after logging in.  Typically
57       * a redirect after login will redirect to the originally request URL; this property is provided mainly as a
58       * fallback in case the original request URL is not available or not specified.
59       * <p/>
60       * The default value is {@link #DEFAULT_SUCCESS_URL}.
61       *
62       * @param successUrl the success URL to redirect the user to after a successful login.
63       */
64      public void setSuccessUrl(String successUrl) {
65          this.successUrl = successUrl;
66      }
67  
68  
69      /**
70       * Determines whether the current subject is authenticated.
71       * <p/>
72       * The default implementation {@link #getSubject(javax.servlet.ServletRequest, javax.servlet.ServletResponse) acquires}
73       * the currently executing Subject and then returns
74       * {@link org.apache.shiro.subject.Subject#isAuthenticated() subject.isAuthenticated()};
75       *
76       * @return true if the subject is authenticated; false if the subject is unauthenticated
77       */
78      protected boolean isAccessAllowed(ServletRequest request, ServletResponse response, Object mappedValue) {
79          Subject subject = getSubject(request, response);
80          return subject.isAuthenticated() && subject.getPrincipal() != null;
81      }
82  
83      /**
84       * Redirects to user to the previously attempted URL after a successful login.  This implementation simply calls
85       * <code>{@link org.apache.shiro.web.util.WebUtils WebUtils}.
86       * {@link WebUtils#redirectToSavedRequest(ServletRequest, ServletResponse, String) redirectToSavedRequest}</code>
87       * using the {@link #getSuccessUrl() successUrl} as the {@code fallbackUrl} argument to that call.
88       *
89       * @param request  the incoming request
90       * @param response the outgoing response
91       * @throws Exception if there is a problem redirecting.
92       */
93      protected void issueSuccessRedirect(ServletRequest request, ServletResponse response) throws Exception {
94          WebUtils.redirectToSavedRequest(request, response, getSuccessUrl());
95      }
96  
97  }
