19  package org.apache.shiro.web.filter.authz;
20  
21  import org.apache.shiro.lang.util.StringUtils;
22  
23  import javax.servlet.ServletRequest;
24  import javax.servlet.ServletResponse;
25  import java.util.regex.Pattern;
26  import java.util.Map;
27  
28  /**
29   * A Filter that can allow or deny access based on the host that sent the request.
30   *
31   * <b>WARNING:</b> NOT YET FULLY IMPLEMENTED!!!  Work in progress.
32   *
33   * @since 1.0
34   */
35  @SuppressWarnings("checkstyle:JavadocVariable")
36  public class HostFilter extends AuthorizationFilter {
37  
38      /**
39       * IPV4 quad regex.
40       */
41      public static final String IPV4_QUAD_REGEX = "(?:[0-9]|[1-9][0-9]|1[0-9][0-9]|2(?:[0-4][0-9]|5[0-5]))";
42  
43      /**
44       * IPV4 regex.
45       */
46      public static final String IPV4_REGEX = "(?:" + IPV4_QUAD_REGEX + "\\.){3}" + IPV4_QUAD_REGEX + "$";
47  
48      /**
49       * IPV4 pattern.
50       */
51      public static final Pattern IPV4_PATTERN = Pattern.compile(IPV4_REGEX);
52  
53      public static final String PRIVATE_CLASS_B_SUBSET = "(?:1[6-9]|2[0-9]|3[0-1])";
54  
55      public static final String PRIVATE_CLASS_A_REGEX = "10\\.(?:" + IPV4_QUAD_REGEX + "\\.){2}" + IPV4_QUAD_REGEX + "$";
56  
57      public static final String PRIVATE_CLASS_B_REGEX =
58              "172\\." + PRIVATE_CLASS_B_SUBSET + "\\." + IPV4_QUAD_REGEX + "\\." + IPV4_QUAD_REGEX + "$";
59  
60      public static final String PRIVATE_CLASS_C_REGEX = "192\\.168\\." + IPV4_QUAD_REGEX + "\\." + IPV4_QUAD_REGEX + "$";
61  
62      //user-configured IP (which can be wildcarded) to constructed regex mapping
63      Map<String, String> authorizedIps;
64      Map<String, String> deniedIps;
65      Map<String, String> authorizedHostnames;
66      Map<String, String> deniedHostnames;
67  
68  
69      public void setAuthorizedHosts(String authorizedHosts) {
70          if (!StringUtils.hasText(authorizedHosts)) {
71              throw new IllegalArgumentException("authorizedHosts argument cannot be null or empty.");
72          }
73          String[] hosts = StringUtils.tokenizeToStringArray(authorizedHosts, ", \t");
74  
75          for (String host : hosts) {
76              //replace any periods with \\. to ensure the regex works:
77              String periodsReplaced = host.replace(".", "\\.");
78              //check for IPv4:
79              String wildcardsReplaced = periodsReplaced.replace("*", IPV4_QUAD_REGEX);
80  
81              if (IPV4_PATTERN.matcher(wildcardsReplaced).matches()) {
82                  authorizedIps.put(host, wildcardsReplaced);
83              }
84          }
85  
86      }
87  
88      public void setDeniedHosts(String deniedHosts) {
89          if (!StringUtils.hasText(deniedHosts)) {
90              throw new IllegalArgumentException("deniedHosts argument cannot be null or empty.");
91          }
92      }
93  
94      protected boolean isIpv4Candidate(String host) {
95          String[] quads = StringUtils.tokenizeToStringArray(host, ".");
96          if (quads == null || quads.length != 4) {
97              return false;
98          }
99          for (String quad : quads) {
100             if (!quad.equals("*")) {
101                 try {
102                     Integer.parseInt(quad);
103                 } catch (NumberFormatException nfe) {
104                     return false;
105                 }
106             }
107         }
108         return true;
109     }
110 
111     protected boolean isAccessAllowed(ServletRequest request, ServletResponse response, Object mappedValue) throws Exception {
112         throw new UnsupportedOperationException("Not yet fully implemented!!!");
113     }
114 }
