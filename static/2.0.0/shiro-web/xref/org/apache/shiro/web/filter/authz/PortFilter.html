19  package org.apache.shiro.web.filter.authz;
20  
21  import org.apache.shiro.config.ConfigurationException;
22  import org.apache.shiro.lang.util.StringUtils;
23  import org.apache.shiro.web.util.WebUtils;
24  
25  import javax.servlet.ServletRequest;
26  import javax.servlet.ServletResponse;
27  import javax.servlet.http.HttpServletRequest;
28  import java.io.IOException;
29  
30  /**
31   * A Filter that requires the request to be on a specific port, and if not, redirects to the same URL on that port.
32   * <p/>
33   * Example config:
34   * <pre>
35   * [filters]
36   * port.port = 80
37   * <p/>
38   * [urls]
39   * /some/path/** = port
40   * # override for just this path:
41   * /another/path/** = port[8080]
42   * </pre>
43   *
44   * @since 1.0
45   */
46  @SuppressWarnings("checkstyle:JavadocVariable")
47  public class PortFilter extends AuthorizationFilter {
48  
49      public static final int DEFAULT_HTTP_PORT = 80;
50      public static final String HTTP_SCHEME = "http";
51  
52      private int port = DEFAULT_HTTP_PORT;
53  
54      public int getPort() {
55          return port;
56      }
57  
58      public void setPort(int port) {
59          this.port = port;
60      }
61  
62      protected int toPort(Object mappedValue) {
63          String[] ports = (String[]) mappedValue;
64          if (ports == null || ports.length == 0) {
65              return getPort();
66          }
67          if (ports.length > 1) {
68              throw new ConfigurationException("PortFilter can only be configured with a single port.  You have "
69                      + "configured " + ports.length + ": " + StringUtils.toString(ports));
70          }
71          return Integer.parseInt(ports[0]);
72      }
73  
74      protected boolean isAccessAllowed(ServletRequest request, ServletResponse response, Object mappedValue) throws Exception {
75          int requiredPort = toPort(mappedValue);
76          int requestPort = request.getServerPort();
77          return requiredPort == requestPort;
78      }
79  
80      protected String getScheme(String requestScheme, int port) {
81          if (port == DEFAULT_HTTP_PORT) {
82              return HTTP_SCHEME;
83          } else if (port == SslFilter.DEFAULT_HTTPS_PORT) {
84              return SslFilter.HTTPS_SCHEME;
85          } else {
86              return requestScheme;
87          }
88      }
89  
90      /**
91       * Redirects the request to the same exact incoming URL, but with the port listed in the filter's configuration.
92       *
93       * @param request     the incoming <code>ServletRequest</code>
94       * @param response    the outgoing <code>ServletResponse</code>
95       * @param mappedValue the config specified for the filter in the matching request's filter chain.
96       * @return {@code false} always to force a redirect.
97       */
98      @Override
99      protected boolean onAccessDenied(ServletRequest request, ServletResponse response, Object mappedValue) throws IOException {
100 
101         //just redirect to the specified port:
102         int port = toPort(mappedValue);
103 
104         String scheme = getScheme(request.getScheme(), port);
105 
106         StringBuilder sb = new StringBuilder();
107         sb.append(scheme).append("://");
108         sb.append(request.getServerName());
109         if (port != DEFAULT_HTTP_PORT && port != SslFilter.DEFAULT_HTTPS_PORT) {
110             sb.append(":");
111             sb.append(port);
112         }
113         if (request instanceof HttpServletRequest) {
114             sb.append(WebUtils.toHttp(request).getRequestURI());
115             String query = WebUtils.toHttp(request).getQueryString();
116             if (query != null) {
117                 sb.append("?").append(query);
118             }
119         }
120 
121         WebUtils.issueRedirect(request, response, sb.toString());
122 
123         return false;
124     }
125 }
