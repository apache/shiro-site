19  package org.apache.shiro.web.filter.mgt;
20  
21  import org.apache.shiro.lang.util.StringUtils;
22  import org.apache.shiro.web.servlet.ProxiedFilterChain;
23  
24  import javax.servlet.Filter;
25  import javax.servlet.FilterChain;
26  import java.util.ArrayList;
27  import java.util.Collection;
28  import java.util.Iterator;
29  import java.util.List;
30  import java.util.ListIterator;
31  
32  /**
33   * Simple {@code NamedFilterList} implementation that is supported by a backing {@link List} instance and a simple
34   * {@link #getName() name} property. All {@link List} method implementations are immediately delegated to the
35   * wrapped backing list.
36   *
37   * @since 1.0
38   */
39  public class SimpleNamedFilterList implements NamedFilterList {
40  
41      private String name;
42      private List<Filter> backingList;
43  
44      /**
45       * Creates a new {@code SimpleNamedFilterList} instance with the specified {@code name}, defaulting to a new
46       * {@link ArrayList ArrayList} instance as the backing list.
47       *
48       * @param name the name to assign to this instance.
49       * @throws IllegalArgumentException if {@code name} is null or empty.
50       */
51      public SimpleNamedFilterList(String name) {
52          this(name, new ArrayList<Filter>());
53      }
54  
55      /**
56       * Creates a new {@code SimpleNamedFilterList} instance with the specified {@code name} and {@code backingList}.
57       *
58       * @param name        the name to assign to this instance.
59       * @param backingList the list instance used to back all of this class's {@link List} method implementations.
60       * @throws IllegalArgumentException if {@code name} is null or empty.
61       * @throws NullPointerException     if the backing list is {@code null}.
62       */
63      public SimpleNamedFilterList(String name, List<Filter> backingList) {
64          if (backingList == null) {
65              throw new NullPointerException("backingList constructor argument cannot be null.");
66          }
67          this.backingList = backingList;
68          setName(name);
69      }
70  
71      protected void setName(String name) {
72          if (!StringUtils.hasText(name)) {
73              throw new IllegalArgumentException("Cannot specify a null or empty name.");
74          }
75          this.name = name;
76      }
77  
78      public String getName() {
79          return name;
80      }
81  
82      public FilterChain proxy(FilterChain orig) {
83          return new ProxiedFilterChain(orig, this);
84      }
85  
86      public boolean add(Filter filter) {
87          return this.backingList.add(filter);
88      }
89  
90      public void add(int index, Filter filter) {
91          this.backingList.add(index, filter);
92      }
93  
94      public boolean addAll(Collection<? extends Filter> c) {
95          return this.backingList.addAll(c);
96      }
97  
98      public boolean addAll(int index, Collection<? extends Filter> c) {
99          return this.backingList.addAll(index, c);
100     }
101 
102     public void clear() {
103         this.backingList.clear();
104     }
105 
106     public boolean contains(Object o) {
107         return this.backingList.contains(o);
108     }
109 
110     public boolean containsAll(Collection<?> c) {
111         return this.backingList.containsAll(c);
112     }
113 
114     public Filter get(int index) {
115         return this.backingList.get(index);
116     }
117 
118     public int indexOf(Object o) {
119         return this.backingList.indexOf(o);
120     }
121 
122     public boolean isEmpty() {
123         return this.backingList.isEmpty();
124     }
125 
126     public Iterator<Filter> iterator() {
127         return this.backingList.iterator();
128     }
129 
130     public int lastIndexOf(Object o) {
131         return this.backingList.lastIndexOf(o);
132     }
133 
134     public ListIterator<Filter> listIterator() {
135         return this.backingList.listIterator();
136     }
137 
138     public ListIterator<Filter> listIterator(int index) {
139         return this.backingList.listIterator(index);
140     }
141 
142     public Filter remove(int index) {
143         return this.backingList.remove(index);
144     }
145 
146     public boolean remove(Object o) {
147         return this.backingList.remove(o);
148     }
149 
150     public boolean removeAll(Collection<?> c) {
151         return this.backingList.removeAll(c);
152     }
153 
154     public boolean retainAll(Collection<?> c) {
155         return this.backingList.retainAll(c);
156     }
157 
158     public Filter set(int index, Filter filter) {
159         return this.backingList.set(index, filter);
160     }
161 
162     public int size() {
163         return this.backingList.size();
164     }
165 
166     public List<Filter> subList(int fromIndex, int toIndex) {
167         return this.backingList.subList(fromIndex, toIndex);
168     }
169 
170     public Object[] toArray() {
171         return this.backingList.toArray();
172     }
173 
174     public <T> T[] toArray(T[] a) {
175         //noinspection SuspiciousToArrayCall
176         return this.backingList.toArray(a);
177     }
178 }
