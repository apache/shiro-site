19  package org.apache.shiro.web.filter.authc;
20  
21  import org.apache.shiro.subject.Subject;
22  
23  import javax.servlet.ServletRequest;
24  import javax.servlet.ServletResponse;
25  
26  /**
27   * An authentication filter that redirects the user to the login page when they are trying to access
28   * a protected resource.  However, if the user is trying to access the login page, the filter lets
29   * the request pass through to the application code.
30   * <p/>
31   * The difference between this filter and the {@link FormAuthenticationFilter FormAuthenticationFilter} is that
32   * on a login submission (by default an HTTP POST to the login URL), the <code>FormAuthenticationFilter</code> filter
33   * attempts to automatically authenticate the user by passing the <code>username</code> and <code>password</code>
34   * request parameter values to
35   * {@link org.apache.shiro.subject.Subject#login(org.apache.shiro.authc.AuthenticationToken) Subject.login(usernamePasswordToken)}
36   * directly.
37   * <p/>
38   * Conversely, this controller always passes all requests to the {@link #setLoginUrl loginUrl} through, both GETs and
39   * POSTs.  This is useful in cases where the developer wants to write their own login behavior, which should include a
40   * call to {@link Subject#login(org.apache.shiro.authc.AuthenticationToken) Subject.login(AuthenticationToken)}
41   * at some point.  For example,  if the developer has their own custom MVC login controller or validator,
42   * this <code>PassThruAuthenticationFilter</code> may be appropriate.
43   *
44   * @see FormAuthenticationFilter
45   * @since 0.9
46   */
47  public class PassThruAuthenticationFilter extends AuthenticationFilter {
48  
49      //TODO - complete JavaDoc
50  
51      protected boolean onAccessDenied(ServletRequest request, ServletResponse response) throws Exception {
52          if (isLoginRequest(request, response)) {
53              return true;
54          } else {
55              saveRequestAndRedirectToLogin(request, response);
56              return false;
57          }
58      }
59  
60  }
