34  /**
35   * Requires the requesting user to be {@link org.apache.shiro.subject.Subject#isAuthenticated() authenticated} for the
36   * request to continue, and if they're not, requires the user to login via the HTTP "Authentication"
37   * header (e.g. BASIC, Bearer, etc.) Upon successful login, they're allowed to continue on to the requested resource/url.
38   * <p/>
39   * The {@link #onAccessDenied(ServletRequest, ServletResponse)} method will
40   * only be called if the subject making the request is not
41   * {@link org.apache.shiro.subject.Subject#isAuthenticated() authenticated}
42   *
43   * @see <a href="https://tools.ietf.org/html/rfc2617">RFC 2617</a>
44   * @see <a href="http://en.wikipedia.org/wiki/Basic_access_authentication">Basic Access Authentication</a>
45   * @see <a href="https://tools.ietf.org/html/rfc6750#section-2.1">OAuth2 Authorization Request Header Field</a>
46   * @since 1.5
47   */
48  abstract class HttpAuthenticationFilter extends AuthenticatingFilter {

166     /**
167      * The Basic authentication filter can be configured with a list of HTTP methods to which it should apply. This
168      * method ensures that authentication is <em>only</em> required for those HTTP methods specified. For example,
169      * if you had the configuration:
170      * <pre>
171      *    [urls]
172      *    /basic/** = authcBasic[POST,PUT,DELETE]
173      * </pre>
174      * then a GET request would not required authentication but a POST would.
175      *
176      * @param request     The current HTTP servlet request.
177      * @param response    The current HTTP servlet response.
178      * @param mappedValue The array of configured HTTP methods as strings. This is empty if no methods are configured.
179      */
180     protected boolean isAccessAllowed(ServletRequest request, ServletResponse response, Object mappedValue) {
181         HttpServletRequest httpRequest = WebUtils.toHttp(request);
182         String httpMethod = httpRequest.getMethod();
183 
184         // Check whether the current request's method requires authentication.
185         // If no methods have been configured, then all of them require auth,
186         // otherwise only the declared ones need authentication.
187 
188         Set<String> methods = httpMethodsFromOptions((String[]) mappedValue);
189         boolean authcRequired = methods.size() == 0;
190         for (String m : methods) {
191             // list of methods is in upper case
192             if (httpMethod.toUpperCase(Locale.ENGLISH).equals(m)) {
193                 authcRequired = true;
194                 break;
195             }
196         }
197 
198         if (authcRequired) {
199             return super.isAccessAllowed(request, response, mappedValue);
200         } else {
201             return true;
202         }
203     }
204 
205     private Set<String> httpMethodsFromOptions(String[] options) {
206         Set<String> methods = new HashSet<String>();
207 
208         if (options != null) {
209             for (String option : options) {
210                 // to be backwards compatible with 1.3, we can ONLY check for known args
211                 // ideally we would just validate HTTP methods, but someone could already be using this for webdav
212                 if (!option.equalsIgnoreCase(PERMISSIVE)) {
213                     methods.add(option.toUpperCase(Locale.ENGLISH));
214                 }
215             }
216         }
217         return methods;
218     }

220     /**
221      * Processes unauthenticated requests. It handles the two-stage request/challenge authentication protocol.
222      *
223      * @param request  incoming ServletRequest
224      * @param response outgoing ServletResponse
225      * @return true if the request should be processed; false if the request should not continue to be processed
226      */
227     protected boolean onAccessDenied(ServletRequest request, ServletResponse response) throws Exception {
228         //false by default or we wouldn't be in this method
229         boolean loggedIn = false;
230         if (isLoginAttempt(request, response)) {
231             loggedIn = executeLogin(request, response);
232         }
233         if (!loggedIn) {
234             sendChallenge(request, response);
235         }
236         return loggedIn;
237     }

239     /**
240      * Determines whether the incoming request is an attempt to log in.
241      * <p/>
242      * The default implementation obtains the value of the request's
243      * {@link #AUTHORIZATION_HEADER AUTHORIZATION_HEADER}, and if it is not <code>null</code>, delegates
244      * to {@link #isLoginAttempt(String) isLoginAttempt(authzHeaderValue)}. If the header is <code>null</code>,
245      * <code>false</code> is returned.
246      *
247      * @param request  incoming ServletRequest
248      * @param response outgoing ServletResponse
249      * @return true if the incoming request is an attempt to log in based, false otherwise
250      */
251     protected boolean isLoginAttempt(ServletRequest request, ServletResponse response) {
252         String authzHeader = getAuthzHeader(request);
253         return authzHeader != null && isLoginAttempt(authzHeader);
254     }

264     /**
265      * Returns the {@link #AUTHORIZATION_HEADER AUTHORIZATION_HEADER} from the specified ServletRequest.
266      * <p/>
267      * This implementation merely casts the request to an <code>HttpServletRequest</code> and returns the header:
268      * <p/>
269      * <code>HttpServletRequest httpRequest = {@link WebUtils#toHttp(ServletRequest) toHttp(request)};<br/>
270      * return httpRequest.getHeader({@link #AUTHORIZATION_HEADER AUTHORIZATION_HEADER});</code>
271      *
272      * @param request the incoming <code>ServletRequest</code>
273      * @return the <code>Authorization</code> header's value.
274      */
275     protected String getAuthzHeader(ServletRequest request) {
276         HttpServletRequest httpRequest = WebUtils.toHttp(request);
277         return httpRequest.getHeader(AUTHORIZATION_HEADER);
278     }

280     /**
281      * Default implementation that returns <code>true</code> if the specified <code>authzHeader</code>
282      * starts with the same (case-insensitive) characters specified by the
283      * {@link #getAuthzScheme() authzScheme}, <code>false</code> otherwise.
284      * <p/>
285      * That is:
286      * <p/>
287      * <code>String authzScheme = getAuthzScheme().toLowerCase();<br/>
288      * return authzHeader.toLowerCase().startsWith(authzScheme);</code>
289      *
290      * @param authzHeader the 'Authorization' header value (guaranteed to be non-null if the
291      *                    {@link #isLoginAttempt(ServletRequest, ServletResponse)} method is not overridden).
292      * @return <code>true</code> if the authzHeader value matches that configured as defined by
293      * the {@link #getAuthzScheme() authzScheme}.
294      */
295     protected boolean isLoginAttempt(String authzHeader) {
296         //SHIRO-415: use English Locale:
297         String authzScheme = getAuthzScheme().toLowerCase(Locale.ENGLISH);
298         return authzHeader.toLowerCase(Locale.ENGLISH).startsWith(authzScheme);
299     }

301     /**
302      * Builds the challenge for authorization by setting a HTTP <code>401</code> (Unauthorized) status as well as the
303      * response's {@link #AUTHENTICATE_HEADER AUTHENTICATE_HEADER}.
304      * <p/>
305      * The header value constructed is equal to:
306      * <p/>
307      * <code>{@link #getAuthcScheme() getAuthcScheme()} + " realm=\""
308      *      + {@link #getApplicationName() getApplicationName()} + "\"";</code>
309      *
310      * @param request  incoming ServletRequest, ignored by this implementation
311      * @param response outgoing ServletResponse
312      * @return false - this sends the challenge to be sent back
313      */
314     protected boolean sendChallenge(ServletRequest request, ServletResponse response) {
315         LOGGER.debug("Authentication required: sending 401 Authentication challenge response.");
316 
317         HttpServletResponse httpResponse = WebUtils.toHttp(response);
318         httpResponse.setStatus(HttpServletResponse.SC_UNAUTHORIZED);
319         String authcHeader = getAuthcScheme() + " realm=\"" + getApplicationName() + "\"";
320         httpResponse.setHeader(AUTHENTICATE_HEADER, authcHeader);
321         return false;
322     }

324     /**
325      * Creates an AuthenticationToken for use during login attempt with the provided credentials in the http header.
326      * <p/>
327      * This implementation:
328      * <ol><li>acquires the username and password based on the request's
329      * {@link #getAuthzHeader(ServletRequest) authorization header} via the
330      * {@link #getPrincipalsAndCredentials(String, ServletRequest) getPrincipalsAndCredentials} method</li>
331      * <li>The return value of that method is converted to an <code>AuthenticationToken</code> via the
332      * {@link #createToken(String, String, ServletRequest, ServletResponse) createToken} method</li>
333      * <li>The created <code>AuthenticationToken</code> is returned.</li>
334      * </ol>
335      *
336      * @param request  incoming ServletRequest
337      * @param response outgoing ServletResponse
338      * @return the AuthenticationToken used to execute the login attempt
339      */
340     protected AuthenticationToken createToken(ServletRequest request, ServletResponse response) {
341         String authorizationHeader = getAuthzHeader(request);
342         if (authorizationHeader == null || authorizationHeader.length() == 0) {
343             // Create an empty authentication token since there is no
344             // Authorization header.
345             return createToken("", "", request, response);
346         }
347 
348         LOGGER.debug("Attempting to execute login with auth header");
349 
350         String[] prinCred = getPrincipalsAndCredentials(authorizationHeader, request);
351         if (prinCred == null || prinCred.length < 2) {
352             // Create an authentication token with an empty password,
353             // since one hasn't been provided in the request.
354             String username = prinCred == null || prinCred.length == 0 ? "" : prinCred[0];
355             return createToken(username, "", request, response);
356         }
357 
358         String username = prinCred[0];
359         String password = prinCred[1];
360 
361         return createToken(username, password, request, response);
362     }

364     /**
365      * Returns the username obtained from the
366      * {@link #getAuthzHeader(ServletRequest) authorizationHeader}.
367      * <p/>
368      * Once the {@code authzHeader} is split per the RFC (based on the space character ' '), the resulting split tokens
369      * are translated into the username/password pair by the
370      * {@link #getPrincipalsAndCredentials(String, String) getPrincipalsAndCredentials(scheme,encoded)} method.
371      *
372      * @param authorizationHeader the authorization header obtained from the request.
373      * @param request             the incoming ServletRequest
374      * @return the username (index 0)/password pair (index 1) submitted by the user for the given header value and request.
375      * @see #getAuthzHeader(ServletRequest)
376      */
377     protected String[] getPrincipalsAndCredentials(String authorizationHeader, ServletRequest request) {
378         if (authorizationHeader == null) {
379             return null;
380         }
381         String[] authTokens = authorizationHeader.split(" ");
382         if (authTokens == null || authTokens.length < 2) {
383             return null;
384         }
385         return getPrincipalsAndCredentials(authTokens[0], authTokens[1]);
386     }
