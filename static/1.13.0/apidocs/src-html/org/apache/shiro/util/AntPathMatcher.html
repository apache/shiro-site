021/**
022 * <p>PathMatcher implementation for Ant-style path patterns.
023 * Examples are provided below.</p>
024 *
025 * <p>Part of this mapping code has been kindly borrowed from
026 * <a href="http://ant.apache.org">Apache Ant</a>.
027 *
028 * <p>The mapping matches URLs using the following rules:<br>
029 * <ul>
030 * <li>? matches one character</li>
031 * <li>* matches zero or more characters</li>
032 * <li>** matches zero or more 'directories' in a path</li>
033 * </ul>
034 *
035 * <p>Some examples:<br>
036 * <ul>
037 * <li><code>com/t?st.jsp</code> - matches <code>com/test.jsp</code> but also
038 * <code>com/tast.jsp</code> or <code>com/txst.jsp</code></li>
039 * <li><code>com/*.jsp</code> - matches all <code>.jsp</code> files in the
040 * <code>com</code> directory</li>
041 * <li><code>com/&#42;&#42;/test.jsp</code> - matches all <code>test.jsp</code>
042 * files underneath the <code>com</code> path</li>
043 * <li><code>org/apache/shiro/&#42;&#42;/*.jsp</code> - matches all <code>.jsp</code>
044 * files underneath the <code>org/apache/shiro</code> path</li>
045 * <li><code>org/&#42;&#42;/servlet/bla.jsp</code> - matches
046 * <code>org/apache/shiro/servlet/bla.jsp</code> but also
047 * <code>org/apache/shiro/testing/servlet/bla.jsp</code> and
048 * <code>org/servlet/bla.jsp</code></li>
049 * </ul>
050 *
051 * <p><b>N.B.</b>: This class was borrowed (with much appreciation) from the
052 * <a href="http://www.springframework.org">Spring Framework</a> with modifications.  We didn't want to reinvent the
053 * wheel of great work they've done, but also didn't want to force every Shiro user to depend on Spring</p>
054 *
055 * <p>As per the Apache 2.0 license, the original copyright notice and all author and copyright information have
056 * remained in tact.</p>
057 *
058 * @since 16.07.2003
059 */
060public class AntPathMatcher implements PatternMatcher {

064    /**
065     * Default path separator: "/"
066     */
067    public static final String DEFAULT_PATH_SEPARATOR = "/";
068
069    private String pathSeparator = DEFAULT_PATH_SEPARATOR;

072    /**
073     * Set the path separator to use for pattern parsing.
074     * Default is "/", as in Ant.
075     */
076    public void setPathSeparator(String pathSeparator) {
077        this.pathSeparator = (pathSeparator != null ? pathSeparator : DEFAULT_PATH_SEPARATOR);
078    }

080    /**
081     * Checks if {@code path} is a pattern (i.e. contains a '*', or '?'). For example the {@code /foo/**} would return {@code true}, while {@code /bar/} would return {@code false}.
082     * @param path the string to check
083     * @return this method returns {@code true} if {@code path} contains a '*' or '?', otherwise, {@code false}
084     */
085    public boolean isPattern(String path) {
086        if (path == null) {
087            return false;
088        }
089        return (path.indexOf('*') != -1 || path.indexOf('?') != -1);
090    }
091
092    public boolean matches(String pattern, String source) {
093        return match(pattern, source);
094    }
095
096    public boolean match(String pattern, String path) {
097        return doMatch(pattern, path, true);
098    }
099
100    public boolean matchStart(String pattern, String path) {
101        return doMatch(pattern, path, false);
102    }

105    /**
106     * Actually match the given <code>path</code> against the given <code>pattern</code>.
107     *
108     * @param pattern   the pattern to match against
109     * @param path      the path String to test
110     * @param fullMatch whether a full pattern match is required
111     *                  (else a pattern match as far as the given base path goes is sufficient)
112     * @return <code>true</code> if the supplied <code>path</code> matched,
113     *         <code>false</code> if it didn't
114     */
115    protected boolean doMatch(String pattern, String path, boolean fullMatch) {
116        if (path == null || path.startsWith(this.pathSeparator) != pattern.startsWith(this.pathSeparator)) {
117            return false;
118        }
119
120        String[] pattDirs = StringUtils.tokenizeToStringArray(pattern, this.pathSeparator, false, true);
121        String[] pathDirs = StringUtils.tokenizeToStringArray(path, this.pathSeparator, false, true);
122
123        int pattIdxStart = 0;
124        int pattIdxEnd = pattDirs.length - 1;
125        int pathIdxStart = 0;
126        int pathIdxEnd = pathDirs.length - 1;
127
128        // Match all elements up to the first **
129        while (pattIdxStart <= pattIdxEnd && pathIdxStart <= pathIdxEnd) {
130            String patDir = pattDirs[pattIdxStart];
131            if ("**".equals(patDir)) {
132                break;
133            }
134            if (!matchStrings(patDir, pathDirs[pathIdxStart])) {
135                return false;
136            }
137            pattIdxStart++;
138            pathIdxStart++;
139        }
140
141        if (pathIdxStart > pathIdxEnd) {
142            // Path is exhausted, only match if rest of pattern is * or **'s
143            if (pattIdxStart > pattIdxEnd) {
144                return (pattern.endsWith(this.pathSeparator) ?
145                        path.endsWith(this.pathSeparator) : !path.endsWith(this.pathSeparator));
146            }
147            if (!fullMatch) {
148                return true;
149            }
150            if (pattIdxStart == pattIdxEnd && pattDirs[pattIdxStart].equals("*") &&
151                    path.endsWith(this.pathSeparator)) {
152                return true;
153            }
154            for (int i = pattIdxStart; i <= pattIdxEnd; i++) {
155                if (!pattDirs[i].equals("**")) {
156                    return false;
157                }
158            }
159            return true;
160        } else if (pattIdxStart > pattIdxEnd) {
161            // String not exhausted, but pattern is. Failure.
162            return false;
163        } else if (!fullMatch && "**".equals(pattDirs[pattIdxStart])) {
164            // Path start definitely matches due to "**" part in pattern.
165            return true;
166        }
167
168        // up to last '**'
169        while (pattIdxStart <= pattIdxEnd && pathIdxStart <= pathIdxEnd) {
170            String patDir = pattDirs[pattIdxEnd];
171            if (patDir.equals("**")) {
172                break;
173            }
174            if (!matchStrings(patDir, pathDirs[pathIdxEnd])) {
175                return false;
176            }
177            pattIdxEnd--;
178            pathIdxEnd--;
179        }
180        if (pathIdxStart > pathIdxEnd) {
181            // String is exhausted
182            for (int i = pattIdxStart; i <= pattIdxEnd; i++) {
183                if (!pattDirs[i].equals("**")) {
184                    return false;
185                }
186            }
187            return true;
188        }
189
190        while (pattIdxStart != pattIdxEnd && pathIdxStart <= pathIdxEnd) {
191            int patIdxTmp = -1;
192            for (int i = pattIdxStart + 1; i <= pattIdxEnd; i++) {
193                if (pattDirs[i].equals("**")) {
194                    patIdxTmp = i;
195                    break;
196                }
197            }
198            if (patIdxTmp == pattIdxStart + 1) {
199                // '**/**' situation, so skip one
200                pattIdxStart++;
201                continue;
202            }
203            // Find the pattern between padIdxStart & padIdxTmp in str between
204            // strIdxStart & strIdxEnd
205            int patLength = (patIdxTmp - pattIdxStart - 1);
206            int strLength = (pathIdxEnd - pathIdxStart + 1);
207            int foundIdx = -1;
208
209            strLoop:
210            for (int i = 0; i <= strLength - patLength; i++) {
211                for (int j = 0; j < patLength; j++) {
212                    String subPat = (String) pattDirs[pattIdxStart + j + 1];
213                    String subStr = (String) pathDirs[pathIdxStart + i + j];
214                    if (!matchStrings(subPat, subStr)) {
215                        continue strLoop;
216                    }
217                }
218                foundIdx = pathIdxStart + i;
219                break;
220            }
221
222            if (foundIdx == -1) {
223                return false;
224            }
225
226            pattIdxStart = patIdxTmp;
227            pathIdxStart = foundIdx + patLength;
228        }
229
230        for (int i = pattIdxStart; i <= pattIdxEnd; i++) {
231            if (!pattDirs[i].equals("**")) {
232                return false;
233            }
234        }
235
236        return true;
237    }

239    /**
240     * Tests whether or not a string matches against a pattern.
241     * The pattern may contain two special characters:<br>
242     * '*' means zero or more characters<br>
243     * '?' means one and only one character
244     *
245     * @param pattern pattern to match against.
246     *                Must not be <code>null</code>.
247     * @param str     string which must be matched against the pattern.
248     *                Must not be <code>null</code>.
249     * @return <code>true</code> if the string matches against the
250     *         pattern, or <code>false</code> otherwise.
251     */
252    private boolean matchStrings(String pattern, String str) {
253        char[] patArr = pattern.toCharArray();
254        char[] strArr = str.toCharArray();
255        int patIdxStart = 0;
256        int patIdxEnd = patArr.length - 1;
257        int strIdxStart = 0;
258        int strIdxEnd = strArr.length - 1;
259        char ch;
260
261        boolean containsStar = false;
262        for (char aPatArr : patArr) {
263            if (aPatArr == '*') {
264                containsStar = true;
265                break;
266            }
267        }
268
269        if (!containsStar) {
270            // No '*'s, so we make a shortcut
271            if (patIdxEnd != strIdxEnd) {
272                return false; // Pattern and string do not have the same size
273            }
274            for (int i = 0; i <= patIdxEnd; i++) {
275                ch = patArr[i];
276                if (ch != '?') {
277                    if (ch != strArr[i]) {
278                        return false;// Character mismatch
279                    }
280                }
281            }
282            return true; // String matches against pattern
283        }
284
285
286        if (patIdxEnd == 0) {
287            return true; // Pattern contains only '*', which matches anything
288        }
289
290        // Process characters before first star
291        while ((ch = patArr[patIdxStart]) != '*' && strIdxStart <= strIdxEnd) {
292            if (ch != '?') {
293                if (ch != strArr[strIdxStart]) {
294                    return false;// Character mismatch
295                }
296            }
297            patIdxStart++;
298            strIdxStart++;
299        }
300        if (strIdxStart > strIdxEnd) {
301            // All characters in the string are used. Check if only '*'s are
302            // left in the pattern. If so, we succeeded. Otherwise failure.
303            for (int i = patIdxStart; i <= patIdxEnd; i++) {
304                if (patArr[i] != '*') {
305                    return false;
306                }
307            }
308            return true;
309        }
310
311        // Process characters after last star
312        while ((ch = patArr[patIdxEnd]) != '*' && strIdxStart <= strIdxEnd) {
313            if (ch != '?') {
314                if (ch != strArr[strIdxEnd]) {
315                    return false;// Character mismatch
316                }
317            }
318            patIdxEnd--;
319            strIdxEnd--;
320        }
321        if (strIdxStart > strIdxEnd) {
322            // All characters in the string are used. Check if only '*'s are
323            // left in the pattern. If so, we succeeded. Otherwise failure.
324            for (int i = patIdxStart; i <= patIdxEnd; i++) {
325                if (patArr[i] != '*') {
326                    return false;
327                }
328            }
329            return true;
330        }
331
332        // process pattern between stars. padIdxStart and patIdxEnd point
333        // always to a '*'.
334        while (patIdxStart != patIdxEnd && strIdxStart <= strIdxEnd) {
335            int patIdxTmp = -1;
336            for (int i = patIdxStart + 1; i <= patIdxEnd; i++) {
337                if (patArr[i] == '*') {
338                    patIdxTmp = i;
339                    break;
340                }
341            }
342            if (patIdxTmp == patIdxStart + 1) {
343                // Two stars next to each other, skip the first one.
344                patIdxStart++;
345                continue;
346            }
347            // Find the pattern between padIdxStart & padIdxTmp in str between
348            // strIdxStart & strIdxEnd
349            int patLength = (patIdxTmp - patIdxStart - 1);
350            int strLength = (strIdxEnd - strIdxStart + 1);
351            int foundIdx = -1;
352            strLoop:
353            for (int i = 0; i <= strLength - patLength; i++) {
354                for (int j = 0; j < patLength; j++) {
355                    ch = patArr[patIdxStart + j + 1];
356                    if (ch != '?') {
357                        if (ch != strArr[strIdxStart + i + j]) {
358                            continue strLoop;
359                        }
360                    }
361                }
362
363                foundIdx = strIdxStart + i;
364                break;
365            }
366
367            if (foundIdx == -1) {
368                return false;
369            }
370
371            patIdxStart = patIdxTmp;
372            strIdxStart = foundIdx + patLength;
373        }
374
375        // All characters in the string are used. Check if only '*'s are left
376        // in the pattern. If so, we succeeded. Otherwise failure.
377        for (int i = patIdxStart; i <= patIdxEnd; i++) {
378            if (patArr[i] != '*') {
379                return false;
380            }
381        }
382
383        return true;
384    }

386    /**
387     * Given a pattern and a full path, determine the pattern-mapped part.
388     * <p>For example:
389     * <ul>
390     * <li>'<code>/docs/cvs/commit.html</code>' and '<code>/docs/cvs/commit.html</code> -> ''</li>
391     * <li>'<code>/docs/*</code>' and '<code>/docs/cvs/commit</code> -> '<code>cvs/commit</code>'</li>
392     * <li>'<code>/docs/cvs/*.html</code>' and '<code>/docs/cvs/commit.html</code> -> '<code>commit.html</code>'</li>
393     * <li>'<code>/docs/**</code>' and '<code>/docs/cvs/commit</code> -> '<code>cvs/commit</code>'</li>
394     * <li>'<code>/docs/**\/*.html</code>' and '<code>/docs/cvs/commit.html</code> -> '<code>cvs/commit.html</code>'</li>
395     * <li>'<code>/*.html</code>' and '<code>/docs/cvs/commit.html</code> -> '<code>docs/cvs/commit.html</code>'</li>
396     * <li>'<code>*.html</code>' and '<code>/docs/cvs/commit.html</code> -> '<code>/docs/cvs/commit.html</code>'</li>
397     * <li>'<code>*</code>' and '<code>/docs/cvs/commit.html</code> -> '<code>/docs/cvs/commit.html</code>'</li>
398     * </ul>
399     * <p>Assumes that {@link #match} returns <code>true</code> for '<code>pattern</code>'
400     * and '<code>path</code>', but does <strong>not</strong> enforce this.
401     */
402    public String extractPathWithinPattern(String pattern, String path) {
403        String[] patternParts = StringUtils.tokenizeToStringArray(pattern, this.pathSeparator, false, true);
404        String[] pathParts = StringUtils.tokenizeToStringArray(path, this.pathSeparator, false, true);
405        StringBuilder builder = new StringBuilder();
406        boolean pathStarted = false;
407
408        for (int segment = 0; segment < patternParts.length; segment++) {
409            String patternPart = patternParts[segment];
410            if (patternPart.indexOf('*') > -1 || patternPart.indexOf('?') > -1) {
411                for (; segment < pathParts.length; segment++) {
412                    if (pathStarted || (segment == 0 && !pattern.startsWith(this.pathSeparator))) {
413                        builder.append(this.pathSeparator);
414                    }
415                    builder.append(pathParts[segment]);
416                    pathStarted = true;
417                }
418            }
419        }
420
421        return builder.toString();
422    }
