051    /**
052     * Check whether the given String has actual text.
053     * More specifically, returns <code>true</code> if the string not <code>null</code>,
054     * its length is greater than 0, and it contains at least one non-whitespace character.
055     * <p/>
056     * <code>StringUtils.hasText(null) == false<br/>
057     * StringUtils.hasText("") == false<br/>
058     * StringUtils.hasText(" ") == false<br/>
059     * StringUtils.hasText("12345") == true<br/>
060     * StringUtils.hasText(" 12345 ") == true</code>
061     * <p/>
062     * <p>Copied from the Spring Framework while retaining all license, copyright and author information.
063     *
064     * @param str the String to check (may be <code>null</code>)
065     * @return <code>true</code> if the String is not <code>null</code>, its length is
066     *         greater than 0, and it does not contain whitespace only
067     * @see java.lang.Character#isWhitespace
068     */
069    public static boolean hasText(String str) {
070        if (!hasLength(str)) {
071            return false;
072        }
073        int strLen = str.length();
074        for (int i = 0; i < strLen; i++) {
075            if (!Character.isWhitespace(str.charAt(i))) {
076                return true;
077            }
078        }
079        return false;
080    }

083     * Check that the given String is neither <code>null</code> nor of length 0.
084     * Note: Will return <code>true</code> for a String that purely consists of whitespace.

102    /**
103     * Test if the given String starts with the specified prefix,
104     * ignoring upper/lower case.
105     * <p/>
106     * <p>Copied from the Spring Framework while retaining all license, copyright and author information.
107     *
108     * @param str    the String to check
109     * @param prefix the prefix to look for
110     * @return <code>true</code> starts with the specified prefix (ignoring case), <code>false</code> if it does not.
111     * @see java.lang.String#startsWith
112     */
113    public static boolean startsWithIgnoreCase(String str, String prefix) {
114        if (str == null || prefix == null) {
115            return false;
116        }
117        if (str.startsWith(prefix)) {
118            return true;
119        }
120        if (str.length() < prefix.length()) {
121            return false;
122        }
123        String lcStr = str.substring(0, prefix.length()).toLowerCase();
124        String lcPrefix = prefix.toLowerCase();
125        return lcStr.equals(lcPrefix);
126    }

128    /**
129     * Returns a 'cleaned' representation of the specified argument.  'Cleaned' is defined as the following:
130     * <p/>
131     * <ol>
132     * <li>If the specified <code>String</code> is <code>null</code>, return <code>null</code></li>
133     * <li>If not <code>null</code>, {@link String#trim() trim()} it, when the trim param is set to <code>true</code>.</li>
134     * <li>If the trimmed string is equal to the empty String (i.e. &quot;&quot;), return <code>null</code></li>
135     * <li>If the trimmed string is not the empty string, return the trimmed version</li>.
136     * </ol>
137     * <p/>
138     * Therefore this method always ensures that any given string has trimmed text, and if it doesn't, <code>null</code>
139     * is returned.
140     *
141     * @param in the input String to clean.
142     * @param trim specifies whether the input String should be trimmed or not
143     * @return a populated-but-trimmed String or <code>null</code> otherwise
144     */
145    public static String clean(String in, boolean trim) {
146        String out = in;
147
148        if (in != null) {
149            if (trim) {
150                out = in.trim();
151            }
152            if (out.equals(EMPTY_STRING)) {
153                out = null;
154            }
155        }
156
157        return out;
158    }

191    /**
192     * Returns the array's contents as a string, with each element delimited by the specified
193     * {@code delimiter} argument.  Useful for {@code toString()} implementations and log messages.
194     *
195     * @param array     the array whose contents will be converted to a string
196     * @param delimiter the delimiter to use between each element
197     * @return a single string, delimited by the specified {@code delimiter}.
198     * @since 1.0
199     */
200    public static String toDelimitedString(Object[] array, String delimiter) {
201        if (array == null || array.length == 0) {
202            return EMPTY_STRING;
203        }
204        StringBuilder sb = new StringBuilder();
205        for (int i = 0; i < array.length; i++) {
206            if (i > 0) {
207                sb.append(delimiter);
208            }
209            sb.append(array[i]);
210        }
211        return sb.toString();
212    }

251    /**
252     * Tokenize the given String into a String array via a StringTokenizer.
253     * <p>The given delimiters string is supposed to consist of any number of
254     * delimiter characters. Each of those characters can be used to separate
255     * tokens. A delimiter is always a single character; for multi-character
256     * delimiters, consider using <code>delimitedListToStringArray</code>
257     * <p/>
258     * <p>Copied from the Spring Framework while retaining all license, copyright and author information.
259     *
260     * @param str               the String to tokenize
261     * @param delimiters        the delimiter characters, assembled as String
262     *                          (each of those characters is individually considered as delimiter)
263     * @param trimTokens        trim the tokens via String's <code>trim</code>
264     * @param ignoreEmptyTokens omit empty tokens from the result array
265     *                          (only applies to tokens that are empty after trimming; StringTokenizer
266     *                          will not consider subsequent delimiters as token in the first place).
267     * @return an array of the tokens (<code>null</code> if the input String
268     *         was <code>null</code>)
269     * @see java.util.StringTokenizer
270     * @see java.lang.String#trim()
271     */
272    @SuppressWarnings({"unchecked"})
273    public static String[] tokenizeToStringArray(
274            String str, String delimiters, boolean trimTokens, boolean ignoreEmptyTokens) {
275
276        if (str == null) {
277            return null;
278        }
279        StringTokenizer st = new StringTokenizer(str, delimiters);
280        List tokens = new ArrayList();
281        while (st.hasMoreTokens()) {
282            String token = st.nextToken();
283            if (trimTokens) {
284                token = token.trim();
285            }
286            if (!ignoreEmptyTokens || token.length() > 0) {
287                tokens.add(token);
288            }
289        }
290        return toStringArray(tokens);
291    }

311    public static String[] splitKeyValue(String aLine) throws ParseException {
312        String line = clean(aLine);
313        if (line == null) {
314            return null;
315        }
316        String[] split = line.split(" ", 2);
317        if (split.length != 2) {
318            //fallback to checking for an equals sign
319            split = line.split("=", 2);
320            if (split.length != 2) {
321                String msg = "Unable to determine Key/Value pair from line [" + line + "].  There is no space from " +
322                        "which the split location could be determined.";
323                throw new ParseException(msg, 0);
324            }
325
326        }
327
328        split[0] = clean(split[0]);
329        split[1] = clean(split[1]);
330        if (split[1].startsWith("=")) {
331            //they used spaces followed by an equals followed by zero or more spaces to split the key/value pair, so
332            //remove the equals sign to result in only the key and values in the
333            split[1] = clean(split[1].substring(1));
334        }
335
336        if (split[0] == null) {
337            String msg = "No valid key could be found in line [" + line + "] to form a key/value pair.";
338            throw new ParseException(msg, 0);
339        }
340        if (split[1] == null) {
341            String msg = "No corresponding value could be found in line [" + line + "] for key [" + split[0] + "]";
342            throw new ParseException(msg, 0);
343        }
344
345        return split;
346    }

364    /**
365     * Splits the specified delimited String into tokens, supporting quoted tokens so that quoted strings themselves
366     * won't be tokenized.
367     * <p/>
368     * This method's implementation is very loosely based (with significant modifications) on
369     * <a href="http://blogs.bytecode.com.au/glen">Glen Smith</a>'s open-source
370     * <a href="http://opencsv.svn.sourceforge.net/viewvc/opencsv/trunk/src/au/com/bytecode/opencsv/CSVReader.java?&view=markup">CSVReader.java</a>
371     * file.
372     * <p/>
373     * That file is Apache 2.0 licensed as well, making Glen's code a great starting point for us to modify to
374     * our needs.
375     *
376     * @param aLine          the String to parse
377     * @param delimiter      the delimiter by which the <tt>line</tt> argument is to be split
378     * @param beginQuoteChar the character signifying the start of quoted text (so the quoted text will not be split)
379     * @param endQuoteChar   the character signifying the end of quoted text
380     * @param retainQuotes   if the quotes themselves should be retained when constructing the corresponding token
381     * @param trimTokens     if leading and trailing whitespace should be trimmed from discovered tokens.
382     * @return the tokens discovered from parsing the given delimited <tt>line</tt>.
383     */
384    public static String[] split(String aLine, char delimiter, char beginQuoteChar, char endQuoteChar,
385                                 boolean retainQuotes, boolean trimTokens) {
386        String line = clean(aLine);
387        if (line == null) {
388            return null;
389        }
390
391        List<String> tokens = new ArrayList<String>();
392        StringBuilder sb = new StringBuilder();
393        boolean inQuotes = false;
394
395        for (int i = 0; i < line.length(); i++) {
396
397            char c = line.charAt(i);
398            if (c == beginQuoteChar) {
399                // this gets complex... the quote may end a quoted block, or escape another quote.
400                // do a 1-char lookahead:
401                if (inQuotes  // we are in quotes, therefore there can be escaped quotes in here.
402                        && line.length() > (i + 1)  // there is indeed another character to check.
403                        && line.charAt(i + 1) == beginQuoteChar) { // ..and that char. is a quote also.
404                    // we have two quote chars in a row == one quote char, so consume them both and
405                    // put one on the token. we do *not* exit the quoted text.
406                    sb.append(line.charAt(i + 1));
407                    i++;
408                } else {
409                    inQuotes = !inQuotes;
410                    if (retainQuotes) {
411                        sb.append(c);
412                    }
413                }
414            } else if (c == endQuoteChar) {
415                inQuotes = !inQuotes;
416                if (retainQuotes) {
417                    sb.append(c);
418                }
419            } else if (c == delimiter && !inQuotes) {
420                String s = sb.toString();
421                if (trimTokens) {
422                    s = s.trim();
423                }
424                tokens.add(s);
425                sb = new StringBuilder(); // start work on next token
426            } else {
427                sb.append(c);
428            }
429        }
430        String s = sb.toString();
431        if (trimTokens) {
432            s = s.trim();
433        }
434        tokens.add(s);
435        return tokens.toArray(new String[tokens.size()]);
436    }

438    /**
439     * Joins the elements of the provided {@code Iterator} into
440     * a single String containing the provided elements.</p>
441     * <p/>
442     * No delimiter is added before or after the list.
443     * A {@code null} separator is the same as an empty String ("").</p>
444     * <p/>
445     * Copied from Commons Lang, version 3 (r1138702).</p>
446     *
447     * @param iterator  the {@code Iterator} of values to join together, may be null
448     * @param separator the separator character to use, null treated as ""
449     * @return the joined String, {@code null} if null iterator input
450     * @since 1.2
451     */
452    public static String join(Iterator<?> iterator, String separator) {
453        final String empty = "";
454
455        // handle null, zero and one elements before building a buffer
456        if (iterator == null) {
457            return null;
458        }
459        if (!iterator.hasNext()) {
460            return empty;
461        }
462        Object first = iterator.next();
463        if (!iterator.hasNext()) {
464            return first == null ? empty : first.toString();
465        }
466
467        // two or more elements
468        StringBuilder buf = new StringBuilder(256); // Java default is 16, probably too small
469        if (first != null) {
470            buf.append(first);
471        }
472
473        while (iterator.hasNext()) {
474            if (separator != null) {
475                buf.append(separator);
476            }
477            Object obj = iterator.next();
478            if (obj != null) {
479                buf.append(obj);
480            }
481        }
482        return buf.toString();
483    }

485    /**
486     * Splits the {@code delimited} string (delimited by the specified {@code separator} character) and returns the
487     * delimited values as a {@code Set}.
488     * <p/>
489     * If either argument is {@code null}, this method returns {@code null}.
490     *
491     * @param delimited the string to split
492     * @param separator the character that delineates individual tokens to split
493     * @return the delimited values as a {@code Set}.
494     * @since 1.2
495     */
496    public static Set<String> splitToSet(String delimited, String separator) {
497        if (delimited == null || separator == null) {
498            return null;
499        }
500        String[] split = split(delimited, separator.charAt(0));
501        return asSet(split);
502    }

504    /**
505     * Returns the input argument, but ensures the first character is capitalized (if possible).
506     * @param in the string to uppercase the first character.
507     * @return the input argument, but with the first character capitalized (if possible).
508     * @since 1.2
509     */
510    public static String uppercaseFirstChar(String in) {
511        if (in == null || in.length() == 0) {
512            return in;
513        }
514        int length = in.length();
515        StringBuilder sb = new StringBuilder(length);
516
517        sb.append(Character.toUpperCase(in.charAt(0)));
518        if (length > 1) {
519            String remaining = in.substring(1);
520            sb.append(remaining);
521        }
522        return sb.toString();
523    }

531    private static <E> Set<E> asSet(E... elements) {
532        if (elements == null || elements.length == 0) {
533            return Collections.emptySet();
534        }
535
536        if (elements.length == 1) {
537            return Collections.singleton(elements[0]);
538        }
539
540        LinkedHashSet<E> set = new LinkedHashSet<E>(elements.length * 4 / 3 + 1);
541        Collections.addAll(set, elements);
542        return set;
543    }
