019package org.apache.shiro.util;
020
021import java.util.regex.Pattern;
022import java.util.regex.Matcher;
023
024/**
025 * {@code PatternMatcher} implementation that uses standard {@link java.util.regex} objects.
026 *
027 * @see Pattern
028 * @since 1.0
029 */
030public class RegExPatternMatcher implements PatternMatcher {
031
032    private static final int DEFAULT = Pattern.DOTALL;
033
034    private static final int CASE_INSENSITIVE = DEFAULT | Pattern.CASE_INSENSITIVE;
035
036    private boolean caseInsensitive = false;
037
038    /**
039     * Simple implementation that merely uses the default pattern comparison logic provided by the
040     * JDK.
041     * <p/>This implementation essentially executes the following:
042     * <pre>
043     * Pattern p = Pattern.compile(pattern, Pattern.DOTALL);
044     * Matcher m = p.matcher(source);
045     * return m.matches();</pre>
046     * @param pattern the pattern to match against
047     * @param source  the source to match
048     * @return {@code true} if the source matches the required pattern, {@code false} otherwise.
049     */
050    public boolean matches(String pattern, String source) {
051        if (pattern == null) {
052            throw new IllegalArgumentException("pattern argument cannot be null.");
053        }
054        Pattern p = Pattern.compile(pattern, caseInsensitive ? CASE_INSENSITIVE : DEFAULT);
055        Matcher m = p.matcher(source);
056        return m.matches();
057    }
058
059    /**
060     * Returns true if regex match should be case-insensitive.
061     * @return true if regex match should be case-insensitive.
062     */
063    public boolean isCaseInsensitive() {
064        return caseInsensitive;
065    }
066
067    /**
068     * Adds the Pattern.CASE_INSENSITIVE flag when compiling patterns.
069     * @param caseInsensitive true if patterns should match case-insensitive.
070     */
071    public void setCaseInsensitive(boolean caseInsensitive) {
072        this.caseInsensitive = caseInsensitive;
073    }
074}
