019package org.apache.shiro.spring.config.web.autoconfigure;
020
021import org.apache.shiro.spring.web.config.ShiroRequestMappingConfig;
022import org.springframework.boot.autoconfigure.condition.ConditionalOnClass;
023import org.springframework.boot.autoconfigure.condition.ConditionalOnProperty;
024import org.springframework.boot.autoconfigure.condition.ConditionalOnWebApplication;
025import org.springframework.context.annotation.Configuration;
026import org.springframework.context.annotation.Import;
027import org.springframework.web.servlet.mvc.method.annotation.RequestMappingHandlerMapping;
028
029@Configuration
030@ConditionalOnClass(RequestMappingHandlerMapping.class)
031@Import(ShiroRequestMappingConfig.class)
032@ConditionalOnWebApplication(type = ConditionalOnWebApplication.Type.SERVLET)
033@ConditionalOnProperty(name = "shiro.web.enabled", matchIfMissing = true)
034public class ShiroWebMvcAutoConfiguration { }
