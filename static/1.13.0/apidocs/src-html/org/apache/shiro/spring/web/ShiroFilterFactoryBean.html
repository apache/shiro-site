053/**
054 * {@link org.springframework.beans.factory.FactoryBean FactoryBean} to be used in Spring-based web applications for
055 * defining the master Shiro Filter.
056 * <h4>Usage</h4>
057 * Declare a DelegatingFilterProxy in {@code web.xml}, matching the filter name to the bean id:
058 * <pre>
059 * &lt;filter&gt;
060 *   &lt;filter-name&gt;<b>shiroFilter</b>&lt;/filter-name&gt;
061 *   &lt;filter-class&gt;org.springframework.web.filter.DelegatingFilterProxy&lt;filter-class&gt;
062 *   &lt;init-param&gt;
063 *    &lt;param-name&gt;targetFilterLifecycle&lt;/param-name&gt;
064 *     &lt;param-value&gt;true&lt;/param-value&gt;
065 *   &lt;/init-param&gt;
066 * &lt;/filter&gt;
067 * </pre>
068 * Then, in your spring XML file that defines your web ApplicationContext:
069 * <pre>
070 * &lt;bean id="<b>shiroFilter</b>" class="org.apache.shiro.spring.web.ShiroFilterFactoryBean"&gt;
071 *    &lt;property name="securityManager" ref="securityManager"/&gt;
072 *    &lt;!-- other properties as necessary ... --&gt;
073 * &lt;/bean&gt;
074 * </pre>
075 * <h4>Filter Auto-Discovery</h4>
076 * While there is a {@link #setFilters(java.util.Map) filters} property that allows you to assign a filter beans
077 * to the 'pool' of filters available when defining {@link #setFilterChainDefinitions(String) filter chains}, it is
078 * optional.
079 * <p/>
080 * This implementation is also a {@link BeanPostProcessor} and will acquire
081 * any {@link javax.servlet.Filter Filter} beans defined independently in your Spring application context.  Upon
082 * discovery, they will be automatically added to the {@link #setFilters(java.util.Map) map} keyed by the bean ID.
083 * That ID can then be used in the filter chain definitions, for example:
084 *
085 * <pre>
086 * &lt;bean id="<b>myCustomFilter</b>" class="com.class.that.implements.javax.servlet.Filter"/&gt;
087 * ...
088 * &lt;bean id="shiroFilter" class="org.apache.shiro.spring.web.ShiroFilterFactoryBean"&gt;
089 *    ...
090 *    &lt;property name="filterChainDefinitions"&gt;
091 *        &lt;value&gt;
092 *            /some/path/** = authc, <b>myCustomFilter</b>
093 *        &lt;/value&gt;
094 *    &lt;/property&gt;
095 * &lt;/bean&gt;
096 * </pre>
097 * <h4>Global Property Values</h4>
098 * Most Shiro servlet Filter implementations exist for defining custom Filter
099 * {@link #setFilterChainDefinitions(String) chain definitions}.  Most implementations subclass one of the
100 * {@link AccessControlFilter}, {@link AuthenticationFilter}, {@link AuthorizationFilter} classes to simplify things,
101 * and each of these 3 classes has configurable properties that are application-specific.
102 * <p/>
103 * A dilemma arises where, if you want to for example set the application's 'loginUrl' for any Filter, you don't want
104 * to have to manually specify that value for <em>each</em> filter instance definied.
105 * <p/>
106 * To prevent configuration duplication, this implementation provides the following properties to allow you
107 * to set relevant values in only one place:
108 * <ul>
109 * <li>{@link #setLoginUrl(String)}</li>
110 * <li>{@link #setSuccessUrl(String)}</li>
111 * <li>{@link #setUnauthorizedUrl(String)}</li>
112 * </ul>
113 *
114 * Then at startup, any values specified via these 3 properties will be applied to all configured
115 * Filter instances so you don't have to specify them individually on each filter instance.  To ensure your own custom
116 * filters benefit from this convenience, your filter implementation should subclass one of the 3 mentioned
117 * earlier.
118 *
119 * @see org.springframework.web.filter.DelegatingFilterProxy DelegatingFilterProxy
120 * @since 1.0
121 */

345     * Each key/value pair must conform to the format defined by the
346     * {@link FilterChainManager#createChain(String,String)} JavaDoc - each property key is an ant URL

348     *

350     *                    where each key/value pair represents a single urlPathExpression-commaDelimitedChainDefinition.
351     */
352    public void setFilterChainDefinitions(String definitions) {
353        Ini ini = new Ini();
354        ini.load(definitions);
355        //did they explicitly state a 'urls' section?  Not necessary, but just in case:
356        Ini.Section section = ini.getSection(IniFilterChainResolverFactory.URLS);
357        if (CollectionUtils.isEmpty(section)) {
358            //no urls section.  Since this _is_ a urls chain definition property, just assume the
359            //default section contains only the definitions:
360            section = ini.getSection(Ini.DEFAULT_SECTION_NAME);
361        }
362        setFilterChainDefinitionMap(section);
363    }

405    protected FilterChainManager createFilterChainManager() {
406
407        DefaultFilterChainManager manager = new DefaultFilterChainManager();
408        Map<String, Filter> defaultFilters = manager.getFilters();
409        //apply global settings if necessary:
410        for (Filter filter : defaultFilters.values()) {
411            applyGlobalPropertiesIfNecessary(filter);
412        }
413
414        //Apply the acquired and/or configured filters:
415        Map<String, Filter> filters = getFilters();
416        if (!CollectionUtils.isEmpty(filters)) {
417            for (Map.Entry<String, Filter> entry : filters.entrySet()) {
418                String name = entry.getKey();
419                Filter filter = entry.getValue();
420                applyGlobalPropertiesIfNecessary(filter);
421                if (filter instanceof Nameable) {
422                    ((Nameable) filter).setName(name);
423                }
424                //'init' argument is false, since Spring-configured filters should be initialized
425                //in Spring (i.e. 'init-method=blah') or implement InitializingBean:
426                manager.addFilter(name, filter, false);
427            }
428        }
429
430        // set the global filters
431        manager.setGlobalFilters(this.globalFilters);
432
433        //build up the chains:
434        Map<String, String> chains = getFilterChainDefinitionMap();
435        if (!CollectionUtils.isEmpty(chains)) {
436            for (Map.Entry<String, String> entry : chains.entrySet()) {
437                String url = entry.getKey();
438                String chainDefinition = entry.getValue();
439                manager.createChain(url, chainDefinition);
440            }
441        }
442
443        // create the default chain, to match anything the path matching would have missed
444        manager.createDefaultChain("/**"); // TODO this assumes ANT path matching, which might be OK here
445
446        return manager;
447    }

449    /**
450     * This implementation:
451     * <ol>
452     * <li>Ensures the required {@link #setSecurityManager(org.apache.shiro.mgt.SecurityManager) securityManager}
453     * property has been set</li>
454     * <li>{@link #createFilterChainManager() Creates} a {@link FilterChainManager} instance that reflects the
455     * configured {@link #setFilters(java.util.Map) filters} and
456     * {@link #setFilterChainDefinitionMap(java.util.Map) filter chain definitions}</li>
457     * <li>Wraps the FilterChainManager with a suitable
458     * {@link org.apache.shiro.web.filter.mgt.FilterChainResolver FilterChainResolver} since the Shiro Filter
459     * implementations do not know of {@code FilterChainManager}s</li>
460     * <li>Sets both the {@code SecurityManager} and {@code FilterChainResolver} instances on a new Shiro Filter
461     * instance and returns that filter instance.</li>
462     * </ol>
463     *
464     * @return a new Shiro Filter reflecting any configured filters and filter chain definitions.
465     * @throws Exception if there is a problem creating the AbstractShiroFilter instance.
466     */
467    protected AbstractShiroFilter createInstance() throws Exception {
468
469        log.debug("Creating Shiro Filter instance.");
470
471        SecurityManager securityManager = getSecurityManager();
472        if (securityManager == null) {
473            String msg = "SecurityManager property must be set.";
474            throw new BeanInitializationException(msg);
475        }
476
477        if (!(securityManager instanceof WebSecurityManager)) {
478            String msg = "The security manager does not implement the WebSecurityManager interface.";
479            throw new BeanInitializationException(msg);
480        }
481
482        FilterChainManager manager = createFilterChainManager();
483
484        //Expose the constructed FilterChainManager by first wrapping it in a
485        // FilterChainResolver implementation. The AbstractShiroFilter implementations
486        // do not know about FilterChainManagers - only resolvers:
487        PathMatchingFilterChainResolver chainResolver = new PathMatchingFilterChainResolver();
488        chainResolver.setFilterChainManager(manager);
489
490        //Now create a concrete ShiroFilter instance and apply the acquired SecurityManager and built
491        //FilterChainResolver.  It doesn't matter that the instance is an anonymous inner class
492        //here - we're just using it because it is a concrete AbstractShiroFilter instance that accepts
493        //injection of the SecurityManager and FilterChainResolver:
494        return new SpringShiroFilter((WebSecurityManager) securityManager, chainResolver, getShiroFilterConfiguration());
495    }

497    private void applyLoginUrlIfNecessary(Filter filter) {
498        String loginUrl = getLoginUrl();
499        if (StringUtils.hasText(loginUrl) && (filter instanceof AccessControlFilter)) {
500            AccessControlFilter acFilter = (AccessControlFilter) filter;
501            //only apply the login url if they haven't explicitly configured one already:
502            String existingLoginUrl = acFilter.getLoginUrl();
503            if (AccessControlFilter.DEFAULT_LOGIN_URL.equals(existingLoginUrl)) {
504                acFilter.setLoginUrl(loginUrl);
505            }
506        }
507    }
508
509    private void applySuccessUrlIfNecessary(Filter filter) {
510        String successUrl = getSuccessUrl();
511        if (StringUtils.hasText(successUrl) && (filter instanceof AuthenticationFilter)) {
512            AuthenticationFilter authcFilter = (AuthenticationFilter) filter;
513            //only apply the successUrl if they haven't explicitly configured one already:
514            String existingSuccessUrl = authcFilter.getSuccessUrl();
515            if (AuthenticationFilter.DEFAULT_SUCCESS_URL.equals(existingSuccessUrl)) {
516                authcFilter.setSuccessUrl(successUrl);
517            }
518        }
519    }
520
521    private void applyUnauthorizedUrlIfNecessary(Filter filter) {
522        String unauthorizedUrl = getUnauthorizedUrl();
523        if (StringUtils.hasText(unauthorizedUrl) && (filter instanceof AuthorizationFilter)) {
524            AuthorizationFilter authzFilter = (AuthorizationFilter) filter;
525            //only apply the unauthorizedUrl if they haven't explicitly configured one already:
526            String existingUnauthorizedUrl = authzFilter.getUnauthorizedUrl();
527            if (existingUnauthorizedUrl == null) {
528                authzFilter.setUnauthorizedUrl(unauthorizedUrl);
529            }
530        }
531    }
532
533    private void applyGlobalPropertiesIfNecessary(Filter filter) {
534        applyLoginUrlIfNecessary(filter);
535        applySuccessUrlIfNecessary(filter);
536        applyUnauthorizedUrlIfNecessary(filter);
537
538        if (filter instanceof OncePerRequestFilter) {
539            ((OncePerRequestFilter) filter).setFilterOncePerRequest(filterConfiguration.isFilterOncePerRequest());
540        }
541    }

543    /**
544     * Inspects a bean, and if it implements the {@link Filter} interface, automatically adds that filter
545     * instance to the internal {@link #setFilters(java.util.Map) filters map} that will be referenced
546     * later during filter chain construction.
547     */
548    public Object postProcessBeforeInitialization(Object bean, String beanName) throws BeansException {
549        if (bean instanceof Filter) {
550            log.debug("Found filter chain candidate filter '{}'", beanName);
551            Filter filter = (Filter) bean;
552            applyGlobalPropertiesIfNecessary(filter);
553            getFilters().put(beanName, filter);
554        } else {
555            log.trace("Ignoring non-Filter bean '{}'", beanName);
556        }
557        return bean;
558    }

568    /**
569     * Ordinarily the {@code AbstractShiroFilter} must be subclassed to additionally perform configuration
570     * and initialization behavior.  Because this {@code FactoryBean} implementation manually builds the
571     * {@link AbstractShiroFilter}'s
572     * {@link AbstractShiroFilter#setSecurityManager(org.apache.shiro.web.mgt.WebSecurityManager) securityManager} and
573     * {@link AbstractShiroFilter#setFilterChainResolver(org.apache.shiro.web.filter.mgt.FilterChainResolver) filterChainResolver}
574     * properties, the only thing left to do is set those properties explicitly.  We do that in a simple
575     * concrete subclass in the constructor.
576     */
