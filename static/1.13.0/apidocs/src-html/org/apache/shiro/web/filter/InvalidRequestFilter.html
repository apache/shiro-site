020package org.apache.shiro.web.filter;
021
022import org.apache.shiro.util.StringUtils;
023import org.apache.shiro.web.util.WebUtils;
024
025import javax.servlet.ServletRequest;
026import javax.servlet.ServletResponse;
027import javax.servlet.http.HttpServletRequest;
028import java.util.Arrays;
029import java.util.Collections;
030import java.util.List;
031import java.util.stream.Stream;
032
033/**
034 * A request filter that blocks malicious requests. Invalid request will respond with a 400 response code.
035 * <p>
036 * This filter checks and blocks the request if the following characters are found in the request URI:
037 * <ul>
038 *     <li>Semicolon - can be disabled by setting {@code blockSemicolon = false}</li>
039 *     <li>Backslash - can be disabled by setting {@code blockBackslash = false}</li>
040 *     <li>Non-ASCII characters - can be disabled by setting {@code blockNonAscii = false}, the ability to disable this check will be removed in future version.</li>
041 *     <li>Path traversals - can be disabled by setting {@code blockTraversal = false}</li>
042 * </ul>
043 *
044 * @see <a href="https://docs.spring.io/spring-security/site/docs/current/api/org/springframework/security/web/firewall/StrictHttpFirewall.html">This class was inspired by Spring Security StrictHttpFirewall</a>
045 * @since 1.6
046 */
047public class InvalidRequestFilter extends AccessControlFilter {
048
049    private static final List<String> SEMICOLON = Collections.unmodifiableList(Arrays.asList(";", "%3b", "%3B"));
050
051    private static final List<String> BACKSLASH = Collections.unmodifiableList(Arrays.asList("\\", "%5c", "%5C"));
052
053    private static final List<String> FORWARDSLASH = Collections.unmodifiableList(Arrays.asList("%2f", "%2F"));
054
055    private static final List<String> PERIOD = Collections.unmodifiableList(Arrays.asList("%2e", "%2E"));
056
057    private boolean blockSemicolon = true;
058
059    private boolean blockBackslash = !Boolean.getBoolean(WebUtils.ALLOW_BACKSLASH);
060
061    private boolean blockNonAscii = true;
062
063    private boolean blockTraversal = true;
064
065    private boolean blockEncodedPeriod = true;
066
067    private boolean blockEncodedForwardSlash = true;
068
069    private boolean blockRewriteTraversal = true;
070
071    @Override
072    protected boolean isAccessAllowed(ServletRequest req, ServletResponse response, Object mappedValue) throws Exception {
073        HttpServletRequest request = WebUtils.toHttp(req);
074        // check the original and decoded values
075        return isValid(request.getRequestURI())      // user request string (not decoded)
076                && isValid(request.getServletPath()) // decoded servlet part
077                && isValid(request.getPathInfo());   // decoded path info (may be null)
078    }
079
080    private boolean isValid(String uri) {
081        return !StringUtils.hasText(uri)
082               || ( !containsSemicolon(uri)
083                 && !containsBackslash(uri)
084                 && !containsNonAsciiCharacters(uri)
085                 && !containsTraversal(uri)
086                 && !containsEncodedPeriods(uri)
087                 && !containsEncodedForwardSlash(uri));
088    }
089
090    @Override
091    protected boolean onAccessDenied(ServletRequest request, ServletResponse response) throws Exception {
092        WebUtils.toHttp(response).sendError(400, "Invalid request");
093        return false;
094    }
095
096    private boolean containsSemicolon(String uri) {
097        if (isBlockSemicolon()) {
098            return SEMICOLON.stream().anyMatch(uri::contains);
099        }
100        return false;
101    }
102
103    private boolean containsBackslash(String uri) {
104        if (isBlockBackslash()) {
105            return BACKSLASH.stream().anyMatch(uri::contains);
106        }
107        return false;
108    }
109
110    private boolean containsNonAsciiCharacters(String uri) {
111        if (isBlockNonAscii()) {
112            return !containsOnlyPrintableAsciiCharacters(uri);
113        }
114        return false;
115    }
116
117    private static boolean containsOnlyPrintableAsciiCharacters(String uri) {
118        int length = uri.length();
119        for (int i = 0; i < length; i++) {
120            char c = uri.charAt(i);
121            if (c < '\u0020' || c > '\u007e') {
122                return false;
123            }
124        }
125        return true;
126    }
127
128    private boolean containsTraversal(String uri) {
129        if (isBlockTraversal()) {
130            return !isNormalized(uri)
131                || (isBlockRewriteTraversal() && Stream.of("/..;", "/.;").anyMatch(uri::contains));
132        }
133        return false;
134    }
135
136    private boolean containsEncodedPeriods(String uri) {
137        if (isBlockEncodedPeriod()) {
138            return PERIOD.stream().anyMatch(uri::contains);
139        }
140        return false;
141    }
142
143    private boolean containsEncodedForwardSlash(String uri) {
144        if (isBlockEncodedForwardSlash()) {
145            return FORWARDSLASH.stream().anyMatch(uri::contains);
146        }
147        return false;
148    }
149
150    /**
151     * Checks whether a path is normalized (doesn't contain path traversal sequences like
152     * "./", "/../" or "/.")
153     * @param path the path to test
154     * @return true if the path doesn't contain any path-traversal character sequences.
155     */
156    private boolean isNormalized(String path) {
157        if (path == null) {
158            return true;
159        }
160        for (int i = path.length(); i > 0;) {
161            int slashIndex = path.lastIndexOf('/', i - 1);
162            int gap = i - slashIndex;
163            if (gap == 2 && path.charAt(slashIndex + 1) == '.') {
164                return false; // ".", "/./" or "/."
165            }
166            if (gap == 3 && path.charAt(slashIndex + 1) == '.' && path.charAt(slashIndex + 2) == '.') {
167                return false;
168            }
169            i = slashIndex;
170        }
171        return true;
172    }
173
174    public boolean isBlockSemicolon() {
175        return blockSemicolon;
176    }
177
178    public void setBlockSemicolon(boolean blockSemicolon) {
179        this.blockSemicolon = blockSemicolon;
180    }
181
182    public boolean isBlockBackslash() {
183        return blockBackslash;
184    }
185
186    public void setBlockBackslash(boolean blockBackslash) {
187        this.blockBackslash = blockBackslash;
188    }
189
190    public boolean isBlockNonAscii() {
191        return blockNonAscii;
192    }
193
194    public void setBlockNonAscii(boolean blockNonAscii) {
195        this.blockNonAscii = blockNonAscii;
196    }
197
198    public boolean isBlockTraversal() {
199        return blockTraversal;
200    }
201
202    public void setBlockTraversal(boolean blockTraversal) {
203        this.blockTraversal = blockTraversal;
204    }
205
206    public boolean isBlockEncodedPeriod() {
207        return blockEncodedPeriod;
208    }
209
210    public void setBlockEncodedPeriod(boolean blockEncodedPeriod) {
211        this.blockEncodedPeriod = blockEncodedPeriod;
212    }
213
214    public boolean isBlockEncodedForwardSlash() {
215        return blockEncodedForwardSlash;
216    }
217
218    public void setBlockEncodedForwardSlash(boolean blockEncodedForwardSlash) {
219        this.blockEncodedForwardSlash = blockEncodedForwardSlash;
220    }
221
222    public boolean isBlockRewriteTraversal() {
223        return blockRewriteTraversal;
224    }
225
226    public void setBlockRewriteTraversal(boolean blockRewriteTraversal) {
227        this.blockRewriteTraversal = blockRewriteTraversal;
228    }
229}
