019package org.apache.shiro.web.tags;
020
021import javax.servlet.jsp.JspException;
022import javax.servlet.jsp.tagext.TagSupport;
023
024import org.slf4j.Logger;
025import org.slf4j.LoggerFactory;
026
027
028/**
029 * JSP tag that renders the tag body if the current user <em>is not</em> known to the system, either because they
030 * haven't logged in yet, or because they have no 'RememberMe' identity.
031 *
032 * <p>The logically opposite tag of this one is the {@link UserTag}.  Please read that class's JavaDoc as it explains
033 * more about the differences between Authenticated/Unauthenticated and User/Guest semantic differences.
034 *
035 * @since 0.9
036 */
037public class GuestTag extends SecureTag {
038
039    //TODO - complete JavaDoc
040
041    private static final Logger log = LoggerFactory.getLogger(GuestTag.class);
042
043    public int onDoStartTag() throws JspException {
044        if (getSubject() == null || getSubject().getPrincipal() == null) {
045            if (log.isTraceEnabled()) {
046                log.trace("Subject does not exist or does not have a known identity (aka 'principal').  " +
047                        "Tag body will be evaluated.");
048            }
049            return TagSupport.EVAL_BODY_INCLUDE;
050        } else {
051            if (log.isTraceEnabled()) {
052                log.trace("Subject exists or has a known identity (aka 'principal').  " +
053                        "Tag body will not be evaluated.");
054            }
055            return TagSupport.SKIP_BODY;
056        }
057    }
058
059}
