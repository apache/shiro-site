019package org.apache.shiro.web.servlet;
020
021import org.slf4j.Logger;
022import org.slf4j.LoggerFactory;
023
024import javax.servlet.FilterChain;
025import javax.servlet.ServletException;
026import javax.servlet.ServletRequest;
027import javax.servlet.ServletResponse;
028import java.io.IOException;
029
030/**
031 * Filter base class that guarantees to be just executed once per request,
032 * on any servlet container. It provides a {@link #doFilterInternal}
033 * method with HttpServletRequest and HttpServletResponse arguments.
034 * <p/>
035 * The {@link #getAlreadyFilteredAttributeName} method determines how
036 * to identify that a request is already filtered. The default implementation
037 * is based on the configured name of the concrete filter instance.
038 * <h3>Controlling filter execution</h3>
039 * 1.2 introduced the {@link #isEnabled(javax.servlet.ServletRequest, javax.servlet.ServletResponse)} method and
040 * {@link #isEnabled()} property to allow explicit control over whether the filter executes (or allows passthrough)
041 * for any given request.
042 * <p/>
043 * <b>NOTE</b> This class was initially borrowed from the Spring framework but has continued modifications.
044 *
045 * @since 0.1
046 */
047public abstract class OncePerRequestFilter extends NameableFilter {
048
049    /**
050     * Private internal log instance.
051     */
052    private static final Logger log = LoggerFactory.getLogger(OncePerRequestFilter.class);
053
054    /**
055     * Suffix that gets appended to the filter name for the "already filtered" request attribute.
056     *
057     * @see #getAlreadyFilteredAttributeName
058     */
059    public static final String ALREADY_FILTERED_SUFFIX = ".FILTERED";
060
061    /**
062     * Determines generally if this filter should execute or let requests fall through to the next chain element.
063     *
064     * @see #isEnabled()
065     */
066    private boolean enabled = true; //most filters wish to execute when configured, so default to true
067
068    /**
069     * Determines if the filter's once per request functionality is enabled, defaults to false. It is recommended
070     * to leave this disabled if you are using a {@link javax.servlet.RequestDispatcher RequestDispatcher} to forward
071     * or include request (JSP tags, programmatically, or via a framework).
072     */
073    private boolean filterOncePerRequest = false;
074
075    /**
076     * Returns {@code true} if this filter should <em>generally</em><b>*</b> execute for any request,
077     * {@code false} if it should let the request/response pass through immediately to the next
078     * element in the {@link FilterChain}.  The default value is {@code true}, as most filters would inherently need
079     * to execute when configured.
080     * <p/>
081     * <b>*</b> This configuration property is for general configuration for any request that comes through
082     * the filter.  The
083     * {@link #isEnabled(javax.servlet.ServletRequest, javax.servlet.ServletResponse) isEnabled(request,response)}
084     * method actually determines whether or not if the filter is enabled based on the current request.
085     *
086     * @return {@code true} if this filter should <em>generally</em> execute, {@code false} if it should let the
087     * request/response pass through immediately to the next element in the {@link FilterChain}.
088     * @since 1.2
089     */
090    public boolean isEnabled() {
091        return enabled;
092    }
093
094    /**
095     * Sets whether or not this filter <em>generally</em> executes for any request.  See the
096     * {@link #isEnabled() isEnabled()} JavaDoc as to what <em>general</em> execution means.
097     *
098     * @param enabled whether or not this filter <em>generally</em> executes.
099     * @since 1.2
100     */
101    public void setEnabled(boolean enabled) {
102        this.enabled = enabled;
103    }
104
105    /**
106     * Returns {@code true} if this filter should only execute once per request. If set to {@code false} this filter
107     * will execute each time it is invoked.
108     * @return {@code true} if this filter should only execute once per request.
109     * @since 1.10
110     */
111    public boolean isFilterOncePerRequest() {
112        return filterOncePerRequest;
113    }
114
115    /**
116     * Sets whether this filter executes once per request or for every invocation of the filter. It is recommended
117     * to leave this disabled if you are using a {@link javax.servlet.RequestDispatcher RequestDispatcher} to forward
118     * or include request (JSP tags, programmatically, or via a framework). 
119     *
120     * @param filterOncePerRequest Whether this filter executes once per request.
121     * @since 1.10
122     */
123    public void setFilterOncePerRequest(boolean filterOncePerRequest) {
124        this.filterOncePerRequest = filterOncePerRequest;
125    }
126
127    /**
128     * This {@code doFilter} implementation stores a request attribute for
129     * "already filtered", proceeding without filtering again if the
130     * attribute is already there.
131     *
132     * @see #getAlreadyFilteredAttributeName
133     * @see #shouldNotFilter
134     * @see #doFilterInternal
135     */
136    public final void doFilter(ServletRequest request, ServletResponse response, FilterChain filterChain)
137            throws ServletException, IOException {
138        String alreadyFilteredAttributeName = getAlreadyFilteredAttributeName();
139        if ( request.getAttribute(alreadyFilteredAttributeName) != null && filterOncePerRequest) {
140            log.trace("Filter '{}' already executed.  Proceeding without invoking this filter.", getName());
141            filterChain.doFilter(request, response);
142        } else //noinspection deprecation
143            if (/* added in 1.2: */ !isEnabled(request, response) ||
144                /* retain backwards compatibility: */ shouldNotFilter(request) ) {
145            log.debug("Filter '{}' is not enabled for the current request.  Proceeding without invoking this filter.",
146                    getName());
147            filterChain.doFilter(request, response);
148        } else {
149            // Do invoke this filter...
150            log.trace("Filter '{}' not yet executed.  Executing now.", getName());
151            request.setAttribute(alreadyFilteredAttributeName, Boolean.TRUE);
152
153            try {
154                doFilterInternal(request, response, filterChain);
155            } finally {
156                // Once the request has finished, we're done and we don't
157                // need to mark as 'already filtered' any more.
158                request.removeAttribute(alreadyFilteredAttributeName);
159            }
160        }
161    }
162
163    /**
164     * Returns {@code true} if this filter should filter the specified request, {@code false} if it should let the
165     * request/response pass through immediately to the next element in the {@code FilterChain}.
166     * <p/>
167     * This default implementation merely returns the value of {@link #isEnabled() isEnabled()}, which is
168     * {@code true} by default (to ensure the filter always executes by default), but it can be overridden by
169     * subclasses for request-specific behavior if necessary.  For example, a filter could be enabled or disabled
170     * based on the request path being accessed.
171     * <p/>
172     * <b>Helpful Hint:</b> if your subclass extends {@link org.apache.shiro.web.filter.PathMatchingFilter PathMatchingFilter},
173     * you may wish to instead override the
174     * {@link org.apache.shiro.web.filter.PathMatchingFilter#isEnabled(javax.servlet.ServletRequest, javax.servlet.ServletResponse, String, Object)
175     * PathMatchingFilter.isEnabled(request,response,path,pathSpecificConfig)}
176     * method if you want to make your enable/disable decision based on any path-specific configuration.
177     *
178     * @param request the incoming servlet request
179     * @param response the outbound servlet response
180     * @return {@code true} if this filter should filter the specified request, {@code false} if it should let the
181     * request/response pass through immediately to the next element in the {@code FilterChain}.
182     * @throws IOException in the case of any IO error
183     * @throws ServletException in the case of any error
184     * @see org.apache.shiro.web.filter.PathMatchingFilter#isEnabled(javax.servlet.ServletRequest, javax.servlet.ServletResponse, String, Object)
185     * @since 1.2
186     */
187    @SuppressWarnings({"UnusedParameters"})
188    protected boolean isEnabled(ServletRequest request, ServletResponse response) throws ServletException, IOException {
189        return isEnabled();
190    }
191
192    /**
193     * Return name of the request attribute that identifies that a request has already been filtered.
194     * <p/>
195     * The default implementation takes the configured {@link #getName() name} and appends &quot;{@code .FILTERED}&quot;.
196     * If the filter is not fully initialized, it falls back to the implementation's class name.
197     *
198     * @return the name of the request attribute that identifies that a request has already been filtered.
199     * @see #getName
200     * @see #ALREADY_FILTERED_SUFFIX
201     */
202    protected String getAlreadyFilteredAttributeName() {
203        String name = getName();
204        if (name == null) {
205            name = getClass().getName();
206        }
207        return name + ALREADY_FILTERED_SUFFIX;
208    }
209
210    /**
211     * Can be overridden in subclasses for custom filtering control,
212     * returning <code>true</code> to avoid filtering of the given request.
213     * <p>The default implementation always returns <code>false</code>.
214     *
215     * @param request current HTTP request
216     * @return whether the given request should <i>not</i> be filtered
217     * @throws ServletException in case of errors
218     * @deprecated in favor of overriding {@link #isEnabled(javax.servlet.ServletRequest, javax.servlet.ServletResponse)}
219     * for custom behavior.  This method will be removed in Shiro 2.0.
220     */
221    @Deprecated
222    @SuppressWarnings({"UnusedDeclaration"})
223    protected boolean shouldNotFilter(ServletRequest request) throws ServletException {
224        return false;
225    }
226
227
228    /**
229     * Same contract as for
230     * {@link #doFilter(javax.servlet.ServletRequest, javax.servlet.ServletResponse, javax.servlet.FilterChain)},
231     * but guaranteed to be invoked only once per request.
232     *
233     * @param request  incoming {@code ServletRequest}
234     * @param response outgoing {@code ServletResponse}
235     * @param chain    the {@code FilterChain} to execute
236     * @throws ServletException if there is a problem processing the request
237     * @throws IOException      if there is an I/O problem processing the request
238     */
239    protected abstract void doFilterInternal(ServletRequest request, ServletResponse response, FilterChain chain)
240            throws ServletException, IOException;
241}
