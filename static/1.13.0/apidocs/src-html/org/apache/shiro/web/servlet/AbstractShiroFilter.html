042/**
043 * Abstract base class that provides all standard Shiro request filtering behavior and expects
044 * subclasses to implement configuration-specific logic (INI, XML, .properties, etc).
045 * <p/>
046 * Subclasses should perform configuration and construction logic in an overridden
047 * {@link #init()} method implementation.  That implementation should make available any constructed
048 * {@code SecurityManager} and {@code FilterChainResolver} by calling
049 * {@link #setSecurityManager(org.apache.shiro.web.mgt.WebSecurityManager)} and
050 * {@link #setFilterChainResolver(org.apache.shiro.web.filter.mgt.FilterChainResolver)} methods respectively.
051 * <h3>Static SecurityManager</h3>
052 * By default the {@code SecurityManager} instance enabled by this filter <em>will not</em> be enabled in static
053 * memory via the {@code SecurityUtils.}{@link SecurityUtils#setSecurityManager(org.apache.shiro.mgt.SecurityManager) setSecurityManager}
054 * method.  Instead, it is expected that Subject instances will always be constructed on a request-processing thread
055 * via instances of this Filter class.
056 * <p/>
057 * However, if you need to construct {@code Subject} instances on separate (non request-processing) threads, it might
058 * be easiest to enable the SecurityManager to be available in static memory via the
059 * {@link SecurityUtils#getSecurityManager()} method.  You can do this by additionally specifying an {@code init-param}:
060 * <pre>
061 * &lt;filter&gt;
062 *     ... other config here ...
063 *     &lt;init-param&gt;
064 *         &lt;param-name&gt;staticSecurityManagerEnabled&lt;/param-name&gt;
065 *         &lt;param-value&gt;true&lt;/param-value&gt;
066 *     &lt;/init-param&gt;
067 * &lt;/filter&gt;
068 * </pre>
069 * See the Shiro <a href="http://shiro.apache.org/subject.html">Subject documentation</a> for more information as to
070 * if you would do this, particularly the sections on the {@code Subject.Builder} and Thread Association.
071 *
072 * @since 1.0
073 * @see <a href="http://shiro.apache.org/subject.html">Subject documentation</a>
074 */
075public abstract class AbstractShiroFilter extends OncePerRequestFilter {
076

144     * <p/>
145     * The default value is {@code false}.
146     *
147     * @param staticSecurityManagerEnabled if the constructed {@link #getSecurityManager() securityManager} reference

157    protected final void onFilterConfigSet() throws Exception {
158        //added in 1.2 for SHIRO-287:
159        applyStaticSecurityManagerEnabledConfig();
160        init();
161        ensureSecurityManager();
162        //added in 1.2 for SHIRO-287:
163        if (isStaticSecurityManagerEnabled()) {
164            SecurityUtils.setSecurityManager(getSecurityManager());
165        }
166    }

168    /**
169     * Checks if the init-param that configures the filter to use static memory has been configured, and if so,
170     * sets the {@link #setStaticSecurityManagerEnabled(boolean)} attribute with the configured value.
171     *
172     * @since 1.2
173     * @see <a href="https://issues.apache.org/jira/browse/SHIRO-287">SHIRO-287</a>
174     */
175    private void applyStaticSecurityManagerEnabledConfig() {
176        String value = getInitParam(STATIC_INIT_PARAM_NAME);
177        if (value != null) {
178            Boolean b = Boolean.valueOf(value);
179            if (b != null) {
180                setStaticSecurityManagerEnabled(b);
181            }
182        }
183    }

186    }

188    /**
189     * A fallback mechanism called in {@link #onFilterConfigSet()} to ensure that the
190     * {@link #getSecurityManager() securityManager} property has been set by configuration, and if not,
191     * creates one automatically.
192     */
193    private void ensureSecurityManager() {
194        WebSecurityManager securityManager = getSecurityManager();
195        if (securityManager == null) {
196            log.info("No SecurityManager configured.  Creating default.");
197            securityManager = createDefaultSecurityManager();
198            setSecurityManager(securityManager);
199        }
200    }

210    /**
211     * Wraps the original HttpServletRequest in a {@link ShiroHttpServletRequest}, which is required for supporting
212     * Servlet Specification behavior backed by a {@link org.apache.shiro.subject.Subject Subject} instance.
213     *
214     * @param orig the original Servlet Container-provided incoming {@code HttpServletRequest} instance.
215     * @return {@link ShiroHttpServletRequest ShiroHttpServletRequest} instance wrapping the original.
216     * @since 1.0
217     */
218    protected ServletRequest wrapServletRequest(HttpServletRequest orig) {
219        return new ShiroHttpServletRequest(orig, getServletContext(), isHttpSessions());
220    }

222    /**
223     * Prepares the {@code ServletRequest} instance that will be passed to the {@code FilterChain} for request
224     * processing.
225     * <p/>
226     * If the {@code ServletRequest} is an instance of {@link HttpServletRequest}, the value returned from this method
227     * is obtained by calling {@link #wrapServletRequest(javax.servlet.http.HttpServletRequest)} to allow Shiro-specific
228     * HTTP behavior, otherwise the original {@code ServletRequest} argument is returned.
229     *
230     * @param request  the incoming ServletRequest
231     * @param response the outgoing ServletResponse
232     * @param chain    the Servlet Container provided {@code FilterChain} that will receive the returned request.
233     * @return the {@code ServletRequest} instance that will be passed to the {@code FilterChain} for request processing.
234     * @since 1.0
235     */
236    @SuppressWarnings({"UnusedDeclaration"})
237    protected ServletRequest prepareServletRequest(ServletRequest request, ServletResponse response, FilterChain chain) {
238        ServletRequest toUse = request;
239        if (request instanceof HttpServletRequest) {
240            HttpServletRequest http = (HttpServletRequest) request;
241            toUse = wrapServletRequest(http);
242        }
243        return toUse;
244    }

246    /**
247     * Returns a new {@link ShiroHttpServletResponse} instance, wrapping the {@code orig} argument, in order to provide
248     * correct URL rewriting behavior required by the Servlet Specification when using Shiro-based sessions (and not
249     * Servlet Container HTTP-based sessions).
250     *
251     * @param orig    the original {@code HttpServletResponse} instance provided by the Servlet Container.
252     * @param request the {@code ShiroHttpServletRequest} instance wrapping the original request.
253     * @return the wrapped ServletResponse instance to use during {@link FilterChain} execution.
254     * @since 1.0
255     */
256    protected ServletResponse wrapServletResponse(HttpServletResponse orig, ShiroHttpServletRequest request) {
257        return new ShiroHttpServletResponse(orig, getServletContext(), request);
258    }

260    /**
261     * Prepares the {@code ServletResponse} instance that will be passed to the {@code FilterChain} for request
262     * processing.
263     * <p/>
264     * This implementation delegates to {@link #wrapServletRequest(javax.servlet.http.HttpServletRequest)}
265     * only if Shiro-based sessions are enabled (that is, !{@link #isHttpSessions()}) and the request instance is a
266     * {@link ShiroHttpServletRequest}.  This ensures that any URL rewriting that occurs is handled correctly using the
267     * Shiro-managed Session's sessionId and not a servlet container session ID.
268     * <p/>
269     * If HTTP-based sessions are enabled (the default), then this method does nothing and just returns the
270     * {@code ServletResponse} argument as-is, relying on the default Servlet Container URL rewriting logic.
271     *
272     * @param request  the incoming ServletRequest
273     * @param response the outgoing ServletResponse
274     * @param chain    the Servlet Container provided {@code FilterChain} that will receive the returned request.
275     * @return the {@code ServletResponse} instance that will be passed to the {@code FilterChain} during request processing.
276     * @since 1.0
277     */
278    @SuppressWarnings({"UnusedDeclaration"})
279    protected ServletResponse prepareServletResponse(ServletRequest request, ServletResponse response, FilterChain chain) {
280        ServletResponse toUse = response;
281        if (!isHttpSessions() && (request instanceof ShiroHttpServletRequest) &&
282                (response instanceof HttpServletResponse)) {
283            //the ShiroHttpServletResponse exists to support URL rewriting for session ids.  This is only needed if
284            //using Shiro sessions (i.e. not simple HttpSession based sessions):
285            toUse = wrapServletResponse((HttpServletResponse) response, (ShiroHttpServletRequest) request);
286        }
287        return toUse;
288    }

290    /**
291     * Creates a {@link WebSubject} instance to associate with the incoming request/response pair which will be used
292     * throughout the request/response execution.
293     *
294     * @param request  the incoming {@code ServletRequest}
295     * @param response the outgoing {@code ServletResponse}
296     * @return the {@code WebSubject} instance to associate with the request/response execution
297     * @since 1.0
298     */
299    protected WebSubject createSubject(ServletRequest request, ServletResponse response) {
300        return new WebSubject.Builder(getSecurityManager(), request, response).buildWebSubject();
301    }

303    /**
304     * Updates any 'native'  Session's last access time that might exist to the timestamp when this method is called.
305     * If native sessions are not enabled (that is, standard Servlet container sessions are being used) or there is no
306     * session ({@code subject.getSession(false) == null}), this method does nothing.
307     * <p/>This method implementation merely calls
308     * <code>Session.{@link org.apache.shiro.session.Session#touch() touch}()</code> on the session.
309     *
310     * @param request  incoming request - ignored, but available to subclasses that might wish to override this method
311     * @param response outgoing response - ignored, but available to subclasses that might wish to override this method
312     * @since 1.0
313     */
314    @SuppressWarnings({"UnusedDeclaration"})
315    protected void updateSessionLastAccessTime(ServletRequest request, ServletResponse response) {
316        if (!isHttpSessions()) { //'native' sessions
317            Subject subject = SecurityUtils.getSubject();
318            //Subject should never _ever_ be null, but just in case:
319            if (subject != null) {
320                Session session = subject.getSession(false);
321                if (session != null) {
322                    try {
323                        session.touch();
324                    } catch (Throwable t) {
325                        log.error("session.touch() method invocation has failed.  Unable to update " +
326                                "the corresponding session's last access time based on the incoming request.", t);
327                    }
328                }
329            }
330        }
331    }

333    /**
334     * {@code doFilterInternal} implementation that sets-up, executes, and cleans-up a Shiro-filtered request.  It
335     * performs the following ordered operations:
336     * <ol>
337     * <li>{@link #prepareServletRequest(ServletRequest, ServletResponse, FilterChain) Prepares}
338     * the incoming {@code ServletRequest} for use during Shiro's processing</li>
339     * <li>{@link #prepareServletResponse(ServletRequest, ServletResponse, FilterChain) Prepares}
340     * the outgoing {@code ServletResponse} for use during Shiro's processing</li>
341     * <li> {@link #createSubject(javax.servlet.ServletRequest, javax.servlet.ServletResponse) Creates} a
342     * {@link Subject} instance based on the specified request/response pair.</li>
343     * <li>Finally {@link Subject#execute(Runnable) executes} the
344     * {@link #updateSessionLastAccessTime(javax.servlet.ServletRequest, javax.servlet.ServletResponse)} and
345     * {@link #executeChain(javax.servlet.ServletRequest, javax.servlet.ServletResponse, javax.servlet.FilterChain)}
346     * methods</li>
347     * </ol>
348     * <p/>
349     * The {@code Subject.}{@link Subject#execute(Runnable) execute(Runnable)} call in step #4 is used as an
350     * implementation technique to guarantee proper thread binding and restoration is completed successfully.
351     *
352     * @param servletRequest  the incoming {@code ServletRequest}
353     * @param servletResponse the outgoing {@code ServletResponse}
354     * @param chain           the container-provided {@code FilterChain} to execute
355     * @throws IOException                    if an IO error occurs
356     * @throws javax.servlet.ServletException if an Throwable other than an IOException
357     */
358    protected void doFilterInternal(ServletRequest servletRequest, ServletResponse servletResponse, final FilterChain chain)
359            throws ServletException, IOException {
360
361        Throwable t = null;
362
363        try {
364            final ServletRequest request = prepareServletRequest(servletRequest, servletResponse, chain);
365            final ServletResponse response = prepareServletResponse(request, servletResponse, chain);
366
367            final Subject subject = createSubject(request, response);
368
369            //noinspection unchecked
370            subject.execute(new Callable() {
371                public Object call() throws Exception {
372                    updateSessionLastAccessTime(request, response);
373                    executeChain(request, response, chain);
374                    return null;
375                }
376            });
377        } catch (ExecutionException ex) {
378            t = ex.getCause();
379        } catch (Throwable throwable) {
380            t = throwable;
381        }
382
383        if (t != null) {
384            if (t instanceof ServletException) {
385                throw (ServletException) t;
386            }
387            if (t instanceof IOException) {
388                throw (IOException) t;
389            }
390            //otherwise it's not one of the two exceptions expected by the filter method signature - wrap it in one:
391            String msg = "Filtered request failed.";
392            throw new ServletException(msg, t);
393        }
394    }

396    /**
397     * Returns the {@code FilterChain} to execute for the given request.
398     * <p/>
399     * The {@code origChain} argument is the
400     * original {@code FilterChain} supplied by the Servlet Container, but it may be modified to provide
401     * more behavior by pre-pending further chains according to the Shiro configuration.
402     * <p/>
403     * This implementation returns the chain that will actually be executed by acquiring the chain from a
404     * {@link #getFilterChainResolver() filterChainResolver}.  The resolver determines exactly which chain to
405     * execute, typically based on URL configuration.  If no chain is returned from the resolver call
406     * (returns {@code null}), then the {@code origChain} will be returned by default.
407     *
408     * @param request   the incoming ServletRequest
409     * @param response  the outgoing ServletResponse
410     * @param origChain the original {@code FilterChain} provided by the Servlet Container
411     * @return the {@link FilterChain} to execute for the given request
412     * @since 1.0
413     */
414    protected FilterChain getExecutionChain(ServletRequest request, ServletResponse response, FilterChain origChain) {
415
416        FilterChain chain = origChain;
417
418        FilterChainResolver resolver = getFilterChainResolver();
419        if (resolver == null) {
420            log.debug("No FilterChainResolver configured.  Returning original FilterChain.");
421            return origChain;
422        }
423
424        FilterChain resolved = resolver.getChain(request, response, origChain);
425        if (resolved != null) {
426            log.trace("Resolved a configured FilterChain for the current request.");
427            chain = resolved;
428        } else {
429            log.trace("No FilterChain configured for the current request.  Using the default.");
430        }
431
432        return chain;
433    }

435    /**
436     * Executes a {@link FilterChain} for the given request.
437     * <p/>
438     * This implementation first delegates to
439     * <code>{@link #getExecutionChain(javax.servlet.ServletRequest, javax.servlet.ServletResponse, javax.servlet.FilterChain) getExecutionChain}</code>
440     * to allow the application's Shiro configuration to determine exactly how the chain should execute.  The resulting
441     * value from that call is then executed directly by calling the returned {@code FilterChain}'s
442     * {@link FilterChain#doFilter doFilter} method.  That is:
443     * <pre>
444     * FilterChain chain = {@link #getExecutionChain}(request, response, origChain);
445     * chain.{@link FilterChain#doFilter doFilter}(request,response);</pre>
446     *
447     * @param request   the incoming ServletRequest
448     * @param response  the outgoing ServletResponse
449     * @param origChain the Servlet Container-provided chain that may be wrapped further by an application-configured
450     *                  chain of Filters.
451     * @throws IOException      if the underlying {@code chain.doFilter} call results in an IOException
452     * @throws ServletException if the underlying {@code chain.doFilter} call results in a ServletException
453     * @since 1.0
454     */
455    protected void executeChain(ServletRequest request, ServletResponse response, FilterChain origChain)
456            throws IOException, ServletException {
457        FilterChain chain = getExecutionChain(request, response, origChain);
458        chain.doFilter(request, response);
459    }
