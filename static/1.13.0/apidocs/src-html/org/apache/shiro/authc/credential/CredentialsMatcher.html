019package org.apache.shiro.authc.credential;
020
021import org.apache.shiro.authc.AuthenticationInfo;
022import org.apache.shiro.authc.AuthenticationToken;
023
024
025/**
026 * Interface implemented by classes that can determine if an AuthenticationToken's provided
027 * credentials matches a corresponding account's credentials stored in the system.
028 *
029 * <p>Simple direct comparisons are handled well by the
030 * {@link SimpleCredentialsMatcher SimpleCredentialsMatcher}.  If you
031 * hash user's credentials before storing them in a realm (a common practice), look at the
032 * {@link HashedCredentialsMatcher HashedCredentialsMatcher} implementations,
033 * as they support this scenario.
034 *
035 * @see SimpleCredentialsMatcher
036 * @see AllowAllCredentialsMatcher
037 * @see Md5CredentialsMatcher
038 * @see Sha1CredentialsMatcher
039 * @since 0.1
040 */
041public interface CredentialsMatcher {
042
043    /**
044     * Returns {@code true} if the provided token credentials match the stored account credentials,
045     * {@code false} otherwise.
046     *
047     * @param token   the {@code AuthenticationToken} submitted during the authentication attempt
048     * @param info the {@code AuthenticationInfo} stored in the system.
049     * @return {@code true} if the provided token credentials match the stored account credentials,
050     *         {@code false} otherwise.
051     */
052    boolean doCredentialsMatch(AuthenticationToken token, AuthenticationInfo info);
053
054}
