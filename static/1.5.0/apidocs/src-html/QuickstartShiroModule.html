019import com.google.inject.Provides;
020import org.apache.shiro.config.Ini;
021import org.apache.shiro.guice.ShiroModule;
022import org.apache.shiro.realm.text.IniRealm;
023
024public class QuickstartShiroModule extends ShiroModule {
025    protected void configureShiro() {
026        try {
027            bindRealm().toConstructor(IniRealm.class.getConstructor(Ini.class));
028        } catch (NoSuchMethodException e) {
029            addError(e);
030        }
031    }
032
033    @Provides
034    Ini loadShiroIni() {
035        return Ini.fromResourcePath("classpath:shiro.ini");
036    }
037}
