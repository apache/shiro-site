19  package org.apache.shiro.web.filter.mgt;
20  
21  import org.apache.shiro.util.AntPathMatcher;
22  import org.apache.shiro.util.PatternMatcher;
23  import org.apache.shiro.web.util.WebUtils;
24  import org.owasp.encoder.Encode;
25  import org.owasp.encoder.Encoder;
26  import org.slf4j.Logger;
27  import org.slf4j.LoggerFactory;
28  
29  import javax.servlet.FilterChain;
30  import javax.servlet.FilterConfig;
31  import javax.servlet.ServletRequest;
32  import javax.servlet.ServletResponse;
33  
34  /**
35   * A {@code FilterChainResolver} that resolves {@link FilterChain}s based on url path
36   * matching, as determined by a configurable {@link #setPathMatcher(org.apache.shiro.util.PatternMatcher) PathMatcher}.
37   * <p/>
38   * This implementation functions by consulting a {@link org.apache.shiro.web.filter.mgt.FilterChainManager} for all configured filter chains (keyed
39   * by configured path pattern).  If an incoming Request path matches one of the configured path patterns (via
40   * the {@code PathMatcher}, the corresponding configured {@code FilterChain} is returned.
41   *
42   * @since 1.0
43   */
44  public class PathMatchingFilterChainResolver implements FilterChainResolver {
45  
46      private static transient final Logger log = LoggerFactory.getLogger(PathMatchingFilterChainResolver.class);
47  
48      private FilterChainManager filterChainManager;
49  
50      private PatternMatcher pathMatcher;
51  
52      private static final String DEFAULT_PATH_SEPARATOR = "/";
53  
54      public PathMatchingFilterChainResolver() {
55          this.pathMatcher = new AntPathMatcher();
56          this.filterChainManager = new DefaultFilterChainManager();
57      }
58  
59      public PathMatchingFilterChainResolver(FilterConfig filterConfig) {
60          this.pathMatcher = new AntPathMatcher();
61          this.filterChainManager = new DefaultFilterChainManager(filterConfig);
62      }
63  
64      /**
65       * Returns the {@code PatternMatcher} used when determining if an incoming request's path
66       * matches a configured filter chain.  Unless overridden, the
67       * default implementation is an {@link org.apache.shiro.util.AntPathMatcher AntPathMatcher}.
68       *
69       * @return the {@code PatternMatcher} used when determining if an incoming request's path
70       *         matches a configured filter chain.
71       */
72      public PatternMatcher getPathMatcher() {
73          return pathMatcher;
74      }
75  
76      /**
77       * Sets the {@code PatternMatcher} used when determining if an incoming request's path
78       * matches a configured filter chain.  Unless overridden, the
79       * default implementation is an {@link org.apache.shiro.util.AntPathMatcher AntPathMatcher}.
80       *
81       * @param pathMatcher the {@code PatternMatcher} used when determining if an incoming request's path
82       *                    matches a configured filter chain.
83       */
84      public void setPathMatcher(PatternMatcher pathMatcher) {
85          this.pathMatcher = pathMatcher;
86      }
87  
88      public FilterChainManager getFilterChainManager() {
89          return filterChainManager;
90      }
91  
92      @SuppressWarnings({"UnusedDeclaration"})
93      public void setFilterChainManager(FilterChainManager filterChainManager) {
94          this.filterChainManager = filterChainManager;
95      }
96  
97      public FilterChain getChain(ServletRequest request, ServletResponse response, FilterChain originalChain) {
98          FilterChainManager filterChainManager = getFilterChainManager();
99          if (!filterChainManager.hasChains()) {
100             return null;
101         }
102 
103         final String requestURI = getPathWithinApplication(request);
104         final String requestURINoTrailingSlash = removeTrailingSlash(requestURI);
105 
106         //the 'chain names' in this implementation are actually path patterns defined by the user.  We just use them
107         //as the chain name for the FilterChainManager's requirements
108         for (String pathPattern : filterChainManager.getChainNames()) {
109             // If the path does match, then pass on to the subclass implementation for specific checks:
110             if (pathMatches(pathPattern, requestURI)) {
111                 if (log.isTraceEnabled()) {
112                     log.trace("Matched path pattern [{}] for requestURI [{}].  " +
113                             "Utilizing corresponding filter chain...", pathPattern, Encode.forHtml(requestURI));
114                 }
115                 return filterChainManager.proxy(originalChain, pathPattern);
116             } else {
117 
118                 // in spring web, the requestURI "/resource/menus" ---- "resource/menus/" bose can access the resource
119                 // but the pathPattern match "/resource/menus" can not match "resource/menus/"
120                 // user can use requestURI + "/" to simply bypassed chain filter, to bypassed shiro protect
121 
122                 pathPattern = removeTrailingSlash(pathPattern);
123 
124                 if (pathMatches(pathPattern, requestURINoTrailingSlash)) {
125                     if (log.isTraceEnabled()) {
126                         log.trace("Matched path pattern [{}] for requestURI [{}].  " +
127                                   "Utilizing corresponding filter chain...", pathPattern, Encode.forHtml(requestURINoTrailingSlash));
128                     }
129                     return filterChainManager.proxy(originalChain, pathPattern);
130                 }
131             }
132         }
133 
134         return null;
135     }
136 
137     /**
138      * Returns {@code true} if an incoming request path (the {@code path} argument)
139      * matches a configured filter chain path (the {@code pattern} argument), {@code false} otherwise.
140      * <p/>
141      * Simply delegates to
142      * <b><code>{@link #getPathMatcher() getPathMatcher()}.{@link org.apache.shiro.util.PatternMatcher#matches(String, String) matches(pattern,path)}</code></b>.
143      * Subclass implementors should think carefully before overriding this method, as typically a custom
144      * {@code PathMatcher} should be configured for custom path matching behavior instead.  Favor OO composition
145      * rather than inheritance to limit your exposure to Shiro implementation details which may change over time.
146      *
147      * @param pattern the pattern to match against
148      * @param path    the value to match with the specified {@code pattern}
149      * @return {@code true} if the request {@code path} matches the specified filter chain url {@code pattern},
150      *         {@code false} otherwise.
151      */
152     protected boolean pathMatches(String pattern, String path) {
153         PatternMatcher pathMatcher = getPathMatcher();
154         return pathMatcher.matches(pattern, path);
155     }
156 
157     /**
158      * Merely returns
159      * <code>WebUtils.{@link org.apache.shiro.web.util.WebUtils#getPathWithinApplication(javax.servlet.http.HttpServletRequest) getPathWithinApplication(request)}</code>
160      * and can be overridden by subclasses for custom request-to-application-path resolution behavior.
161      *
162      * @param request the incoming {@code ServletRequest}
163      * @return the request's path within the appliation.
164      */
165     protected String getPathWithinApplication(ServletRequest request) {
166         return WebUtils.getPathWithinApplication(WebUtils.toHttp(request));
167     }
168 
169     private static String removeTrailingSlash(String path) {
170         if(path != null && !DEFAULT_PATH_SEPARATOR.equals(path)
171            && path.endsWith(DEFAULT_PATH_SEPARATOR)) {
172             return path.substring(0, path.length() - 1);
173         }
174         return path;
175     }
176 }
