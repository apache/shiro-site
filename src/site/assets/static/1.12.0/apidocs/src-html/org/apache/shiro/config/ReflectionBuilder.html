057/**
058 * Object builder that uses reflection and Apache Commons BeanUtils to build objects given a
059 * map of "property values".  Typically these come from the Shiro INI configuration and are used
060 * to construct or modify the SecurityManager, its dependencies, and web-based security filters.
061 * <p/>
062 * Recognizes {@link Factory} implementations and will call
063 * {@link org.apache.shiro.util.Factory#getInstance() getInstance} to satisfy any reference to this bean.
064 *
065 * @since 0.9
066 */
067public class ReflectionBuilder {

124    public ReflectionBuilder(Map<String, ?> defaults) {
125
126        // SHIRO-619
127        // SHIRO-739
128        beanUtilsBean = new BeanUtilsBean(new ConvertUtilsBean() {
129            @Override
130            public Object convert(String value, Class clazz) {
131                if (clazz.isEnum()){
132                    return Enum.valueOf(clazz, value);
133                }else{
134                    return super.convert(value, clazz);
135                }
136            }
137        });
138        beanUtilsBean.getPropertyUtils().addBeanIntrospector(SuppressPropertiesBeanIntrospector.SUPPRESS_CLASS);
139        beanUtilsBean.getPropertyUtils().addBeanIntrospector(new FluentPropertyBeanIntrospector());
140
141        this.interpolator = createInterpolator();
142
143        this.objects = createDefaultObjectMap();
144        this.registeredEventSubscribers = new LinkedHashMap<String,Object>();
145        apply(defaults);
146    }

148    private void apply(Map<String, ?> objects) {
149        if(!isEmpty(objects)) {
150            this.objects.putAll(objects);
151        }
152        EventBus found = findEventBus(this.objects);
153        Assert.notNull(found, "An " + EventBus.class.getName() + " instance must be present in the object defaults");
154        enableEvents(found);
155    }

170    //@since 1.3
171    private void enableEvents(EventBus eventBus) {
172        Assert.notNull(eventBus, "EventBus argument cannot be null.");
173        //clean up old auto-registered subscribers:
174        for (Object subscriber : this.registeredEventSubscribers.values()) {
175            this.eventBus.unregister(subscriber);
176        }
177        this.registeredEventSubscribers.clear();
178
179        this.eventBus = eventBus;
180
181        for(Map.Entry<String,Object> entry : this.objects.entrySet()) {
182            enableEventsIfNecessary(entry.getValue(), entry.getKey());
183        }
184    }

186    //@since 1.3
187    private void enableEventsIfNecessary(Object bean, String name) {
188        boolean applied = applyEventBusIfNecessary(bean);
189        if (!applied) {
190            //if the event bus is applied, and the bean wishes to be a subscriber as well (not just a publisher),
191            // we assume that the implementation registers itself with the event bus, i.e. eventBus.register(this);
192
193            //if the event bus isn't applied, only then do we need to check to see if the bean is an event subscriber,
194            // and if so, register it on the event bus automatically since it has no ability to do so itself:
195            if (isEventSubscriber(bean, name)) {
196                //found an event subscriber, so register them with the EventBus:
197                this.eventBus.register(bean);
198                this.registeredEventSubscribers.put(name, bean);
199            }
200        }
201    }

204    private boolean isEventSubscriber(Object bean, String name) {
205        List annotatedMethods = ClassUtils.getAnnotatedMethods(bean.getClass(), Subscribe.class);
206        return !isEmpty(annotatedMethods);
207    }

209    //@since 1.3
210    protected EventBus findEventBus(Map<String,?> objects) {
211
212        if (isEmpty(objects)) {
213            return null;
214        }
215
216        //prefer a named object first:
217        Object value = objects.get(EVENT_BUS_NAME);
218        if (value != null && value instanceof EventBus) {
219            return (EventBus)value;
220        }
221
222        //couldn't find a named 'eventBus' EventBus object.  Try to find the first typed value we can:
223        for( Object v : objects.values()) {
224            if (v instanceof EventBus) {
225                return (EventBus)v;
226            }
227        }
228
229        return null;
230    }
231
232    private boolean applyEventBusIfNecessary(Object value) {
233        if (value instanceof EventBusAware) {
234            ((EventBusAware)value).setEventBus(this.eventBus);
235            return true;
236        }
237        return false;
238    }

244    @SuppressWarnings({"unchecked"})
245    public <T> T getBean(String id, Class<T> requiredType) {
246        if (requiredType == null) {
247            throw new NullPointerException("requiredType argument cannot be null.");
248        }
249        Object bean = getBean(id);
250        if (bean == null) {
251            return null;
252        }
253        Assert.state(requiredType.isAssignableFrom(bean.getClass()),
254                "Bean with id [" + id + "] is not of the required type [" + requiredType.getName() + "].");
255        return (T) bean;
256    }

258    private String parseBeanId(String lhs) {
259        Assert.notNull(lhs);
260        if (lhs.indexOf('.') < 0) {
261            return lhs;
262        }
263        String classSuffix = ".class";
264        int index = lhs.indexOf(classSuffix);
265        if (index >= 0) {
266            return lhs.substring(0, index);
267        }
268        return null;
269    }

271    @SuppressWarnings({"unchecked"})
272    public Map<String, ?> buildObjects(Map<String, String> kvPairs) {
273
274        if (kvPairs != null && !kvPairs.isEmpty()) {
275
276            BeanConfigurationProcessor processor = new BeanConfigurationProcessor();
277
278            for (Map.Entry<String, String> entry : kvPairs.entrySet()) {
279                String lhs = entry.getKey();
280                String rhs = interpolator.interpolate(entry.getValue());
281
282                String beanId = parseBeanId(lhs);
283                if (beanId != null) { //a beanId could be parsed, so the line is a bean instance definition
284                    processor.add(new InstantiationStatement(beanId, rhs));
285                } else { //the line must be a property configuration
286                    processor.add(new AssignmentStatement(lhs, rhs));
287                }
288            }
289
290            processor.execute();
291        }
292
293        //SHIRO-413: init method must be called for constructed objects that are Initializable
294        LifecycleUtils.init(objects.values());
295
296        return objects;
297    }

299    public void destroy() {
300        final Map<String, Object> immutableObjects = Collections.unmodifiableMap(objects);
301
302        //destroy objects in the opposite order they were initialized:
303        List<Map.Entry<String,?>> entries = new ArrayList<Map.Entry<String,?>>(objects.entrySet());
304        Collections.reverse(entries);
305
306        for(Map.Entry<String, ?> entry: entries) {
307            String id = entry.getKey();
308            Object bean = entry.getValue();
309
310            //don't destroy the eventbus until the end - we need it to still be 'alive' while publishing destroy events:
311            if (bean != this.eventBus) { //memory equality check (not .equals) on purpose
312                LifecycleUtils.destroy(bean);
313                BeanEvent event = new DestroyedBeanEvent(id, bean, immutableObjects);
314                eventBus.publish(event);
315                this.eventBus.unregister(bean); //bean is now destroyed - it should not receive any other events
316            }
317        }
318        //only now destroy the event bus:
319        LifecycleUtils.destroy(this.eventBus);
320    }

322    protected void createNewInstance(Map<String, Object> objects, String name, String value) {
323
324        Object currentInstance = objects.get(name);
325        if (currentInstance != null) {
326            log.info("An instance with name '{}' already exists.  " +
327                    "Redefining this object as a new instance of type {}", name, value);
328        }
329
330        Object instance;//name with no property, assume right hand side of equals sign is the class name:
331        try {
332            instance = ClassUtils.newInstance(value);
333            if (instance instanceof Nameable) {
334                ((Nameable) instance).setName(name);
335            }
336        } catch (Exception e) {
337            String msg = "Unable to instantiate class [" + value + "] for object named '" + name + "'.  " +
338                    "Please ensure you've specified the fully qualified class name correctly.";
339            throw new ConfigurationException(msg, e);
340        }
341        objects.put(name, instance);
342    }

344    protected void applyProperty(String key, String value, Map objects) {
345
346        int index = key.indexOf('.');
347
348        if (index >= 0) {
349            String name = key.substring(0, index);
350            String property = key.substring(index + 1, key.length());
351
352            if (GLOBAL_PROPERTY_PREFIX.equalsIgnoreCase(name)) {
353                applyGlobalProperty(objects, property, value);
354            } else {
355                applySingleProperty(objects, name, property, value);
356            }
357
358        } else {
359            throw new IllegalArgumentException("All property keys must contain a '.' character. " +
360                    "(e.g. myBean.property = value)  These should already be separated out by buildObjects().");
361        }
362    }
363
364    protected void applyGlobalProperty(Map objects, String property, String value) {
365        for (Object instance : objects.values()) {
366            try {
367                PropertyDescriptor pd = beanUtilsBean.getPropertyUtils().getPropertyDescriptor(instance, property);
368                if (pd != null) {
369                    applyProperty(instance, property, value);
370                }
371            } catch (Exception e) {
372                String msg = "Error retrieving property descriptor for instance " +
373                        "of type [" + instance.getClass().getName() + "] " +
374                        "while setting property [" + property + "]";
375                throw new ConfigurationException(msg, e);
376            }
377        }
378    }
379
380    protected void applySingleProperty(Map objects, String name, String property, String value) {
381        Object instance = objects.get(name);
382        if (property.equals("class")) {
383            throw new IllegalArgumentException("Property keys should not contain 'class' properties since these " +
384                    "should already be separated out by buildObjects().");
385
386        } else if (instance == null) {
387            String msg = "Configuration error.  Specified object [" + name + "] with property [" +
388                    property + "] without first defining that object's class.  Please first " +
389                    "specify the class property first, e.g. myObject = fully_qualified_class_name " +
390                    "and then define additional properties.";
391            throw new IllegalArgumentException(msg);
392
393        } else {
394            applyProperty(instance, property, value);
395        }
396    }

399        return value != null && value.startsWith(OBJECT_REFERENCE_BEGIN_TOKEN);

406    protected Object getReferencedObject(String id) {
407        Object o = objects != null && !objects.isEmpty() ? objects.get(id) : null;
408        if (o == null) {
409            String msg = "The object with id [" + id + "] has not yet been defined and therefore cannot be " +
410                    "referenced.  Please ensure objects are defined in the order in which they should be " +
411                    "created and made available for future reference.";
412            throw new UnresolveableReferenceException(msg);
413        }
414        return o;
415    }

417    protected String unescapeIfNecessary(String value) {
418        if (value != null && value.startsWith(ESCAPED_OBJECT_REFERENCE_BEGIN_TOKEN)) {
419            return value.substring(ESCAPED_OBJECT_REFERENCE_BEGIN_TOKEN.length() - 1);
420        }
421        return value;
422    }

424    protected Object resolveReference(String reference) {
425        String id = getId(reference);
426        log.debug("Encountered object reference '{}'.  Looking up object with id '{}'", reference, id);
427        final Object referencedObject = getReferencedObject(id);
428        if (referencedObject instanceof Factory) {
429            return ((Factory) referencedObject).getInstance();
430        }
431        return referencedObject;
432    }

434    protected boolean isTypedProperty(Object object, String propertyName, Class clazz) {
435        if (clazz == null) {
436            throw new NullPointerException("type (class) argument cannot be null.");
437        }
438        try {
439            PropertyDescriptor descriptor = beanUtilsBean.getPropertyUtils().getPropertyDescriptor(object, propertyName);
440            if (descriptor == null) {
441                String msg = "Property '" + propertyName + "' does not exist for object of " +
442                        "type " + object.getClass().getName() + ".";
443                throw new ConfigurationException(msg);
444            }
445            Class propertyClazz = descriptor.getPropertyType();
446            return clazz.isAssignableFrom(propertyClazz);
447        } catch (ConfigurationException ce) {
448            //let it propagate:
449            throw ce;
450        } catch (Exception e) {
451            String msg = "Unable to determine if property [" + propertyName + "] represents a " + clazz.getName();
452            throw new ConfigurationException(msg, e);
453        }
454    }

456    protected Set<?> toSet(String sValue) {
457        String[] tokens = StringUtils.split(sValue);
458        if (tokens == null || tokens.length <= 0) {
459            return null;
460        }
461
462        //SHIRO-423: check to see if the value is a referenced Set already, and if so, return it immediately:
463        if (tokens.length == 1 && isReference(tokens[0])) {
464            Object reference = resolveReference(tokens[0]);
465            if (reference instanceof Set) {
466                return (Set)reference;
467            }
468        }
469
470        Set<String> setTokens = new LinkedHashSet<String>(Arrays.asList(tokens));
471
472        //now convert into correct values and/or references:
473        Set<Object> values = new LinkedHashSet<Object>(setTokens.size());
474        for (String token : setTokens) {
475            Object value = resolveValue(token);
476            values.add(value);
477        }
478        return values;
479    }

481    protected Map<?, ?> toMap(String sValue) {
482        String[] tokens = StringUtils.split(sValue, StringUtils.DEFAULT_DELIMITER_CHAR,
483                StringUtils.DEFAULT_QUOTE_CHAR, StringUtils.DEFAULT_QUOTE_CHAR, true, true);
484        if (tokens == null || tokens.length <= 0) {
485            return null;
486        }
487
488        //SHIRO-423: check to see if the value is a referenced Map already, and if so, return it immediately:
489        if (tokens.length == 1 && isReference(tokens[0])) {
490            Object reference = resolveReference(tokens[0]);
491            if (reference instanceof Map) {
492                return (Map)reference;
493            }
494        }
495
496        Map<String, String> mapTokens = new LinkedHashMap<String, String>(tokens.length);
497        for (String token : tokens) {
498            String[] kvPair = StringUtils.split(token, MAP_KEY_VALUE_DELIMITER);
499            if (kvPair == null || kvPair.length != 2) {
500                String msg = "Map property value [" + sValue + "] contained key-value pair token [" +
501                        token + "] that does not properly split to a single key and pair.  This must be the " +
502                        "case for all map entries.";
503                throw new ConfigurationException(msg);
504            }
505            mapTokens.put(kvPair[0], kvPair[1]);
506        }
507
508        //now convert into correct values and/or references:
509        Map<Object, Object> map = new LinkedHashMap<Object, Object>(mapTokens.size());
510        for (Map.Entry<String, String> entry : mapTokens.entrySet()) {
511            Object key = resolveValue(entry.getKey());
512            Object value = resolveValue(entry.getValue());
513            map.put(key, value);
514        }
515        return map;
516    }

566    protected byte[] toBytes(String sValue) {
567        if (sValue == null) {
568            return null;
569        }
570        byte[] bytes;
571        if (sValue.startsWith(HEX_BEGIN_TOKEN)) {
572            String hex = sValue.substring(HEX_BEGIN_TOKEN.length());
573            bytes = Hex.decode(hex);
574        } else {
575            //assume base64 encoded:
576            bytes = Base64.decode(sValue);
577        }
578        return bytes;
579    }

581    protected Object resolveValue(String stringValue) {
582        Object value;
583        if (isReference(stringValue)) {
584            value = resolveReference(stringValue);
585        } else {
586            value = unescapeIfNecessary(stringValue);
587        }
588        return value;
589    }

591    protected String checkForNullOrEmptyLiteral(String stringValue) {
592        if (stringValue == null) {
593            return null;
594        }
595        //check if the value is the actual literal string 'null' (expected to be wrapped in quotes):
596        if (stringValue.equals("\"null\"")) {
597            return NULL_VALUE_TOKEN;
598        }
599        //or the actual literal string of two quotes '""' (expected to be wrapped in quotes):
600        else if (stringValue.equals("\"\"\"\"")) {
601            return EMPTY_STRING_VALUE_TOKEN;
602        } else {
603            return stringValue;
604        }
605    }

607    protected void applyProperty(Object object, String propertyPath, Object value) {
608
609        int mapBegin = propertyPath.indexOf(MAP_PROPERTY_BEGIN_TOKEN);
610        int mapEnd = -1;
611        String mapPropertyPath = null;
612        String keyString = null;
613
614        String remaining = null;
615        
616        if (mapBegin >= 0) {
617            //a map is being referenced in the overall property path.  Find just the map's path:
618            mapPropertyPath = propertyPath.substring(0, mapBegin);
619            //find the end of the map reference:
620            mapEnd = propertyPath.indexOf(MAP_PROPERTY_END_TOKEN, mapBegin);
621            //find the token in between the [ and the ] (the map/array key or index):
622            keyString = propertyPath.substring(mapBegin+1, mapEnd);
623
624            //find out if there is more path reference to follow.  If not, we're at a terminal of the OGNL expression
625            if (propertyPath.length() > (mapEnd+1)) {
626                remaining = propertyPath.substring(mapEnd+1);
627                if (remaining.startsWith(".")) {
628                    remaining = StringUtils.clean(remaining.substring(1));
629                }
630            }
631        }
632        
633        if (remaining == null) {
634            //we've terminated the OGNL expression.  Check to see if we're assigning a property or a map entry:
635            if (keyString == null) {
636                //not a map or array value assignment - assign the property directly:
637                setProperty(object, propertyPath, value);
638            } else {
639                //we're assigning a map or array entry.  Check to see which we should call:
640                if (isTypedProperty(object, mapPropertyPath, Map.class)) {
641                    Map map = (Map)getProperty(object, mapPropertyPath);
642                    Object mapKey = resolveValue(keyString);
643                    //noinspection unchecked
644                    map.put(mapKey, value);
645                } else {
646                    //must be an array property.  Convert the key string to an index:
647                    int index = Integer.valueOf(keyString);
648                    setIndexedProperty(object, mapPropertyPath, index, value);
649                }
650            }
651        } else {
652            //property is being referenced as part of a nested path.  Find the referenced map/array entry and
653            //recursively call this method with the remaining property path
654            Object referencedValue = null;
655            if (isTypedProperty(object, mapPropertyPath, Map.class)) {
656                Map map = (Map)getProperty(object, mapPropertyPath);
657                Object mapKey = resolveValue(keyString);
658                referencedValue = map.get(mapKey);
659            } else {
660                //must be an array property:
661                int index = Integer.valueOf(keyString);
662                referencedValue = getIndexedProperty(object, mapPropertyPath, index);
663            }
664
665            if (referencedValue == null) {
666                throw new ConfigurationException("Referenced map/array value '" + mapPropertyPath + "[" +
667                keyString + "]' does not exist.");
668            }
669
670            applyProperty(referencedValue, remaining, value);
671        }
672    }

674    private void setProperty(Object object, String propertyPath, Object value) {
675        try {
676            if (log.isTraceEnabled()) {
677                log.trace("Applying property [{}] value [{}] on object of type [{}]",
678                        new Object[]{propertyPath, value, object.getClass().getName()});
679            }
680            beanUtilsBean.setProperty(object, propertyPath, value);
681        } catch (Exception e) {
682            String msg = "Unable to set property '" + propertyPath + "' with value [" + value + "] on object " +
683                    "of type " + (object != null ? object.getClass().getName() : null) + ".  If " +
684                    "'" + value + "' is a reference to another (previously defined) object, prefix it with " +
685                    "'" + OBJECT_REFERENCE_BEGIN_TOKEN + "' to indicate that the referenced " +
686                    "object should be used as the actual value.  " +
687                    "For example, " + OBJECT_REFERENCE_BEGIN_TOKEN + value;
688            throw new ConfigurationException(msg, e);
689        }
690    }

720    protected void applyProperty(Object object, String propertyName, String stringValue) {
721
722        Object value;
723
724        if (NULL_VALUE_TOKEN.equals(stringValue)) {
725            value = null;
726        } else if (EMPTY_STRING_VALUE_TOKEN.equals(stringValue)) {
727            value = StringUtils.EMPTY_STRING;
728        } else if (isIndexedPropertyAssignment(propertyName)) {
729            String checked = checkForNullOrEmptyLiteral(stringValue);
730            value = resolveValue(checked);
731        } else if (isTypedProperty(object, propertyName, Set.class)) {
732            value = toSet(stringValue);
733        } else if (isTypedProperty(object, propertyName, Map.class)) {
734            value = toMap(stringValue);
735        } else if (isTypedProperty(object, propertyName, List.class)) {
736            value = toList(stringValue);
737        } else if (isTypedProperty(object, propertyName, Collection.class)) {
738            value = toCollection(stringValue);
739        } else if (isTypedProperty(object, propertyName, byte[].class)) {
740            value = toBytes(stringValue);
741        } else if (isTypedProperty(object, propertyName, ByteSource.class)) {
742            byte[] bytes = toBytes(stringValue);
743            value = ByteSource.Util.bytes(bytes);
744        } else {
745            String checked = checkForNullOrEmptyLiteral(stringValue);
746            value = resolveValue(checked);
747        }
748
749        applyProperty(object, propertyName, value);
750    }

769    private class BeanConfigurationProcessor {
770
771        private final List<Statement> statements = new ArrayList<Statement>();
772        private final List<BeanConfiguration> beanConfigurations = new ArrayList<BeanConfiguration>();
773
774        public void add(Statement statement) {
775
776            statements.add(statement); //we execute bean configuration statements in the order they are declared.
777
778            if (statement instanceof InstantiationStatement) {
779                InstantiationStatement is = (InstantiationStatement)statement;
780                beanConfigurations.add(new BeanConfiguration(is));
781            } else {
782                AssignmentStatement as = (AssignmentStatement)statement;
783                //statements always apply to the most recently defined bean configuration with the same name, so we
784                //have to traverse the configuration list starting at the end (most recent elements are appended):
785                boolean addedToConfig = false;
786                String beanName = as.getRootBeanName();
787                for( int i = beanConfigurations.size()-1; i >= 0; i--) {
788                    BeanConfiguration mostRecent = beanConfigurations.get(i);
789                    String mostRecentBeanName = mostRecent.getBeanName();
790                    if (beanName.equals(mostRecentBeanName)) {
791                        mostRecent.add(as);
792                        addedToConfig = true;
793                        break;
794                    }
795                }
796
797                if (!addedToConfig) {
798                    // the AssignmentStatement must be for an existing bean that does not yet have a corresponding
799                    // configuration object (this would happen if the bean is in the default objects map). Because
800                    // BeanConfiguration instances don't exist for default (already instantiated) beans,
801                    // we simulate a creation of one to satisfy this processors implementation:
802                    beanConfigurations.add(new BeanConfiguration(as));
803                }
804            }
805        }
806
807        public void execute() {
808
809            for( Statement statement : statements) {
810
811                statement.execute();
812
813                BeanConfiguration bd = statement.getBeanConfiguration();
814
815                if (bd.isExecuted()) { //bean is fully configured, no more statements to execute for it:
816
817                    //bean configured overrides the 'eventBus' bean - replace the existing eventBus with the one configured:
818                    if (bd.getBeanName().equals(EVENT_BUS_NAME)) {
819                        EventBus eventBus = (EventBus)bd.getBean();
820                        enableEvents(eventBus);
821                    }
822
823                    //ignore global 'shiro.' shortcut mechanism:
824                    if (!bd.isGlobalConfig()) {
825                        BeanEvent event = new ConfiguredBeanEvent(bd.getBeanName(), bd.getBean(),
826                                Collections.unmodifiableMap(objects));
827                        eventBus.publish(event);
828                    }
829
830                    //initialize the bean if necessary:
831                    LifecycleUtils.init(bd.getBean());
832
833                    //ignore global 'shiro.' shortcut mechanism:
834                    if (!bd.isGlobalConfig()) {
835                        BeanEvent event = new InitializedBeanEvent(bd.getBeanName(), bd.getBean(),
836                                Collections.unmodifiableMap(objects));
837                        eventBus.publish(event);
838                    }
839                }
840            }
841        }
842    }

844    private class BeanConfiguration {
845
846        private final InstantiationStatement instantiationStatement;
847        private final List<AssignmentStatement> assignments = new ArrayList<AssignmentStatement>();
848        private final String beanName;
849        private Object bean;
850
851        private BeanConfiguration(InstantiationStatement statement) {
852            statement.setBeanConfiguration(this);
853            this.instantiationStatement = statement;
854            this.beanName = statement.lhs;
855        }
856
857        private BeanConfiguration(AssignmentStatement as) {
858            this.instantiationStatement = null;
859            this.beanName = as.getRootBeanName();
860            add(as);
861        }
862
863        public String getBeanName() {
864            return this.beanName;
865        }
866
867        public boolean isGlobalConfig() { //BeanConfiguration instance representing the global 'shiro.' properties
868            // (we should remove this concept).
869            return GLOBAL_PROPERTY_PREFIX.equals(getBeanName());
870        }
871
872        public void add(AssignmentStatement as) {
873            as.setBeanConfiguration(this);
874            assignments.add(as);
875        }
876
877        /**
878         * When this configuration is parsed sufficiently to create (or find) an actual bean instance, that instance
879         * will be associated with its configuration by setting it via this method.
880         *
881         * @param bean the bean instantiated (or found) that corresponds to this BeanConfiguration instance.
882         */
883        public void setBean(Object bean) {
884            this.bean = bean;
885        }
886
887        public Object getBean() {
888            return this.bean;
889        }
890
891        /**
892         * Returns true if all configuration statements have been executed.
893         * @return true if all configuration statements have been executed.
894         */
895        public boolean isExecuted() {
896            if (instantiationStatement != null && !instantiationStatement.isExecuted()) {
897                return false;
898            }
899            for (AssignmentStatement as : assignments) {
900                if (!as.isExecuted()) {
901                    return false;
902                }
903            }
904            return true;
905        }
906    }

908    private abstract class Statement {
909
910        protected final String lhs;
911        protected final String rhs;
912        protected Object bean;
913        private Object result;
914        private boolean executed;
915        private BeanConfiguration beanConfiguration;
916
917        private Statement(String lhs, String rhs) {
918            this.lhs = lhs;
919            this.rhs = rhs;
920            this.executed = false;
921        }
922
923        public void setBeanConfiguration(BeanConfiguration bd) {
924            this.beanConfiguration = bd;
925        }
926
927        public BeanConfiguration getBeanConfiguration() {
928            return this.beanConfiguration;
929        }
930
931        public Object execute() {
932            if (!isExecuted()) {
933                this.result = doExecute();
934                this.executed = true;
935            }
936            if (!getBeanConfiguration().isGlobalConfig()) {
937                Assert.notNull(this.bean, "Implementation must set the root bean for which it executed.");
938            }
939            return this.result;
940        }
941
942        public Object getBean() {
943            return this.bean;
944        }
945
946        protected void setBean(Object bean) {
947            this.bean = bean;
948            if (this.beanConfiguration.getBean() == null) {
949                this.beanConfiguration.setBean(bean);
950            }
951        }
952
953        public Object getResult() {
954            return result;
955        }
956
957        protected abstract Object doExecute();
958
959        public boolean isExecuted() {
960            return executed;
961        }
962    }

964    private class InstantiationStatement extends Statement {
965
966        private InstantiationStatement(String lhs, String rhs) {
967            super(lhs, rhs);
968        }
969
970        @Override
971        protected Object doExecute() {
972            String beanName = this.lhs;
973            createNewInstance(objects, beanName, this.rhs);
974            Object instantiated = objects.get(beanName);
975            setBean(instantiated);
976
977            //also ensure the instantiated bean has access to the event bus or is subscribed to events if necessary:
978            //Note: because events are being enabled on this bean here (before the instantiated event below is
979            //triggered), beans can react to their own instantiation events.
980            enableEventsIfNecessary(instantiated, beanName);
981
982            BeanEvent event = new InstantiatedBeanEvent(beanName, instantiated, Collections.unmodifiableMap(objects));
983            eventBus.publish(event);
984
985            return instantiated;
986        }
987    }

989    private class AssignmentStatement extends Statement {
990
991        private final String rootBeanName;
992
993        private AssignmentStatement(String lhs, String rhs) {
994            super(lhs, rhs);
995            int index = lhs.indexOf('.');
996            this.rootBeanName = lhs.substring(0, index);
997        }
998
999        @Override
1000        protected Object doExecute() {
1001            applyProperty(lhs, rhs, objects);
1002            Object bean = objects.get(this.rootBeanName);
1003            setBean(bean);
1004            return null;
1005        }
1006
1007        public String getRootBeanName() {
1008            return this.rootBeanName;
1009        }
