45          assertEquals(1, ini.getSections().size());

48          assertEquals(Ini.DEFAULT_SECTION_NAME, section.getName());

55      @Test
56      public void testIsContinued() {
57          //no slashes
58          String line = "prop = value ";
59          assertFalse(Ini.Section.isContinued(line));
60  
61          //1 slash (odd number, but edge case):
62          line = "prop = value" + Ini.ESCAPE_TOKEN;
63          assertTrue(Ini.Section.isContinued(line));
64  
65          //2 slashes = even number
66          line = "prop = value" + Ini.ESCAPE_TOKEN + Ini.ESCAPE_TOKEN;
67          assertFalse(Ini.Section.isContinued(line));
68  
69          //3 slashes = odd number
70          line = "prop = value" + Ini.ESCAPE_TOKEN + Ini.ESCAPE_TOKEN + Ini.ESCAPE_TOKEN;
71          assertTrue(Ini.Section.isContinued(line));
72      }

145                 "# comment.";
