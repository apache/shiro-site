019package org.apache.shiro.spring.config;
020
021
022import org.apache.shiro.mgt.SecurityManager;
023import org.apache.shiro.spring.security.interceptor.AuthorizationAttributeSourceAdvisor;
024import org.springframework.aop.framework.autoproxy.DefaultAdvisorAutoProxyCreator;
025
026/**
027 * @since 1.4.0
028 */
029public class AbstractShiroAnnotationProcessorConfiguration {
030
031    protected DefaultAdvisorAutoProxyCreator defaultAdvisorAutoProxyCreator() {
032        return new DefaultAdvisorAutoProxyCreator();
033    }
034
035    protected AuthorizationAttributeSourceAdvisor authorizationAttributeSourceAdvisor(SecurityManager securityManager) {
036        AuthorizationAttributeSourceAdvisor advisor = new AuthorizationAttributeSourceAdvisor();
037        advisor.setSecurityManager(securityManager);
038        return advisor;
039    }
040
041}
