019package org.apache.shiro.samples;
020
021import org.springframework.beans.factory.annotation.Autowired;
022import org.springframework.boot.web.servlet.error.ErrorAttributes;
023import org.springframework.boot.web.servlet.error.ErrorController;
024import org.springframework.stereotype.Controller;
025import org.springframework.ui.Model;
026import org.springframework.web.bind.annotation.RequestMapping;
027import org.springframework.web.context.request.ServletWebRequest;
028
029import javax.servlet.http.HttpServletRequest;
030import java.util.Map;
031
032/**
033 */
034@Controller
035public class RestrictedErrorController implements ErrorController {
036    private static final String ERROR_PATH = "/error";
037
038    @Autowired
039    private ErrorAttributes errorAttributes;
040
041    @Override
042    public String getErrorPath() {
043        return ERROR_PATH;
044    }
045
046    @RequestMapping(ERROR_PATH)
047    String error(HttpServletRequest request, Model model) {
048        Map<String, Object> errorMap = errorAttributes.getErrorAttributes(new ServletWebRequest(request), false);
049        model.addAttribute("errors", errorMap);
050        return "error";
051    }
052}
