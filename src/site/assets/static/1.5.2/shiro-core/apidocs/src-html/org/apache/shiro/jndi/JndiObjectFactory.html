019package org.apache.shiro.jndi;
020
021import org.apache.shiro.util.Factory;
022
023import javax.naming.NamingException;
024
025/**
026 * A factory implementation intended to be used to look up objects in jndi.
027 * @param <T>
028 * @since 1.2
029 */
030public class JndiObjectFactory<T> extends JndiLocator implements Factory<T> {
031
032    private String resourceName;
033    private Class<? extends T> requiredType;
034
035    public T getInstance() {
036        try {
037            if(requiredType != null) {
038                return requiredType.cast(this.lookup(resourceName, requiredType));
039            } else {
040                return (T) this.lookup(resourceName);
041            }
042        } catch (NamingException e) {
043            final String typeName = requiredType != null ? requiredType.getName() : "object";
044            throw new IllegalStateException("Unable to look up " + typeName + " with jndi name '" + resourceName + "'.", e);
045        }
046    }
047
048    public String getResourceName() {
049        return resourceName;
050    }
051
052    public void setResourceName(String resourceName) {
053        this.resourceName = resourceName;
054    }
055
056    public Class<? extends T> getRequiredType() {
057        return requiredType;
058    }
059
060    public void setRequiredType(Class<? extends T> requiredType) {
061        this.requiredType = requiredType;
062    }
063}
