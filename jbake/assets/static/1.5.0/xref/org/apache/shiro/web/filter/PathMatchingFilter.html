36  import static org.apache.shiro.util.StringUtils.split;

38  /**
39   * <p>Base class for Filters that will process only specified paths and allow all others to pass through.</p>
40   *
41   * @since 0.9
42   */
43  public abstract class PathMatchingFilter extends AdviceFilter implements PathConfigProcessor {
44  
45      /**
46       * Log available to this class only
47       */
48      private static final Logger log = LoggerFactory.getLogger(PathMatchingFilter.class);
49  
50      private static final String DEFAULT_PATH_SEPARATOR = "/";
51  
52      /**
53       * PatternMatcher used in determining which paths to react to for a given request.
54       */
55      protected PatternMatcher pathMatcher = new AntPathMatcher();
56  
57      /**
58       * A collection of path-to-config entries where the key is a path which this filter should process and
59       * the value is the (possibly null) configuration element specific to this Filter for that specific path.
60       * <p/>
61       * <p>To put it another way, the keys are the paths (urls) that this Filter will process.
62       * <p>The values are filter-specific data that this Filter should use when processing the corresponding
63       * key (path).  The values can be null if no Filter-specific config was specified for that url.
64       */
65      protected Map<String, Object> appliedPaths = new LinkedHashMap<String, Object>();
66  
67      /**
68       * Splits any comma-delmited values that might be found in the <code>config</code> argument and sets the resulting
69       * <code>String[]</code> array on the <code>appliedPaths</code> internal Map.
70       * <p/>
71       * That is:
72       * <pre><code>
73       * String[] values = null;
74       * if (config != null) {
75       *     values = split(config);
76       * }
77       * <p/>
78       * this.{@link #appliedPaths appliedPaths}.put(path, values);
79       * </code></pre>
80       *
81       * @param path   the application context path to match for executing this filter.
82       * @param config the specified for <em>this particular filter only</em> for the given <code>path</code>
83       * @return this configured filter.
84       */
85      public Filter processPathConfig(String path, String config) {
86          String[] values = null;
87          if (config != null) {
88              values = split(config);
89          }
90  
91          this.appliedPaths.put(path, values);
92          return this;
93      }
94  
95      /**
96       * Returns the context path within the application based on the specified <code>request</code>.
97       * <p/>
98       * This implementation merely delegates to
99       * {@link WebUtils#getPathWithinApplication(javax.servlet.http.HttpServletRequest) WebUtils.getPathWithinApplication(request)},
100      * but can be overridden by subclasses for custom logic.
101      *
102      * @param request the incoming <code>ServletRequest</code>
103      * @return the context path within the application.
104      */
105     protected String getPathWithinApplication(ServletRequest request) {
106         return WebUtils.getPathWithinApplication(WebUtils.toHttp(request));
107     }
108 
109     /**
110      * Returns <code>true</code> if the incoming <code>request</code> matches the specified <code>path</code> pattern,
111      * <code>false</code> otherwise.
112      * <p/>
113      * The default implementation acquires the <code>request</code>'s path within the application and determines
114      * if that matches:
115      * <p/>
116      * <code>String requestURI = {@link #getPathWithinApplication(javax.servlet.ServletRequest) getPathWithinApplication(request)};<br/>
117      * return {@link #pathsMatch(String, String) pathsMatch(path,requestURI)}</code>
118      *
119      * @param path    the configured url pattern to check the incoming request against.
120      * @param request the incoming ServletRequest
121      * @return <code>true</code> if the incoming <code>request</code> matches the specified <code>path</code> pattern,
122      *         <code>false</code> otherwise.
123      */
124     protected boolean pathsMatch(String path, ServletRequest request) {
125         String requestURI = getPathWithinApplication(request);
126         if (requestURI != null && requestURI.endsWith(DEFAULT_PATH_SEPARATOR)) {
127             requestURI = requestURI.substring(0, requestURI.length() - 1);
128         }
129         if (path != null && path.endsWith(DEFAULT_PATH_SEPARATOR)) {
130             path = path.substring(0, path.length() - 1);
131         }
132         log.trace("Attempting to match pattern '{}' with current requestURI '{}'...", path, Encode.forHtml(requestURI));
133         return pathsMatch(path, requestURI);
134     }
135 
136     /**
137      * Returns <code>true</code> if the <code>path</code> matches the specified <code>pattern</code> string,
138      * <code>false</code> otherwise.
139      * <p/>
140      * Simply delegates to
141      * <b><code>this.pathMatcher.{@link PatternMatcher#matches(String, String) matches(pattern,path)}</code></b>,
142      * but can be overridden by subclasses for custom matching behavior.
143      *
144      * @param pattern the pattern to match against
145      * @param path    the value to match with the specified <code>pattern</code>
146      * @return <code>true</code> if the <code>path</code> matches the specified <code>pattern</code> string,
147      *         <code>false</code> otherwise.
148      */
149     protected boolean pathsMatch(String pattern, String path) {
150         return pathMatcher.matches(pattern, path);
151     }
152 
153     /**
154      * Implementation that handles path-matching behavior before a request is evaluated.  If the path matches and
155      * the filter
156      * {@link #isEnabled(javax.servlet.ServletRequest, javax.servlet.ServletResponse, String, Object) isEnabled} for
157      * that path/config, the request will be allowed through via the result from
158      * {@link #onPreHandle(javax.servlet.ServletRequest, javax.servlet.ServletResponse, Object) onPreHandle}.  If the
159      * path does not match or the filter is not enabled for that path, this filter will allow passthrough immediately
160      * to allow the {@code FilterChain} to continue executing.
161      * <p/>
162      * In order to retain path-matching functionality, subclasses should not override this method if at all
163      * possible, and instead override
164      * {@link #onPreHandle(javax.servlet.ServletRequest, javax.servlet.ServletResponse, Object) onPreHandle} instead.
165      *
166      * @param request  the incoming ServletRequest
167      * @param response the outgoing ServletResponse
168      * @return {@code true} if the filter chain is allowed to continue to execute, {@code false} if a subclass has
169      *         handled the request explicitly.
170      * @throws Exception if an error occurs
171      */
172     protected boolean preHandle(ServletRequest request, ServletResponse response) throws Exception {
173 
174         if (this.appliedPaths == null || this.appliedPaths.isEmpty()) {
175             if (log.isTraceEnabled()) {
176                 log.trace("appliedPaths property is null or empty.  This Filter will passthrough immediately.");
177             }
178             return true;
179         }
180 
181         for (String path : this.appliedPaths.keySet()) {
182             // If the path does match, then pass on to the subclass implementation for specific checks
183             //(first match 'wins'):
184             if (pathsMatch(path, request)) {
185                 log.trace("Current requestURI matches pattern '{}'.  Determining filter chain execution...", path);
186                 Object config = this.appliedPaths.get(path);
187                 return isFilterChainContinued(request, response, path, config);
188             }
189         }
190 
191         //no path matched, allow the request to go through:
192         return true;
193     }
194 
195     /**
196      * Simple method to abstract out logic from the preHandle implementation - it was getting a bit unruly.
197      *
198      * @since 1.2
199      */
200     @SuppressWarnings({"JavaDoc"})
201     private boolean isFilterChainContinued(ServletRequest request, ServletResponse response,
202                                            String path, Object pathConfig) throws Exception {
203 
204         if (isEnabled(request, response, path, pathConfig)) { //isEnabled check added in 1.2
205             if (log.isTraceEnabled()) {
206                 log.trace("Filter '{}' is enabled for the current request under path '{}' with config [{}].  " +
207                         "Delegating to subclass implementation for 'onPreHandle' check.",
208                         new Object[]{getName(), path, pathConfig});
209             }
210             //The filter is enabled for this specific request, so delegate to subclass implementations
211             //so they can decide if the request should continue through the chain or not:
212             return onPreHandle(request, response, pathConfig);
213         }
214 
215         if (log.isTraceEnabled()) {
216             log.trace("Filter '{}' is disabled for the current request under path '{}' with config [{}].  " +
217                     "The next element in the FilterChain will be called immediately.",
218                     new Object[]{getName(), path, pathConfig});
219         }
220         //This filter is disabled for this specific request,
221         //return 'true' immediately to indicate that the filter will not process the request
222         //and let the request/response to continue through the filter chain:
223         return true;
224     }
225 
226     /**
227      * This default implementation always returns {@code true} and should be overridden by subclasses for custom
228      * logic if necessary.
229      *
230      * @param request     the incoming ServletRequest
231      * @param response    the outgoing ServletResponse
232      * @param mappedValue the filter-specific config value mapped to this filter in the URL rules mappings.
233      * @return {@code true} if the request should be able to continue, {@code false} if the filter will
234      *         handle the response directly.
235      * @throws Exception if an error occurs
236      * @see #isEnabled(javax.servlet.ServletRequest, javax.servlet.ServletResponse, String, Object)
237      */
238     protected boolean onPreHandle(ServletRequest request, ServletResponse response, Object mappedValue) throws Exception {
239         return true;
240     }
241 
242     /**
243      * Path-matching version of the parent class's
244      * {@link #isEnabled(javax.servlet.ServletRequest, javax.servlet.ServletResponse)} method, but additionally allows
245      * for inspection of any path-specific configuration values corresponding to the specified request.  Subclasses
246      * may wish to inspect this additional mapped configuration to determine if the filter is enabled or not.
247      * <p/>
248      * This method's default implementation ignores the {@code path} and {@code mappedValue} arguments and merely
249      * returns the value from a call to {@link #isEnabled(javax.servlet.ServletRequest, javax.servlet.ServletResponse)}.
250      * It is expected that subclasses override this method if they need to perform enable/disable logic for a specific
251      * request based on any path-specific config for the filter instance.
252      *
253      * @param request     the incoming servlet request
254      * @param response    the outbound servlet response
255      * @param path        the path matched for the incoming servlet request that has been configured with the given {@code mappedValue}.
256      * @param mappedValue the filter-specific config value mapped to this filter in the URL rules mappings for the given {@code path}.
257      * @return {@code true} if this filter should filter the specified request, {@code false} if it should let the
258      *         request/response pass through immediately to the next element in the {@code FilterChain}.
259      * @throws Exception in the case of any error
260      * @since 1.2
261      */
262     @SuppressWarnings({"UnusedParameters"})
263     protected boolean isEnabled(ServletRequest request, ServletResponse response, String path, Object mappedValue)
264             throws Exception {
265         return isEnabled(request, response);
266     }
