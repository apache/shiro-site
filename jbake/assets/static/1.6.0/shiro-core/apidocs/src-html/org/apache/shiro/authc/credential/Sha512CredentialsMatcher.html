019package org.apache.shiro.authc.credential;
020
021import org.apache.shiro.crypto.hash.AbstractHash;
022import org.apache.shiro.crypto.hash.Hash;
023import org.apache.shiro.crypto.hash.Sha512Hash;
024
025
026/**
027 * {@code HashedCredentialsMatcher} implementation that expects the stored {@code AuthenticationInfo} credentials to be
028 * SHA-512 hashed.
029 *
030 * @since 0.9
031 * @deprecated since 1.1 - use the HashedCredentialsMatcher directly and set its
032 *             {@link HashedCredentialsMatcher#setHashAlgorithmName(String) hashAlgorithmName} property.
033 */
034public class Sha512CredentialsMatcher extends HashedCredentialsMatcher {
035
036    public Sha512CredentialsMatcher() {
037        super();
038        setHashAlgorithmName(Sha512Hash.ALGORITHM_NAME);
039    }
040}
