169     /**
170      * Normalize a relative URI path that may have relative values ("/./",
171      * "/../", and so on ) it it.  <strong>WARNING</strong> - This method is
172      * useful only for normalizing application-generated paths.  It does not
173      * try to perform security checks for malicious input.
174      * Normalize operations were was happily taken from org.apache.catalina.util.RequestUtil in
175      * Tomcat trunk, r939305
176      *
177      * @param path             Relative path to be normalized
178      * @param replaceBackSlash Should '\\' be replaced with '/'
179      * @return normalized path
180      */
181     private static String normalize(String path, boolean replaceBackSlash) {
182 
183         if (path == null)
184             return null;
185 
186         // Create a place for the normalized path
187         String normalized = path;
188 
189         if (replaceBackSlash && normalized.indexOf('\\') >= 0)
190             normalized = normalized.replace('\\', '/');
191 
192         if (normalized.equals("/."))
193             return "/";
194 
195         // Add a leading "/" if necessary
196         if (!normalized.startsWith("/"))
197             normalized = "/" + normalized;
198 
199         // Resolve occurrences of "//" in the normalized path
200         while (true) {
201             int index = normalized.indexOf("//");
202             if (index < 0)
203                 break;
204             normalized = normalized.substring(0, index) +
205                     normalized.substring(index + 1);
206         }
207 
208         // Resolve occurrences of "/./" in the normalized path
209         while (true) {
210             int index = normalized.indexOf("/./");
211             if (index < 0)
212                 break;
213             normalized = normalized.substring(0, index) +
214                     normalized.substring(index + 2);
215         }
216 
217         // Resolve occurrences of "/../" in the normalized path
218         while (true) {
219             int index = normalized.indexOf("/../");
220             if (index < 0)
221                 break;
222             if (index == 0)
223                 return (null);  // Trying to go outside our context
224             int index2 = normalized.lastIndexOf('/', index - 1);
225             normalized = normalized.substring(0, index2) +
226                     normalized.substring(index + 3);
227         }
228 
229         // Return the normalized path that we have completed
230         return (normalized);
231 
232     }

252     /**
253      * Return the context path for the given request, detecting an include request
254      * URL if called within a RequestDispatcher include.
255      * <p>As the value returned by <code>request.getContextPath()</code> is <i>not</i>
256      * decoded by the servlet container, this method will decode it.
257      *
258      * @param request current HTTP request
259      * @return the context path
260      */
261     public static String getContextPath(HttpServletRequest request) {
262         String contextPath = (String) request.getAttribute(INCLUDE_CONTEXT_PATH_ATTRIBUTE);
263         if (contextPath == null) {
264             contextPath = request.getContextPath();
265         }
266         contextPath = normalize(decodeRequestString(request, contextPath));
267         if ("/".equals(contextPath)) {
268             // the normalize method will return a "/" and includes on Jetty, will also be a "/".
269             contextPath = "";
270         }
271         return contextPath;
272     }

313     /**
314      * Find the Shiro {@link WebEnvironment} for this web application.
315      *
316      * @param sc       ServletContext to find the web application context for
317      * @param attrName the name of the ServletContext attribute to look for
318      * @return the desired WebEnvironment for this web app, or <code>null</code> if none
319      * @since 1.2
320      */
321     public static WebEnvironment getWebEnvironment(ServletContext sc, String attrName) {
322         if (sc == null) {
323             throw new IllegalArgumentException("ServletContext argument must not be null.");
324         }
325         Object attr = sc.getAttribute(attrName);
326         if (attr == null) {
327             return null;
328         }
329         if (attr instanceof RuntimeException) {
330             throw (RuntimeException) attr;
331         }
332         if (attr instanceof Error) {
333             throw (Error) attr;
334         }
335         if (attr instanceof Exception) {
336             throw new IllegalStateException((Exception) attr);
337         }
338         if (!(attr instanceof WebEnvironment)) {
339             throw new IllegalStateException("Context attribute is not of type WebEnvironment: " + attr);
340         }
341         return (WebEnvironment) attr;
342     }

346      * Decode the given source string with a URLDecoder. The encoding will be taken
347      * from the request, falling back to the default "ISO-8859-1".
348      * <p>The default implementation uses <code>URLDecoder.decode(input, enc)</code>.
349      *
350      * @param request current HTTP request

352      * @return the decoded String

357      */
358     @SuppressWarnings({"deprecation"})
359     public static String decodeRequestString(HttpServletRequest request, String source) {
360         String enc = determineEncoding(request);
361         try {
362             return URLDecoder.decode(source, enc);
363         } catch (UnsupportedEncodingException ex) {
364             if (log.isWarnEnabled()) {
365                 log.warn("Could not decode request string [" + Encode.forHtml(source) + "] with encoding '" + Encode.forHtml(enc) +
366                         "': falling back to platform default encoding; exception message: " + ex.getMessage());
367             }
368             return URLDecoder.decode(source);
369         }

587      * <p>Checks to see if a request param is considered true using a loose matching strategy for
588      * general values that indicate that something is true or enabled, etc.</p>
589      * <p/>
590      * <p>Values that are considered "true" include (case-insensitive): true, t, 1, enabled, y, yes, on.</p>
591      *
592      * @param request   the servlet request

596     public static boolean isTrue(ServletRequest request, String paramName) {
597         String value = getCleanParam(request, paramName);
598         return value != null &&
599                 (value.equalsIgnoreCase("true") ||
600                         value.equalsIgnoreCase("t") ||
601                         value.equalsIgnoreCase("1") ||
602                         value.equalsIgnoreCase("enabled") ||
603                         value.equalsIgnoreCase("y") ||
604                         value.equalsIgnoreCase("yes") ||
605                         value.equalsIgnoreCase("on"));
606     }

648     /**
649      * Redirects the to the request url from a previously
650      * {@link #saveRequest(javax.servlet.ServletRequest) saved} request, or if there is no saved request, redirects the
651      * end user to the specified {@code fallbackUrl}.  If there is no saved request or fallback url, this method
652      * throws an {@link IllegalStateException}.
653      * <p/>
654      * This method is primarily used to support a common login scenario - if an unauthenticated user accesses a
655      * page that requires authentication, it is expected that request is
656      * {@link #saveRequest(javax.servlet.ServletRequest) saved} first and then redirected to the login page. Then,
657      * after a successful login, this method can be called to redirect them back to their originally requested URL, a
658      * nice usability feature.
659      *
660      * @param request     the incoming request
661      * @param response    the outgoing response
662      * @param fallbackUrl the fallback url to redirect to if there is no saved request available.
663      * @throws IllegalStateException if there is no saved request and the {@code fallbackUrl} is {@code null}.
664      * @throws IOException           if there is an error redirecting
665      * @since 1.0
666      */
667     public static void redirectToSavedRequest(ServletRequest request, ServletResponse response, String fallbackUrl)
668             throws IOException {
669         String successUrl = null;
670         boolean contextRelative = true;
671         SavedRequest savedRequest = WebUtils.getAndClearSavedRequest(request);
672         if (savedRequest != null && savedRequest.getMethod().equalsIgnoreCase(AccessControlFilter.GET_METHOD)) {
673             successUrl = savedRequest.getRequestUrl();
674             contextRelative = false;
675         }
676 
677         if (successUrl == null) {
678             successUrl = fallbackUrl;
679         }
680 
681         if (successUrl == null) {
682             throw new IllegalStateException("Success URL not available via saved request or via the " +
683                     "successUrlFallback method parameter. One of these must be non-null for " +
684                     "issueSuccessRedirect() to work.");
685         }
686 
687         WebUtils.issueRedirect(request, response, successUrl, null, contextRelative);
688     }
