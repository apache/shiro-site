26  /**
27   * Indicates a method is an event consumer.  The method must have a single argument and the argument's type determines
28   * what type of events should be delivered to the method for consumption.
29   * <p/>
30   * For example:
31   * <pre>
32   * &#64;Subscribe
33   * public void onSomeEvent(SomeEvent event) { ... }
34   * </pre>
35   * <p/>
36   * Because the method argument is declared as a {@code SomeEvent} type, the method will be called by the event
37   * dispatcher whenever a {@code SomeEvent} instance (or one of its subclass instances that is not already registered)
38   * is published.
39   *
40   * @since 1.3
41   */
42  @Retention(value = RetentionPolicy.RUNTIME)
43  @Target(value = ElementType.METHOD)
44  public @interface Subscribe {
45  }
